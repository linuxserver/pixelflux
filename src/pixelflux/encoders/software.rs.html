7//! CPU-based striped encoder: x264 for H.264 and turbojpeg for JPEG.
8//!
9//! Frames are split into horizontal stripes processed in parallel via rayon. Each stripe is
10//! independently hashed against the previous frame for change detection, and only dirty stripes
11//! are encoded. The x264 path maintains per-stripe encoder state across frames for inter-prediction;
12//! the JPEG path is stateless. An optional recording sink receives the H.264 NAL units for
13//! muxing into a file or socket stream.

24/// Upper bound on the horizontal stripes the CPU encoder splits a frame into, so the
25/// persistent per-stripe state vector can be reserved to a fixed capacity once, up front.
26///
27/// With the vector reserved to this size at startup, the per-frame resize to the actual stripe count
28/// stays a cheap in-place adjustment that preserves each stripe's reused encoder and scratch buffers,
29/// rather than a reallocation that would churn them whenever the count changes.

32/// Convert a packed BGRA/RGBA buffer to planar YUV (4:2:0 or 4:4:4) for the software H.264
33/// encoders, spreading the conversion across up to `bands` threads so it never bottlenecks a frame.
34///
35/// **Why the band split exists.** Colour conversion is a non-trivial slice of per-frame CPU. The
36/// striped x264 path already parallelizes it for free — each stripe converts on its own rayon
37/// worker — but a single full-frame consumer (the whole-frame x264 stripe, or the OpenH264
38/// full-frame encoder, which passes `bands = 4`) would otherwise convert its entire image on one
39/// thread and stall the frame there. Splitting into horizontal bands hands that lone conversion the
40/// same multi-threading the striped path enjoys. The cut is horizontal because YUV planes are
41/// row-major, so a horizontal boundary yields contiguous, non-overlapping plane sub-slices with no
42/// per-row seam bookkeeping.
43///
44/// 1. **Plane strides**: the Y plane is `width` wide; the chroma planes are `width` for 4:4:4
45///    (`i444 == true`) or `width / 2` for 4:2:0. `rgba_input` selects the source byte order and
46///    `i444` the subsampling, together choosing one of four `yuv` crate routines — 4:4:4 uses
47///    **Full** range, 4:2:0 uses **Limited** range, and both use the **BT.709** matrix and the
48///    **Fast** conversion mode.
49/// 2. **Band split**: `band_h` is `height / bands` floored to an even number and at least 2 rows
50///    (a band under 2 rows is not worth a thread). Keeping band boundaries even ensures a 4:2:0
51///    chroma pair never straddles a seam. When `bands <= 1` or the whole image fits one band, the
52///    conversion runs single-threaded in place.
53/// 3. **Parallel bands**: otherwise a `std::thread::scope` carves `src` and the three output planes
54///    into contiguous per-band sub-slices (chroma rows scaled by `uv_rows` — full height for 4:4:4,
55///    half for 4:2:0) and spawns one thread per band. The final band absorbs any leftover rows,
56///    taking all remaining rows whenever fewer than `band_h + 2` are left. Each thread's result is
57///    joined and collected; a panicked join degrades to a `PointerOverflow` error, and the first
58///    error wins.
59#[allow(clippy::too_many_arguments)]
60pub(crate) fn convert_to_yuv_mt(
61    src: &[u8],
62    src_stride: u32,
63    width: usize,
64    height: usize,
65    rgba_input: bool,
66    i444: bool,
67    y_buf: &mut [u8],
68    u_buf: &mut [u8],
69    v_buf: &mut [u8],
70    bands: usize,
71) -> Result<(), yuv::YuvError> {
72    let y_stride = width;
73    let uv_stride = if i444 { width } else { width / 2 };
74
75    let convert_band = |src_band: &[u8], y: &mut [u8], u: &mut [u8], v: &mut [u8], h: usize| {
76        let mut img = YuvPlanarImageMut {
77            y_plane: BufferStoreMut::Borrowed(y),
78            y_stride: y_stride as u32,
79            u_plane: BufferStoreMut::Borrowed(u),
80            u_stride: uv_stride as u32,
81            v_plane: BufferStoreMut::Borrowed(v),
82            v_stride: uv_stride as u32,
83            width: width as u32,
84            height: h as u32,
85        };
86        match (i444, rgba_input) {
87            (true, true) => yuv::rgba_to_yuv444(
88                &mut img, src_band, src_stride, YuvRange::Full,
89                YuvStandardMatrix::Bt709, YuvConversionMode::Fast,
90            ),
91            (true, false) => yuv::bgra_to_yuv444(
92                &mut img, src_band, src_stride, YuvRange::Full,
93                YuvStandardMatrix::Bt709, YuvConversionMode::Fast,
94            ),
95            (false, true) => yuv::rgba_to_yuv420(
96                &mut img, src_band, src_stride, YuvRange::Limited,
97                YuvStandardMatrix::Bt709, YuvConversionMode::Fast,
98            ),
99            (false, false) => yuv::bgra_to_yuv420(
100                &mut img, src_band, src_stride, YuvRange::Limited,
101                YuvStandardMatrix::Bt709, YuvConversionMode::Fast,
102            ),
103        }
104    };
105
106    let band_h = ((height / bands.max(1)) & !1).max(2);
107    if bands <= 1 || height <= band_h {
108        return convert_band(src, y_buf, u_buf, v_buf, height);
109    }
110
111    let uv_rows = |rows: usize| if i444 { rows } else { rows / 2 };
112    let mut results: Vec<Result<(), yuv::YuvError>> = Vec::new();
113    std::thread::scope(|s| {
114        let mut handles = Vec::new();
115        let (mut src_rest, mut y_rest, mut u_rest, mut v_rest) = (src, y_buf, u_buf, v_buf);
116        let mut row = 0;
117        while row < height {
118            let h = if height - row < band_h + 2 { height - row } else { band_h };
119            let (src_band, s_next) = src_rest.split_at(h * src_stride as usize);
120            let (y_band, y_next) = y_rest.split_at_mut(h * y_stride);
121            let (u_band, u_next) = u_rest.split_at_mut(uv_rows(h) * uv_stride);
122            let (v_band, v_next) = v_rest.split_at_mut(uv_rows(h) * uv_stride);
123            src_rest = s_next;
124            y_rest = y_next;
125            u_rest = u_next;
126            v_rest = v_next;
127            row += h;
128            handles.push(s.spawn(move || convert_band(src_band, y_band, u_band, v_band, h)));
129        }
130        for hnd in handles {
131            results.push(hnd.join().unwrap_or(Err(yuv::YuvError::PointerOverflow)));
132        }
133    });
134    results.into_iter().collect()
135}

149/// Process-global lock that serializes libx264 encoder open/close, because those calls are
150/// not thread-safe yet the striped path opens encoders concurrently from many stripe workers.
151///
152/// libx264 mutates process-global state inside `x264_encoder_open`/`x264_encoder_close`, so two
153/// stripe encoders opening at once — or two capture instances sharing one process — can race that
154/// state and corrupt the heap. The lock is deliberately held only around open and close, never
155/// around `x264_encoder_encode`, so serializing setup costs nothing in the hot per-stripe encode
156/// path where the real parallelism lives.
157static X264_OPEN_CLOSE_LOCK: std::sync::Mutex<()> = std::sync::Mutex::new(());
158
159/// One long-lived libx264 session for a stripe, holding the raw `x264_t` handle alongside a
160/// mirror of its live parameters so the encoder can be retuned per frame instead of rebuilt.
161///
162/// Rebuilding an x264 encoder is expensive and forces a fresh IDR, so a stripe keeps its instance
163/// across frames and only nudges CRF, bitrate, VBV, and frame rate live; the tracked `current_*`
164/// fields are that mirror, letting a reconfigure skip the FFI call whenever nothing actually changed.
165/// `is_i444` (4:4:4 vs 4:2:0) is baked into the encoder's colour space at open, so a change to it is
166/// one of the few things that forces a full rebuild; `is_cbr` records which rate-control mode was
167/// chosen at open and gates which of the live reconfigures apply. The manual `Send` impl exists only
168/// because a raw pointer is not `Send` by default and the handle must move onto the rayon stripe
169/// workers; `Drop` closes it under the global open/close lock for the same reason that lock exists.
170pub struct H264EncoderWrapper {
171    encoder: *mut x264_sys::x264_t,
172    pub width: i32,
173    pub height: i32,
174    current_crf: i32,
175    pub is_i444: bool,
176    is_cbr: bool,
177    current_bitrate: i32,
178    current_vbv: i32,
179    current_fps: u32,
180    #[allow(dead_code)]
181    full_range: bool,
182}
183
184unsafe impl Send for H264EncoderWrapper {}
185
186impl Drop for H264EncoderWrapper {
187    fn drop(&mut self) {
188        if !self.encoder.is_null() {
189            let _guard = X264_OPEN_CLOSE_LOCK.lock().unwrap_or_else(|e| e.into_inner());
190            unsafe { x264_sys::x264_encoder_close(self.encoder) };
191            self.encoder = ptr::null_mut();
192        }
193    }
194}

196impl H264EncoderWrapper {
197    /// Open an x264 encoder tuned for real-time screen streaming, or `None` on failure.
198    ///
199    /// **Why this configuration.** These frames are captured live and must ship immediately, so the
200    /// encoder is optimized for latency over compression ratio: the `ultrafast` preset keeps encode
201    /// time under the frame budget, and `zerolatency` bars the frame reordering and lookahead
202    /// buffering that would otherwise add pipeline delay. Everything below then bends x264 toward the
203    /// pipeline's own keyframe and colour model instead of its broadcast-oriented defaults.
204    ///
205    /// 1. **Preset/tune**: starts from the `ultrafast` preset with the `zerolatency` tune, then
206    ///    overrides resolution, frame rate (floored to 30 fps when under 1), and thread count.
207    /// 2. **Infinite GOP**: `i_keyint_max` is set to x264's infinite sentinel and adaptive scene-cut
208    ///    is disabled (`i_scenecut_threshold = 0`), so the encoder never injects an unrequested IDR
209    ///    on a scene change — keyframes are purely on-demand via the forced-IDR path, matching the
210    ///    strict infinite-GOP model.
211    /// 3. **Rate control**:
212    ///    - **CBR** (`cbr_mode`): ABR targeting `bitrate_kbps` with a VBV cap pinned to the same
213    ///      value (buffer `vbv_kbit`, precomputed by the caller from the frame-time multiplier
214    ///      policy) and filler disabled. Optional QP clamps apply only when non-zero — `max_qp` is
215    ///      the legibility floor (caps how ugly a rate-starved frame gets) and `min_qp` the waste
216    ///      ceiling (stops over-spending on easy content); both are clamped to 51.
217    ///    - **CRF** (default): constant-quality with `f_rf_constant = crf`.
218    /// 4. **Colour**: I444 (full range) or I420 (limited range) CSP, BT.709 VUI primaries/transfer/
219    ///    matrix, and the matching `high444` / `baseline` profile.
220    /// 5. **Coding tools**: CABAC and the 8x8 transform are disabled, matching the low-latency
221    ///    baseline profile — CAVLC entropy coding with no 8x8 DCT — for minimal encode cost.
222    /// 6. **Output**: repeated headers (SPS/PPS before each keyframe) and Annex-B framing, with
223    ///    x264's own logging silenced.
224    ///
225    /// The `x264_encoder_open` call is serialized under `X264_OPEN_CLOSE_LOCK` because it mutates
226    /// libx264 global state.
227    #[allow(clippy::too_many_arguments)]
228    pub fn new(width: i32, height: i32, crf: i32, is_i444: bool, fps: f64, threads: i32,
229               cbr_mode: bool, bitrate_kbps: i32, vbv_kbit: i32,
230               min_qp: i32, max_qp: i32) -> Option<Self> {
231        unsafe {
232            let mut param: x264_sys::x264_param_t = std::mem::zeroed();
233            let preset = CString::new("ultrafast").unwrap();
234            let tune = CString::new("zerolatency").unwrap();
235
236            if x264_sys::x264_param_default_preset(&mut param, preset.as_ptr(), tune.as_ptr()) < 0 {
237                return None;
238            }
239
240            param.i_width = width;
241            param.i_height = height;
242            param.i_fps_num = if fps < 1.0 { 30 } else { fps as u32 };
243            param.i_fps_den = 1;
244            param.i_keyint_max = x264_sys::X264_KEYINT_MAX_INFINITE as i32;
245            param.i_scenecut_threshold = 0;
246            if cbr_mode {
247                let bk = bitrate_kbps.saturating_abs();
248                param.rc.i_rc_method = x264_sys::X264_RC_ABR as i32;
249                param.rc.i_bitrate = bk;
250                param.rc.i_vbv_max_bitrate = bk;
251                param.rc.i_vbv_buffer_size = vbv_kbit.max(1);
252                param.rc.b_filler = 0;
253                if min_qp > 0 {
254                    param.rc.i_qp_min = min_qp.min(51);
255                }
256                if max_qp > 0 {
257                    param.rc.i_qp_max = max_qp.min(51);
258                }
259            } else {
260                param.rc.i_rc_method = x264_sys::X264_RC_CRF as i32;
261                param.rc.f_rf_constant = crf as f32;
262            }
263            param.i_csp = if is_i444 {
264                x264_sys::X264_CSP_I444
265            } else {
266                x264_sys::X264_CSP_I420
267            } as i32;
268            param.vui.b_fullrange = if is_i444 { 1 } else { 0 };
269            param.vui.i_colorprim = 1;
270            param.vui.i_transfer = 1;
271            param.vui.i_colmatrix = 1;
272
273            let profile = CString::new(if is_i444 { "high444" } else { "baseline" }).unwrap();
274            x264_sys::x264_param_apply_profile(&mut param, profile.as_ptr());
275
276            param.i_threads = threads;
277            param.b_repeat_headers = 1;
278            param.b_annexb = 1;
279            param.i_log_level = x264_sys::X264_LOG_NONE;
280
281            let encoder = {
282                let _guard = X264_OPEN_CLOSE_LOCK.lock().unwrap_or_else(|e| e.into_inner());
283                x264_sys::x264_encoder_open(&mut param)
284            };
285            if encoder.is_null() {
286                None
287            } else {
288                Some(Self {
289                    encoder,
290                    width,
291                    height,
292                    current_crf: crf,
293                    is_i444,
294                    is_cbr: cbr_mode,
295                    current_bitrate: bitrate_kbps.saturating_abs(),
296                    current_vbv: vbv_kbit,
297                    current_fps: if fps < 1.0 { 30 } else { fps as u32 },
298                    full_range: param.vui.b_fullrange == 1,
299                })
300            }
301        }
302    }

304    /// Retune the constant-quality CRF on the running encoder, so a quality change costs a
305    /// parameter push rather than tearing down and rebuilding the session (a rebuild would force an
306    /// IDR and drop encoder state).
307    ///
308    /// It is a no-op in CBR mode, where rate is bitrate-controlled and CRF simply does not apply, and
309    /// a no-op when the value is unchanged — the tracked `current_crf` is what makes that cheap
310    /// early-out possible. Otherwise it reads the encoder's live parameters, overwrites
311    /// `f_rf_constant`, and pushes the change via `x264_encoder_reconfig`, advancing the tracked CRF
312    /// only once the reconfig has actually succeeded so the mirror never drifts from the encoder.
313    pub fn reconfigure_crf(&mut self, new_crf: i32) {
314        if self.is_cbr || self.current_crf == new_crf {
315            return;
316        }
317        unsafe {
318            let mut param: x264_sys::x264_param_t = std::mem::zeroed();
319            x264_sys::x264_encoder_parameters(self.encoder, &mut param);
320            param.rc.f_rf_constant = new_crf as f32;
321            if x264_sys::x264_encoder_reconfig(self.encoder, &mut param) == 0 {
322                self.current_crf = new_crf;
323            }
324        }
325    }

327    /// Retune bitrate/VBV (CBR only) and/or frame rate on the running encoder, structured to
328    /// be called unconditionally every frame so the caller need not track what changed itself.
329    ///
330    /// Because `encode_cpu` fires it on every frame, the first thing it does is compute the would-be
331    /// values and bail before touching the encoder when neither the CBR bitrate/VBV nor the frame
332    /// rate actually differs from what is live — that self-gating is what keeps a per-frame call
333    /// nearly free. Bitrate and VBV apply only in CBR mode (they are meaningless under CRF); the
334    /// frame rate applies in either. Changed fields are written into the live parameter set and
335    /// pushed via `x264_encoder_reconfig`, and the tracked mirror advances only on a successful
336    /// reconfig so it cannot drift from the encoder's real state.
337    pub fn reconfigure_rate(&mut self, bitrate_kbps: i32, vbv_kbit: i32, fps: f64) {
338        let bk = bitrate_kbps.saturating_abs();
339        let new_fps = if fps < 1.0 { 30 } else { fps as u32 };
340        let rate_changed =
341            self.is_cbr && (self.current_bitrate != bk || self.current_vbv != vbv_kbit);
342        let fps_changed = self.current_fps != new_fps;
343        if !rate_changed && !fps_changed {
344            return;
345        }
346        unsafe {
347            let mut param: x264_sys::x264_param_t = std::mem::zeroed();
348            x264_sys::x264_encoder_parameters(self.encoder, &mut param);
349            if rate_changed {
350                param.rc.i_bitrate = bk;
351                param.rc.i_vbv_max_bitrate = bk;
352                param.rc.i_vbv_buffer_size = vbv_kbit.max(1);
353            }
354            if fps_changed {
355                param.i_fps_num = new_fps;
356                param.i_fps_den = 1;
357            }
358            if x264_sys::x264_encoder_reconfig(self.encoder, &mut param) == 0 {
359                if rate_changed {
360                    self.current_bitrate = bk;
361                    self.current_vbv = vbv_kbit;
362                }
363                if fps_changed {
364                    self.current_fps = new_fps;
365                }
366            }
367        }
368    }

370    /// Encode one YUV frame into H.264 and frame it for the wire, reporting whether the
371    /// encoder actually emitted a bitstream this call.
372    ///
373    /// The boolean return is load-bearing: `x264_encoder_encode` can legitimately produce nothing on
374    /// a given call, and the caller must forward a stripe only when real bytes exist — never an empty
375    /// or header-only packet. The output also serves two consumers at once, which is why framing is
376    /// conditional: the transport needs the pipeline's small wire header to route the stripe, while
377    /// the optional recording sink needs the *bare* Annex-B elementary stream with no wire header so
378    /// it can be muxed directly.
379    ///
380    /// 1. **Picture setup**: wraps the borrowed Y/U/V planes and their strides in an
381    ///    `x264_picture_t` with the encoder's CSP, stamps the presentation timestamp with `frame_id`,
382    ///    and requests an IDR when `force_idr` is set (otherwise `X264_TYPE_AUTO`).
383    /// 2. **Encode**: calls `x264_encoder_encode`; a non-positive returned size means no frame was
384    ///    emitted this call, so the function returns `false` without writing output.
385    /// 3. **Framing**: `output_buf` is cleared and refilled. Unless `omit_headers` is set, a header
386    ///    is prepended — a `0x04` codec tag, then a type byte read from the *actual* output picture
387    ///    type rather than from `force_idr`, because the encoder may not honor a keyframe request and
388    ///    the client keys its decode-recovery on the frame type it truly received (IDR = `0x01`,
389    ///    I = `0x02`, else `0x00`), then the caller's `fixed_header` (frame number, y-start, width,
390    ///    height). With `omit_headers` the output is bare Annex-B.
391    /// 4. **Payload + recording**: every NAL payload is appended to `output_buf`, and each is also
392    ///    forwarded to the recording sink when one is attached — giving the sink raw Annex-B without
393    ///    the wire header.
394    #[allow(clippy::too_many_arguments)]
395    pub fn encode_with_headers(
396        &mut self,
397        y: &[u8],
398        u: &[u8],
399        v: &[u8],
400        y_stride: i32,
401        u_stride: i32,
402        v_stride: i32,
403        frame_id: i64,
404        force_idr: bool,
405        fixed_header: &[u8],
406        omit_headers: bool,
407        output_buf: &mut Vec<u8>,
408        recording_sink: Option<&Arc<RecordingSink>>,
409    ) -> bool {
410        unsafe {
411            let mut pic_in: x264_sys::x264_picture_t = std::mem::zeroed();
412            x264_sys::x264_picture_init(&mut pic_in);
413
414            pic_in.img.i_csp = if self.is_i444 {
415                x264_sys::X264_CSP_I444
416            } else {
417                x264_sys::X264_CSP_I420
418            } as i32;
419            pic_in.img.i_plane = 3;
420            pic_in.img.plane[0] = y.as_ptr() as *mut u8;
421            pic_in.img.plane[1] = u.as_ptr() as *mut u8;
422            pic_in.img.plane[2] = v.as_ptr() as *mut u8;
423            pic_in.img.i_stride[0] = y_stride;
424            pic_in.img.i_stride[1] = u_stride;
425            pic_in.img.i_stride[2] = v_stride;
426            pic_in.i_pts = frame_id;
427            pic_in.i_type = if force_idr {
428                x264_sys::X264_TYPE_IDR
429            } else {
430                x264_sys::X264_TYPE_AUTO
431            } as i32;
432
433            let mut pic_out: x264_sys::x264_picture_t = std::mem::zeroed();
434            let mut nals: *mut x264_sys::x264_nal_t = ptr::null_mut();
435            let mut i_nals: i32 = 0;
436
437            let frame_size = x264_sys::x264_encoder_encode(
438                self.encoder,
439                &mut nals,
440                &mut i_nals,
441                &mut pic_in,
442                &mut pic_out,
443            );
444
445            if frame_size > 0 {
446                let header_len = if omit_headers { 0 } else { 2 + fixed_header.len() };
447                let total_len = header_len + frame_size as usize;
448
449                output_buf.clear();
450                output_buf.reserve(total_len);
451
452                if !omit_headers {
453                    output_buf.push(0x04);
454                    let type_byte = if pic_out.i_type == x264_sys::X264_TYPE_IDR as i32 {
455                        0x01
456                    } else if pic_out.i_type == x264_sys::X264_TYPE_I as i32 {
457                        0x02
458                    } else {
459                        0x00
460                    };
461                    output_buf.push(type_byte);
462                    output_buf.extend_from_slice(fixed_header);
463                }
464
465                let nal_slice = std::slice::from_raw_parts(nals, i_nals as usize);
466                for nal in nal_slice {
467                    let payload = std::slice::from_raw_parts(nal.p_payload, nal.i_payload as usize);
468                    output_buf.extend_from_slice(payload);
469                    if let Some(sink) = recording_sink {
470                        sink.write_frame(payload);
471                    }
472                }
473                return true;
474            }
475        }
476        false
477    }

480/// Everything one horizontal stripe must remember between frames: its reused buffers, its own
481/// live encoder, and the motion / paint-over / damage bookkeeping that drives its send decision.
482///
483/// The frame is striped so independent screen regions can encode in parallel and an unchanged region
484/// can be skipped on its own, and that only works if each stripe carries its *own* cross-frame
485/// history. So one instance lives per stripe for the whole session and nothing per-stripe is rebuilt
486/// or recomputed from scratch each frame:
487/// - **Reused buffers**: `y_buf` / `u_buf` / `v_buf` hold the stripe's YUV planes and `packet_buf`
488///   the encoded output, grown in place rather than reallocated per frame.
489/// - **Encoder**: `h264_encoder` is the stripe's x264 instance, reused until its geometry or chroma
490///   format changes.
491/// - **Paint-over / recovery**: `no_motion_frame_count` counts consecutive static frames,
492///   `paint_over_sent` guards against re-sending a high-quality repaint of a still region, and
493///   `h264_burst_frames_remaining` tracks a post-repaint or recovery streaming burst.
494/// - **Content-hash damage** (only for sources without external damage, i.e. X11): `last_hash` is
495///   the previous frame's content hash, `consecutive_changes` counts changed frames toward the
496///   damage-block threshold, and `in_damage_block` / `damage_block_frames_remaining` /
497///   `hash_at_block_start` drive the sustained-motion damage block managed by `content_dirty`.
498#[derive(Default)]
499pub struct StripeState {
500    pub no_motion_frame_count: u32,
501    pub paint_over_sent: bool,
502    pub h264_encoder: Option<H264EncoderWrapper>,
503    pub h264_burst_frames_remaining: i32,
504    pub y_buf: Vec<u8>,
505    pub u_buf: Vec<u8>,
506    pub v_buf: Vec<u8>,
507    pub packet_buf: Vec<u8>,
508    pub last_hash: u64,
509    pub consecutive_changes: u32,
510    pub in_damage_block: bool,
511    pub damage_block_frames_remaining: i32,
512    pub hash_at_block_start: u64,
513}

515/// Fast, non-cryptographic 64-bit content hash used only for in-memory change detection.
516///
517/// Uses xxh3: a SIMD-friendly hash that processes 64-byte blocks with parallel lanes,
518/// delivering near memory-bandwidth throughput. The value is never persisted or sent on the
519/// wire, so only the property that identical bytes hash identically matters. A collision
520/// between two distinct stripes is ~2^-64, and the next real content change or a requested
521/// keyframe repaints any missed update anyway.
522fn fast_hash(bytes: &[u8]) -> u64 {
523    xxhash_rust::xxh3::xxh3_64_with_seed(bytes, 0)
524}
525
526impl StripeState {
527    /// Stand in for the compositor damage that X11 capture does not provide: hash this stripe
528    /// to decide whether it changed since last frame, and once it is clearly in motion, stop
529    /// re-hashing it every frame by committing to a sustained-motion "damage block".
530    ///
531    /// The hash is not free, and a region that changes every frame would otherwise be re-hashed
532    /// forever while always reporting dirty anyway. So after `threshold` consecutive changes the
533    /// stripe enters a damage block that just reports dirty for `duration` frames and re-hashes only
534    /// once, at the end, to decide whether to extend the block or let it lapse — trading a little
535    /// extra sending for far fewer hashes on exactly the regions that need them least:
536    ///
537    /// 1. **Inside a damage block**: the stripe is treated as dirty without re-hashing, and the
538    ///    block's remaining-frame counter is decremented. Only when the counter reaches zero is the
539    ///    stripe re-hashed — if it differs from the hash captured at block start the block is renewed
540    ///    for another `duration` frames, otherwise the block exits and the change counter resets.
541    ///    This keeps a continuously-moving region streaming for `duration` frames per re-check rather
542    ///    than hashing every frame.
543    /// 2. **Outside a block**: the stripe is hashed and compared to the previous frame. A change
544    ///    increments `consecutive_changes`, and reaching `threshold` consecutive changes opens a new
545    ///    damage block; an unchanged frame resets the counter to zero.
546    ///
547    /// Returns `true` whenever the stripe is considered dirty (always true while inside a block).
548    pub fn content_dirty(&mut self, bytes: &[u8], threshold: u32, duration: i32) -> bool {
549        if self.in_damage_block {
550            self.damage_block_frames_remaining -= 1;
551            if self.damage_block_frames_remaining <= 0 {
552                let h = fast_hash(bytes);
553                if h != self.hash_at_block_start {
554                    self.damage_block_frames_remaining = duration;
555                    self.hash_at_block_start = h;
556                } else {
557                    self.in_damage_block = false;
558                    self.consecutive_changes = 0;
559                }
560                self.last_hash = h;
561            }
562            return true;
563        }
564        let h = fast_hash(bytes);
565        let changed = h != self.last_hash;
566        self.last_hash = h;
567        if changed {
568            self.consecutive_changes += 1;
569            if self.consecutive_changes >= threshold {
570                self.in_damage_block = true;
571                self.damage_block_frames_remaining = duration;
572                self.hash_at_block_start = h;
573            }
574        } else {
575            self.consecutive_changes = 0;
576        }
577        changed
578    }
579}

602/// The software encoder's per-frame entry point: split the frame into horizontal stripes,
603/// decide per stripe whether it needs sending, and encode only those as JPEG or x264 H.264
604/// across the rayon pool.
605///
606/// Two pressures drive the design: CPU H.264/JPEG is expensive, so the frame is cut into
607/// parallel stripes; bandwidth is precious, so unchanged stripes are skipped. Each stripe is
608/// independently hashed against the previous frame for change detection, and only dirty stripes
609/// are encoded. The x264 path maintains per-stripe encoder state across frames for
610/// inter-prediction; the JPEG path is stateless. The recording sink is attached only in
611/// single-stripe mode because several independent sub-frame bitstreams cannot be muxed.
612///
613/// # Arguments
614///
615/// * `stripes` - Persistent per-stripe state vector (resized as needed, encoder state preserved).
616/// * `raw_pixels` - Packed BGRA/RGBA pixel buffer (`width * height * 4` bytes).
617/// * `width` - Frame width in pixels.
618/// * `height` - Frame height in pixels.
619/// * `damage_rects` - Wayland damage rectangles (empty for X11 hash-based detection).
620/// * `settings` - Capture settings (quality, mode, rate control, etc.).
621/// * `frame_counter` - Current frame number (wrapping `u16`).
622/// * `use_gpu` - `true` when the source is RGBA (GLES readback); `false` for BGRA (X11 host).
623/// * `hash_damage` - `true` for X11 stripe-hash change detection; `false` when damage rects
624///   are provided.
625/// * `recording_sink` - Optional Unix-socket H.264 fan-out (only active in single-stripe mode).
626/// * `force_idr_all` - Force a keyframe on every stripe (client join / reset / periodic IDR).
627///
628/// # Returns
629///
630/// Vec of [`EncodedStripe`] — empty when nothing changed.
631/// repainting a stalled region at full quality, and letting a freshly-joined or reset client recover
632/// a clean picture. Persistent `StripeState` is what makes both affordable: encoders and buffers
633/// survive across frames instead of being rebuilt, and the motion/paint-over history the decision
634/// needs lives right beside them. The per-stripe decision mirrors `decide_hw_fullframe`'s policy for
635/// the hardware full-frame encoders; it is kept as separate code here because the striped path also
636/// chooses JPEG-vs-H.264 and derives its own damage.
637///
638/// 1. **Stripe count**: defaults to the core count so the fan-out matches the hardware, but
639///    collapses to a single full-frame stripe when H.264 full-frame is requested or the frame is
640///    shorter than the 64-row minimum, and is otherwise capped so no stripe is thinner than 64 rows —
641///    below that the per-stripe encoder and thread overhead outweighs the parallelism and the tiny
642///    H.264 slices compress poorly. The persistent `stripes` vector is resized to match, preserving
643///    per-stripe state across frames.
644/// 2. **Idle fast path**: a frame on which no stripe can emit anything (no damage / clean
645///    hashes, no paint-over due, no burst, no recovery IDR, not streaming) only advances the
646///    per-stripe no-motion bookkeeping inline and returns without dispatching the stripe
647///    fan-out, so a static capture never wakes the rayon pool.
648/// 3. **Dirty map**: with external compositor damage (`hash_damage == false`) each `damage_rects`
649///    rectangle marks every stripe whose row range it overlaps. With `hash_damage == true` (X11,
650///    which has no compositor damage) per-stripe content hashing drives dirtiness instead — except
651///    in streaming H.264, where every stripe is sent unconditionally so the hash is skipped.
652/// 4. **Per-stripe decision** (in `stripe_body`): a stripe is sent when it is dirty, when a
653///    paint-over / recovery burst is in flight, when streaming mode is on, or when `force_idr_all`
654///    is set. Quality is chosen per case — base JPEG quality / base CRF for live content, the
655///    paint-over quality/CRF after `paint_over_trigger_frames` static frames (once per still region,
656///    guarded by `paint_over_sent`), and `burst_crf` during a burst (the paint-over CRF when it is
657///    enabled and actually lower, else the base CRF, since a recovery burst still needs to stream so
658///    CBR can refine it). A newly dirty frame cancels any pending burst or paint-over and reverts to
659///    base quality.
660/// 5. **Recovery IDR** (`force_idr_all`): forces a send on every stripe even when static so a
661///    reconnecting client can resume. For H.264 it forces an IDR and arms a short streaming burst
662///    (unless one is already pending, so it cannot preempt an in-flight burst) because the keyframe
663///    is base-quality — worsened further by CBR — and a damage-gated static stream would otherwise
664///    never refine it; for JPEG, where every stripe is already intra, it resends a
665///    previously-painted-over stripe at the paint-over quality already on screen so a joining viewer
666///    does not see a downgrade.
667/// 6. **Encoding**:
668///    - **JPEG** (`output_mode 0`): source byte order is RGBA on the GPU readback path and BGRA on
669///      X11; each worker thread reuses its thread-local TurboJPEG compressor. Header-less output
670///      hands the compressed buffer straight through; otherwise a 6-byte stripe header (`0x03` tag,
671///      a reserved byte, frame number, y-start) is prepended to match the H.264 path's native
672///      framing so the transport can forward the buffer without re-framing.
673///    - **H.264** (`output_mode 1`): the stripe's x264 encoder is reused unless the width, height,
674///      or chroma format changed, in which case it is rebuilt and an IDR forced; otherwise CRF and
675///      rate are reconfigured live. ARGB is converted to YUV (a conversion failure skips the stripe
676///      rather than encoding garbage), and an 8-byte fixed header (frame number, y-start, width,
677///      height) is emitted. The live CBR VBV budget is recomputed here from the bitrate/fps so it
678///      rescales with live changes.
679/// 7. **Dispatch**: a single full-frame stripe runs inline (sequential — empirically faster than a
680///    one-element rayon job) with a single-band colour conversion and the same thread policy as the
681///    OpenH264 encoder — one fewer than the available cores, clamped to `[1, 4]`. The slice threads
682///    keep the in-frame encode latency inside the frame budget at high resolutions; the cap is four
683///    because `zerolatency` makes x264 slice-threaded and more than four slices trips decode
684///    glitches in some Chromium builds, and the minus-one leaves headroom for the capture thread. Multiple stripes instead run across the
685///    rayon pool with a single x264 thread and one conversion band each, since the parallelism there
686///    already comes from encoding the stripes concurrently. The recording sink is attached only in single-stripe mode, because the several
687///    independent sub-frame bitstreams of striped mode cannot be muxed into one recording.
688#[allow(clippy::too_many_arguments)]
689pub fn encode_cpu(
690    stripes: &mut Vec<StripeState>,
691    raw_pixels: &[u8],
692    width: i32,
693    height: i32,
694    damage_rects: &[Rectangle<i32, Physical>],
695    settings: &RustCaptureSettings,
696    frame_counter: u16,
697    use_gpu: bool,
698    hash_damage: bool,
699    recording_sink: Option<&Arc<RecordingSink>>,
700    force_idr_all: bool,
701) -> Vec<EncodedStripe> {
702    let num_cores = std::thread::available_parallelism()
703        .map(|n| n.get())
704        .unwrap_or(1);
705    let min_stripe_height = 64;
706    let mut n_processing_stripes = num_cores;
707
708    if (settings.output_mode == 1 && settings.video_fullframe) || height < min_stripe_height {
709        n_processing_stripes = 1;
710    } else {
711        let max_stripes_by_height = (height as usize) / (min_stripe_height as usize);
712        n_processing_stripes = n_processing_stripes.min(max_stripes_by_height).max(1);
713    }
714
715    if stripes.len() != n_processing_stripes {
716        stripes.resize_with(n_processing_stripes, StripeState::default);
717    }
718
719    let stripe_geometries =
720        compute_stripe_geometries(height as usize, n_processing_stripes, settings.output_mode);
721
722    // Idle fast path: a static frame must still advance every stripe's paint-over countdown,
723    // but nothing else — so when no stripe can emit anything this frame, do that bookkeeping
724    // inline and return before the rayon fan-out. Waking the whole worker pool 60x/s for
725    // no-op stripes is the dominant idle cost (tens of percent of a core), dwarfing the real
726    // per-frame work. "Static" is known up front for damage-authoritative sources (Wayland:
727    // empty damage list); hash-damage sources (X11) instead take a sequential early-exit
728    // hash scan, probing the most-recently-dirty stripe first so live content bails out
729    // after a single stripe hash. A clean scan performs exactly the state transitions
730    // `content_dirty` would (hash unchanged, change streak reset), so the damage-block
731    // machinery observes no difference.
732    let idle_candidate = damage_rects.is_empty()
733        && !force_idr_all
734        && !(settings.output_mode == 1 && settings.video_streaming_mode);
735    if idle_candidate {
736        let paint_over_armed = settings.use_paint_over_quality
737            && if settings.output_mode == 0 {
738                settings.paint_over_jpeg_quality > settings.jpeg_quality
739            } else {
740                settings.video_paintover_crf < settings.video_crf
741            };
742        let no_pending_send = |st: &StripeState| {
743            (settings.output_mode == 0 || st.h264_burst_frames_remaining <= 0)
744                && (!paint_over_armed
745                    || st.paint_over_sent
746                    || st.no_motion_frame_count.saturating_add(1)
747                        < settings.paint_over_trigger_frames)
748        };
749        let quiescent = if !hash_damage {
750            stripes.iter().all(no_pending_send)
751        } else {
752            let width_bytes = width as usize * 4;
753            let hint = stripes
754                .iter()
755                .enumerate()
756                .min_by_key(|(_, st)| st.no_motion_frame_count)
757                .map(|(i, _)| i)
758                .unwrap_or(0);
759            let clean = |i: usize| {
760                let st = &stripes[i];
761                if !no_pending_send(st) || st.in_damage_block {
762                    return false;
763                }
764                let (y, h) = stripe_geometries[i];
765                let bytes = &raw_pixels[y * width_bytes..(y + h) * width_bytes];
766                fast_hash(bytes) == st.last_hash
767            };
768            clean(hint) && (0..stripes.len()).filter(|&i| i != hint).all(clean)
769        };
770        if quiescent {
771            for st in stripes.iter_mut() {
772                st.no_motion_frame_count = st.no_motion_frame_count.saturating_add(1);
773                st.consecutive_changes = 0;
774            }
775            return Vec::new();
776        }
777    }
778    let mut stripe_is_dirty = vec![false; n_processing_stripes];
779    if !damage_rects.is_empty() {
780        for rect in damage_rects {
781            let r_y_start = rect.loc.y.max(0) as usize;
782            let r_y_end = (rect.loc.y + rect.size.h).min(height) as usize;
783            if r_y_start < r_y_end {
784                for (i, &(s_y, s_h)) in stripe_geometries.iter().enumerate() {
785                    let s_end = s_y + s_h;
786                    if r_y_start < s_end && r_y_end > s_y {
787                        stripe_is_dirty[i] = true;
788                    }
789                }
790            }
791        }
792    }
793
794    let width_usize = width as usize;
795    let output_mode = settings.output_mode;
796    let video_crf = settings.video_crf;
797    let video_po_crf = settings.video_paintover_crf;
798    let video_burst = settings.video_paintover_burst_frames;
799    let video_fullcolor = settings.video_fullcolor;
800    let video_streaming = settings.video_streaming_mode;
801    let jpeg_q = settings.jpeg_quality;
802    let paint_q = settings.paint_over_jpeg_quality;
803    let trigger_frames = settings.paint_over_trigger_frames;
804    let use_paint_over = settings.use_paint_over_quality;
805    let burst_crf = if use_paint_over && video_po_crf < video_crf { video_po_crf } else { video_crf };
806    let target_fps = settings.target_fps;
807    let omit_headers = settings.omit_stripe_headers;
808    let damage_block_threshold = settings.damage_block_threshold;
809    let damage_block_duration = settings.damage_block_duration as i32;
810    let video_cbr = settings.video_cbr_mode;
811    let video_bitrate = settings.video_bitrate_kbps;
812    let video_vbv = (crate::encoders::vbv_bits(
813        (video_bitrate.max(0) as u32).saturating_mul(1000),
814        target_fps,
815        settings.keyframe_interval_s,
816        settings.video_vbv_multiplier,
817    ) / 1000)
818        .max(1) as i32;
819    // Full-frame x264 threads follow the same policy as the OpenH264 encoder:
820    // one fewer than the cores (headroom for the capture thread), clamped to
821    // [1, 4] to match the four-slice ceiling below.
822    let h264_threads = if n_processing_stripes == 1 {
823        num_cores.saturating_sub(1).clamp(1, 4) as i32
824    } else {
825        1
826    };
827    let csc_bands = 1;
828    let stripe_sink: Option<Arc<RecordingSink>> = if n_processing_stripes == 1 {
829        recording_sink.cloned()
830    } else {
831        None
832    };
833
834    let stripe_body = |(i, stripe_state): (usize, &mut StripeState)| -> Option<EncodedStripe> {
835            if i >= stripe_geometries.len() {
836                return None;
837            }
838            let (y_start, actual_height) = stripe_geometries[i];
839            let start_idx = y_start * width_usize * 4;
840            let end_idx = start_idx + (actual_height * width_usize * 4);
841            let stripe_bytes = &raw_pixels[start_idx..end_idx];
842
843            let mut send_this_stripe = false;
844            let mut quality_or_crf = if output_mode == 0 { jpeg_q } else { video_crf };
845            let mut force_idr = false;
846            let is_dirty = if !hash_damage {
847                stripe_is_dirty[i]
848            } else if output_mode == 1 && video_streaming {
849                false
850            } else {
851                stripe_state.content_dirty(stripe_bytes, damage_block_threshold, damage_block_duration)
852            };
853
854            if output_mode == 1 && stripe_state.h264_burst_frames_remaining > 0 {
855                send_this_stripe = true;
856                quality_or_crf = burst_crf;
857                stripe_state.h264_burst_frames_remaining -= 1;
858
859                if is_dirty {
860                    stripe_state.h264_burst_frames_remaining = 0;
861                    stripe_state.paint_over_sent = false;
862                    quality_or_crf = video_crf;
863                }
864            }
865
866            if !send_this_stripe && output_mode == 1 && video_streaming {
867                send_this_stripe = true;
868            }
869
870            if is_dirty {
871                send_this_stripe = true;
872                stripe_state.no_motion_frame_count = 0;
873                stripe_state.paint_over_sent = false;
874                stripe_state.h264_burst_frames_remaining = 0;
875                quality_or_crf = if output_mode == 0 { jpeg_q } else { video_crf };
876            } else if !send_this_stripe {
877                stripe_state.no_motion_frame_count += 1;
878
879                if use_paint_over
880                    && stripe_state.no_motion_frame_count >= trigger_frames
881                    && !stripe_state.paint_over_sent
882                {
883                    if output_mode == 0 && paint_q > jpeg_q {
884                        send_this_stripe = true;
885                        quality_or_crf = paint_q;
886                        stripe_state.paint_over_sent = true;
887                    } else if output_mode == 1 && video_po_crf < video_crf {
888                        send_this_stripe = true;
889                        stripe_state.paint_over_sent = true;
890                        quality_or_crf = video_po_crf;
891                        force_idr = true;
892                        stripe_state.h264_burst_frames_remaining = video_burst - 1;
893                    }
894                }
895            }
896
897            if force_idr_all {
898                send_this_stripe = true;
899                if output_mode == 1 {
900                    force_idr = true;
901                    if stripe_state.h264_burst_frames_remaining <= 0 && video_burst > 0 {
902                        stripe_state.paint_over_sent = true;
903                        stripe_state.h264_burst_frames_remaining = video_burst;
904                    }
905                } else if stripe_state.paint_over_sent && use_paint_over && paint_q > jpeg_q {
906                    quality_or_crf = paint_q;
907                }
908            }
909
910            if send_this_stripe {
911                if output_mode == 0 {
912                    let pixel_format = if use_gpu {
913                        turbojpeg::PixelFormat::RGBA
914                    } else {
915                        turbojpeg::PixelFormat::BGRA
916                    };
917                    let img = turbojpeg::Image {
918                        pixels: stripe_bytes,
919                        width: width_usize,
920                        pitch: width_usize * 4,
921                        height: actual_height,
922                        format: pixel_format,
923                    };
924                    JPEG_COMPRESSOR.with(|cell| -> Option<EncodedStripe> {
925                        let mut slot = cell.borrow_mut();
926                        if slot.is_none() {
927                            *slot = Some(turbojpeg::Compressor::new().ok()?);
928                        }
929                        let compressor = slot.as_mut().unwrap();
930                        compressor.set_quality(quality_or_crf).ok()?;
931                        let jpeg = compressor.compress_to_vec(img).ok()?;
932                        let data = if omit_headers {
933                            jpeg
934                        } else {
935                            stripe_state.packet_buf.clear();
936                            stripe_state.packet_buf.push(0x03);
937                            stripe_state.packet_buf.push(0x00);
938                            stripe_state
939                                .packet_buf
940                                .extend_from_slice(&frame_counter.to_be_bytes());
941                            stripe_state
942                                .packet_buf
943                                .extend_from_slice(&(y_start as u16).to_be_bytes());
944                            stripe_state.packet_buf.extend_from_slice(&jpeg);
945                            std::mem::take(&mut stripe_state.packet_buf)
946                        };
947                        Some(EncodedStripe {
948                            data,
949                            data_type: 1,
950                            stripe_y_start: y_start as i32,
951                            stripe_height: actual_height as i32,
952                            frame_id: frame_counter as i32,
953                        })
954                    })
955                } else {
956                    let needs_reinit = if let Some(ref enc) = stripe_state.h264_encoder {
957                        enc.width != width_usize as i32
958                            || enc.height != actual_height as i32
959                            || enc.is_i444 != video_fullcolor
960                    } else {
961                        true
962                    };
963
964                    if needs_reinit {
965                        stripe_state.h264_encoder = H264EncoderWrapper::new(
966                            width_usize as i32,
967                            actual_height as i32,
968                            quality_or_crf,
969                            video_fullcolor,
970                            target_fps,
971                            h264_threads,
972                            video_cbr,

1071/// Divide `height` into `n` contiguous stripes as `(y_start, stripe_height)`, with the split
1072/// rule differing by codec because only H.264 constrains stripe height.
1073///
1074/// - **JPEG** (`output_mode 0`): JPEG has no vertical subsampling, so stripes may be any height; the
1075///   heights differ by at most one row — the first `remainder` stripes take one extra each — and
1076///   every row of the frame is covered.
1077/// - **H.264** (`output_mode 1`): 4:2:0 pairs chroma rows vertically, so every stripe height is
1078///   forced even and the remainder is handed out two rows at a time. The deliberate cost is that a
1079///   single trailing odd row may be left uncovered — preferable to an odd-height stripe the encoder
1080///   cannot represent.
1081fn compute_stripe_geometries(height: usize, n: usize, output_mode: i32) -> Vec<(usize, usize)> {
1082    let mut geoms = Vec::with_capacity(n);
1083    let mut current_y = 0;
1084    if output_mode == 0 {
1085        let base_h = height / n;
1086        let remainder = height - base_h * n;
1087        for i in 0..n {
1088            let s_h = base_h + if i < remainder { 1 } else { 0 };
1089            geoms.push((current_y, s_h));
1090            current_y += s_h;
1091        }
1092    } else {
1093        let base_h = (height / n) & !1;
1094        let remainder = height - base_h * n;
1095        let stripes_with_extra = remainder / 2;
1096        for i in 0..n {
1097            let s_h = base_h + if i < stripes_with_extra { 2 } else { 0 };
1098            geoms.push((current_y, s_h));
1099            current_y += s_h;
1100        }
1101    }
1102    geoms
1103}

1287    /// scrolled 4 px per frame — the worst case for screen-share rate control, with dense
1288    /// high-contrast detail (~40% lit pixels per glyph row) under full-frame motion.
1289    fn text_luma(frame: usize) -> Vec<u8> {
1290        let mut y = vec![18u8; W * H];
1291        let scroll = frame * 4;
1292        for row in 0..H {

1296            if in_glyph_y >= 10 {
1297                continue;
1298            }
1299            for col in 0..W {
1300                let cell_x = col / 8;
1301                let in_glyph_x = col % 8;
1302                if in_glyph_x >= 7 {
1303                    continue;
1304                }
1305                let mut s = (cell_x as u32)
1306                    .wrapping_mul(2654435761)
1307                    .wrapping_add((cell_y as u32).wrapping_mul(40503))
1308                    .wrapping_add(1);
1309                s ^= s << 13;
1310                s ^= s >> 17;
1311                s ^= s << 5;
1312                if (s >> ((in_glyph_y * 3 + in_glyph_x) % 29)) & 1 == 1 {
1313                    y[row * W + col] = 224;
1314                }
1315            }
1316        }

1396    /// Mean per-frame luma PSNR (dB) between two decoded sequences, treating a zero-MSE frame
1397    /// as 99 dB.
1398    fn mean_psnr(a: &[Vec<u8>], b: &[Vec<u8>]) -> f64 {
1399        let n = a.len().min(b.len());
1400        let mut acc = 0.0;
1401        for i in 0..n {
1402            let mse: f64 = a[i]
1403                .iter()
1404                .zip(b[i].iter())
1405                .map(|(&x, &y)| {
1406                    let d = x as f64 - y as f64;
1407                    d * d
1408                })
1409                .sum::<f64>()
1410                / a[i].len() as f64;
1411            acc += if mse <= 0.0 { 99.0 } else { 10.0 * (255.0f64 * 255.0 / mse).log10() };
1412        }
1413        acc / n.max(1) as f64
1414    }
