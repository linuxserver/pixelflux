7//! NVENC hardware H.264 encoder: CUDA-bound sessions that encode ARGB — or raw NV12 / YUV444 —
8//! frames from the X11 host-ARGB and Wayland dmabuf capture paths.
9//!
10//! The module dynamically loads `libcuda`, `libnvidia-encode` and `libEGL` at runtime, negotiates
11//! the NVENC API version against the installed driver (set-once per process), and stamps every
12//! NVENCAPI struct with the exact `NV_ENC_*_VER` word the negotiated SDK defines, so one binary
13//! drives drivers from NVENC 10.0 (~R445) through 13.0. Frames reach the GPU three ways: a
14//! zero-copy dmabuf import (EGLImage → CUDA), a pinned host→device copy of packed ARGB, and a raw
15//! planar upload. Sessions reconfigure resolution and rate control in place, so a resize or
16//! bitrate change costs a few milliseconds instead of a full rebuild.

174/// Negotiated NVENC API version `(major, minor)`, resolved once per process. `None` until
175/// `nvenc_negotiate` runs; every struct-version word and the session `apiVersion` derive from it.
176static NVENC_NEG_VER: std::sync::OnceLock<(u32, u32)> = std::sync::OnceLock::new();

178/// The NVENCAPI structs this encoder must stamp with a per-SDK version word — enumerated
179/// here precisely because getting that word exactly right is the whole mechanism that lets one
180/// compiled binary satisfy every driver's version check.
181///
182/// Each NVENCAPI struct carries a `version` field that the driver validates against the exact word
183/// its own SDK defined for that struct, rejecting anything else outright with
184/// `NV_ENC_ERR_INVALID_VERSION`. Only two parts of that packed word move between SDKs — the struct
185/// **revision** (bits 16-23) and the **`1<<31` flag** — so a session that has down-negotiated to an
186/// older API cannot send the compiled 13.0 words; it must stamp each struct with precisely the word
187/// that older SDK defined, while a current driver still receives its own native word. Naming the
188/// structs here is what lets `NvStruct::rev` supply the per-version `(revision, flag)` and
189/// `nvenc_struct_ver` assemble the word.
190#[derive(Clone, Copy, Debug)]
191enum NvStruct {
192    FunctionList,
193    OpenSessionExParams,
194    Config,
195    RcParams,
196    PresetConfig,
197    InitializeParams,
198    ReconfigureParams,
199    RegisterResource,
200    MapInputResource,
201    CreateBitstreamBuffer,
202    PicParams,
203    LockBitstream,
204}

206impl NvStruct {
207    /// The `(struct revision, 1<<31 flag)` this struct uses under the SDK identified by the
208    /// packed API version `api` (`(major<<4)|minor`) — the only two sub-fields that move between
209    /// SDKs, and thus the entire per-version knowledge stamping a struct actually needs.
210    ///
211    /// The revision lands in bits 16-23 of the version word and the flag in bit 31; every other bit
212    /// is fixed, which is exactly why matching just these two on `api` reproduces each SDK's word.
213    /// The values are transcribed verbatim from `nvEncodeAPI.h` at the FFmpeg nv-codec-headers tags
214    /// n10.0.26.2, n11.0.10.3, n11.1.5.3, n12.0.16.1, n12.1.14.0, n12.2.72.0 and n13.0.19.0, so they
215    /// are ground truth rather than anything derived that could drift. Structs whose layout is stable
216    /// across those SDKs return a constant pair; the rest match on `api`. 10.0 is the negotiation
217    /// floor, so the oldest match arm also covers anything below it.
218    fn rev(self, api: u32) -> (u32, bool) {
219        match self {
220            NvStruct::FunctionList => (2, false),
221            NvStruct::OpenSessionExParams => (1, false),
222            NvStruct::Config => match api {
223                0xC2.. => (9, true),
224                0xC0..=0xC1 => (8, true),
225                _ => (7, true),
226            },
227            NvStruct::RcParams => (1, false),
228            NvStruct::PresetConfig => (if api >= 0xC2 { 5 } else { 4 }, true),
229            NvStruct::InitializeParams => match api {
230                0xC2.. => (7, true),
231                0xC1 => (6, true),
232                _ => (5, true),
233            },
234            NvStruct::ReconfigureParams => (if api >= 0xC2 { 2 } else { 1 }, true),
235            NvStruct::RegisterResource => match api {
236                0xC2.. => (5, false),
237                0xC0..=0xC1 => (4, false),
238                _ => (3, false),
239            },
240            NvStruct::MapInputResource => (4, false),
241            NvStruct::CreateBitstreamBuffer => (1, false),
242            NvStruct::PicParams => match api {
243                0xC2.. => (7, true),
244                0xC0..=0xC1 => (6, true),
245                _ => (4, true),
246            },
247            NvStruct::LockBitstream => match api {
248                0xC2.. => (2, true),
249                0xC1 => (1, true),
250                0xC0 => (2, false),
251                _ => (1, false),
252            },
253        }
254    }
255}

257/// Assemble the `NVENCAPI_STRUCT_VERSION` word for struct `s` at API version `(maj, min)`.
258///
259/// The 32-bit word packs, from `NvStruct::rev` and the API version:
260///
261/// 1. **API major** in bits 0-7, **API minor** in bits 24-27.
262/// 2. **Struct revision** (`rev`) in bits 16-23.
263/// 3. **Magic `0x7`** in bits 28-30.
264/// 4. **The `1<<31` flag** in bit 31, when this struct sets it at this version.
265///
266/// For the pinned nvcodec-sys headers this reproduces the compile-time `NV_ENC_*_VER` constants
267/// exactly, so a current driver is stamped byte-for-byte identically to its own SDK's constant; the
268/// `version_tests` module asserts that identity.
269fn nvenc_struct_ver(s: NvStruct, maj: u32, min: u32) -> u32 {
270    let (rev, high_bit) = s.rev((maj << 4) | (min & 0xF));
271    (maj & 0xFF) | ((min & 0xF) << 24) | (rev << 16) | (0x7 << 28) | ((high_bit as u32) << 31)
272}

274/// The process's effective NVENC API version `(major, minor)`: the negotiated value once
275/// `nvenc_negotiate` has run, otherwise the pinned `NVENCAPI_VERSION` decomposed (major in the low
276/// byte, minor at bit 24) as the pre-negotiation fallback.
277#[inline]
278fn nvenc_cur_ver() -> (u32, u32) {
279    NVENC_NEG_VER
280        .get()
281        .copied()
282        .unwrap_or((NVENCAPI_VERSION & 0xFF, (NVENCAPI_VERSION >> 24) & 0xFF))
283}

285/// The struct-version word for `s` tagged with the process's negotiated API version — the
286/// value every NVENCAPI struct literal assigns to its `version` field.
287#[inline]
288fn sv(s: NvStruct) -> u32 {
289    let (m, n) = nvenc_cur_ver();
290    nvenc_struct_ver(s, m, n)
291}

293/// The negotiated session `apiVersion` word (`major | minor<<24`) passed to
294/// `NvEncOpenEncodeSessionEx` — note the minor sits at bit 24 here, unlike the `(major<<4)|minor`
295/// packing that `NvStruct::rev` matches on.
296#[inline]
297fn neg_api() -> u32 {
298    let (m, n) = nvenc_cur_ver();
299    m | (n << 24)
300}

302/// Resolve the process-wide NVENC API version once, by probing the driver newest-first and
303/// remembering the highest version it accepts.
304///
305/// The bundled nv-codec-headers are NVENC 13.0 (`pinned`), so a current driver negotiates 13.0
306/// natively while older drivers down-negotiate through 12.x / 11.x to the 10.0 floor (~R445). The
307/// compiled struct *layouts* are always the 13.0 ones; only the version *words* change per
308/// negotiated version (via the `NvStruct::rev` table), so each struct is stamped with the exact word
309/// the negotiated SDK defined. Steps:
310///
311/// 1. **Cap the search** by the driver's max: query `get_max_version` when present, then optionally
312///    lower it further from `PIXELFLUX_NVENC_MAX_API` (e.g. `"11.0"`) for testing / pinning. A cap
313///    of 0 means unknown, and probing then relies on `create_instance` acceptance alone.
314/// 2. **Probe candidates** newest-first (`pinned`, 12.1, 12.0, 11.1, 11.0, 10.0), skipping any
315///    above the cap. Each probe stamps an `NV_ENCODE_API_FUNCTION_LIST` with that version's word and
316///    calls `create_instance`.
317/// 3. **Require the whole encode path**, not just a success code: the session opener plus
318///    `nvEncInitializeEncoder`, `nvEncGetEncodePresetConfigEx`, `nvEncEncodePicture` and
319///    `nvEncLockBitstream` must all be non-null, because a driver can accept the function-list word
320///    yet leave newer entry points null. The first fully-populated version wins.
321/// 4. **Fall back** to `pinned` if nothing qualifies. Stored in `NVENC_NEG_VER`, set-once.
322fn nvenc_negotiate(lib: &NvencLibrary) {
323    NVENC_NEG_VER.get_or_init(|| {
324        let pinned = (NVENCAPI_VERSION & 0xFF, (NVENCAPI_VERSION >> 24) & 0xFF);
325        let mut drv_max: u32 = 0;
326        if let Some(get_max) = lib.get_max_version {
327            let mut m: u32 = 0;
328            if unsafe { get_max(&mut m) } == NVENCSTATUS::NV_ENC_SUCCESS {
329                drv_max = m;
330            }
331        }
332        if let Ok(cap) = std::env::var("PIXELFLUX_NVENC_MAX_API") {
333            let mut it = cap.split('.');
334            if let (Some(a), Some(b)) = (it.next(), it.next()) {
335                if let (Ok(cm), Ok(cn)) = (a.parse::<u32>(), b.parse::<u32>()) {
336                    let capv = (cm << 4) | (cn & 0xF);
337                    if capv != 0 && (drv_max == 0 || capv < drv_max) {
338                        drv_max = capv;
339                    }
340                }
341            }
342        }
343        let candidates = [pinned, (12, 1), (12, 0), (11, 1), (11, 0), (10, 0)];
344        for (maj, min) in candidates {
345            let vv = (maj << 4) | min;
346            if drv_max != 0 && vv > drv_max {
347                continue;
348            }
349            let mut probe = NV_ENCODE_API_FUNCTION_LIST {
350                version: nvenc_struct_ver(NvStruct::FunctionList, maj, min),
351                ..Default::default()
352            };
353            let st = unsafe { (lib.create_instance)(&mut probe) };
354            if st == NVENCSTATUS::NV_ENC_SUCCESS
355                && probe.nvEncOpenEncodeSessionEx.is_some()
356                && probe.nvEncInitializeEncoder.is_some()
357                && probe.nvEncGetEncodePresetConfigEx.is_some()
358                && probe.nvEncEncodePicture.is_some()
359                && probe.nvEncLockBitstream.is_some()
360            {
361                eprintln!("[pixelflux] NVENC API version negotiated: {}.{}", maj, min);
362                return (maj, min);
363            }
364        }
365        pinned
366    });
367}

445/// Release every GPU resource the session holds, in the one teardown order the drivers
446/// tolerate, so nothing leaks and no still-referenced handle is ever freed out from under the
447/// driver.
448///
449/// The whole sequence runs with the owning CUDA context pushed current, because the `cuMemFree` /
450/// `cuGraphicsUnregisterResource` / `cuMemHostUnregister` calls each act on the *current* context —
451/// pop it first and the frees silently do nothing, leaking device memory. Within that, resources
452/// go inner-handle before the outer handle that owns it, since freeing an owner first orphans or
453/// faults on what still points into it: unmap the ARGB and raw-plane inputs before unregistering
454/// them, free their device buffers, destroy the bitstream buffers, and unregister-and-destroy every
455/// cached dmabuf import — all session-owned — before the encoder session itself, and destroy that
456/// session before the CUDA context it was opened against. The page-locked host sources are unpinned
457/// in the same pass, each only when its recorded length is non-zero (a `0` marks a registration that
458/// failed and so was never pinned).
459impl Drop for NvencEncoder {
460    fn drop(&mut self) {
461        unsafe {
462            let _ = (self.cuda.cuCtxPushCurrent_v2)(self.cuda_context);
463
464            if !self.mapped_input_buffer.is_null() {
465                (self.nvenc_funcs.nvEncUnmapInputResource.unwrap())(
466                    self.encoder_session,
467                    self.mapped_input_buffer,
468                );
469            }
470            if !self.registered_input_resource.is_null() {
471                (self.nvenc_funcs.nvEncUnregisterResource.unwrap())(
472                    self.encoder_session,
473                    self.registered_input_resource,
474                );
475            }
476            if self.input_device_ptr != 0 {
477                (self.cuda.cuMemFree_v2)(self.input_device_ptr);
478            }
479
480            if let Some(mapped) = self.nv12_mapped_buffer {
481                (self.nvenc_funcs.nvEncUnmapInputResource.unwrap())(
482                    self.encoder_session,
483                    mapped,
484                );
485            }
486            if let Some(registered) = self.nv12_registered_resource {
487                (self.nvenc_funcs.nvEncUnregisterResource.unwrap())(
488                    self.encoder_session,
489                    registered,
490                );
491            }
492            if let Some(ptr) = self.nv12_device_ptr {
493                (self.cuda.cuMemFree_v2)(ptr);
494            }
495
496            for &bs in &self.bitstream_buffers {
497                (self.nvenc_funcs.nvEncDestroyBitstreamBuffer.unwrap())(
498                    self.encoder_session,
499                    bs,
500                );
501            }
502
503            for (_, cache) in self.dmabuf_cache.drain() {
504                (self.cuda.cuGraphicsUnregisterResource)(cache.cuda_resource);
505                (self.egl.eglDestroyImageKHR)(self.egl_display, cache.egl_image);
506            }
507
508            for (base, len) in &self.pinned_hosts {
509                if *len > 0 {
510                    (self.cuda.cuMemHostUnregister)(*base as *mut c_void);
511                }
512            }
513
514            if !self.encoder_session.is_null() {
515                (self.nvenc_funcs.nvEncDestroyEncoder.unwrap())(self.encoder_session);
516            }
517
518            (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
519            (self.cuda.cuCtxDestroy_v2)(self.cuda_context);
520        }

524/// The lowest H.264 level (nvcodec-sys numeric encoding: 52/60/61/62 for 5.2/6.0/6.1/6.2)
525/// whose Annex-A limits fit this resolution and frame rate, floored at 5.2.
526///
527/// A frame's cost is measured two ways against Annex-A Table A-1: **MaxFS**, the frame size in
528/// macroblocks, and **MaxMBPS**, the macroblock rate (frame size × fps). The candidate table is
529/// ascending and begins at 5.2 — there is no 5.1 entry — so every resolution up to 4K resolves
530/// deterministically to High@5.2. Only above 4K does the level step up to 6.0 / 6.1 / 6.2; a flat
531/// 5.2 would fail NVENC init there (5.2's MaxFS of 36864 MBs is ≈ 4096×2304). Picking the lowest
532/// fitting level makes the SPS advertise the smallest level a decoder must support, and because the
533/// level is fixed from frame 1 (with the profile held at High) the stream never bumps its level
534/// mid-GOP — which would force some hardware decoders to re-initialize. Above 6.2's limits there is
535/// no higher level, so it returns 6.2 as a best effort.
536fn min_h264_level(width: u32, height: u32, fps: u32) -> u32 {
537    let mbs = (width as u64).div_ceil(16) * (height as u64).div_ceil(16);
538    let mbps = mbs * fps.max(1) as u64;
539    const LEVELS: [(u32, u64, u64); 4] = [
540        (52, 36864, 2073600),
541        (60, 139264, 4177920),
542        (61, 139264, 8355840),
543        (62, 139264, 16711680),
544    ];
545    for &(level, max_fs, max_mbps) in &LEVELS {
546        if mbs <= max_fs && mbps <= max_mbps {
547            return level;
548        }
549    }
550    62
551}

555    /// without EGL — it is needed only by the zero-copy dmabuf path — and reach the
556    /// `eglCreateImageKHR` / `eglDestroyImageKHR` entry points through `eglGetProcAddress` because
557    /// they are KHR *extensions* the base `libEGL` is not obliged to export as plain symbols.
558    /// Erroring when the library or either extension is missing lets the caller fall back to another
559    /// encoder instead of crashing at the first dmabuf import.
560    fn load_egl() -> Result<EglFunctions, String> {
561        unsafe {
562            let lib_name = "libEGL.so.1";
563            let lib = Library::new(lib_name)
564                .or_else(|_| Library::new("libEGL.so"))
565                .map_err(|e| format!("Could not load EGL library: {}", e))?;
566
567            let get_proc_addr_sym: Symbol<unsafe extern "C" fn(*const c_char) -> *mut c_void> = lib
568                .get(b"eglGetProcAddress\0")
569                .map_err(|e| format!("Missing symbol eglGetProcAddress: {}", e))?;
570
571            let eglGetProcAddress = *get_proc_addr_sym;
572
573            let load_extension = |name: &str| -> Result<*mut c_void, String> {
574                let c_name = CString::new(name).unwrap();

576                if addr.is_null() {
577                    Err(format!("EGL Extension not found: {}", name))
578                } else {
579                    Ok(addr)
580                }
581            };

583            let create_addr = load_extension("eglCreateImageKHR")?;
584            let destroy_addr = load_extension("eglDestroyImageKHR")?;

586            Ok(EglFunctions {
587                _lib: lib,
588                eglGetProcAddress,
589                eglCreateImageKHR: std::mem::transmute::<*mut c_void, EglCreateImageKhrFn>(create_addr),

725    /// Build a live NVENC session: bind CUDA to the target GPU, open and configure the H.264
726    /// encoder, and allocate its input and output buffers.
727    ///
728    /// The sequence:
729    ///
730    /// 1. **Load and negotiate**: dlopen EGL / CUDA / NVENC, then `nvenc_negotiate` resolves the API
731    ///    version against the driver (set-once) before any struct is version-tagged. The multi-GPU
732    ///    `GET_ATTACHED_IDS` ioctl filter is installed after the NVIDIA libraries are loaded — so
733    ///    their GOTs can be patched — and before `cuInit` enumerates devices (a no-op unless a host
734    ///    GPU is hidden from this container). The three library `Arc`s are leaked once per process so
735    ///    the resolved function pointers stay valid for the program's life.
736    /// 2. **Bind the device**: `cuInit`, then bind by the render node's PCI bus ID
737    ///    (`encode_node_index`, with auto `<0` meaning device 0), falling back to CUDA device 0, and
738    ///    create the CUDA context.
739    /// 3. **Allocate input**: a pitched ARGB device buffer (`cuMemAllocPitch`, 16-byte element
740    ///    alignment) that hardware CSC turns into YUV.
741    /// 4. **Open the session**: create the function-list instance, open the session with the
742    ///    negotiated `apiVersion`, and pull a preset config (P4, ultra-low-latency). A failed preset
743    ///    lookup logs the driver's error string and proceeds with the zeroed default rather than
744    ///    aborting.
745    /// 5. **Configure the stream** (mutating the returned preset config, whose `version` word is
746    ///    re-stamped while its embedded `rcParams` keeps the version the preset fill set): High or
747    ///    High-4:4:4 profile; CBR (two-pass quarter-resolution for tighter per-frame rate adherence,
748    ///    VBV sizing, optional min/max QP clamps) or ConstQP; infinite GOP (`gopLength` / `idrPeriod`
749    ///    = `0xFFFFFFFF`); `zeroReorderDelay` plus a bitstream-restriction VUI (`max_num_reorder_frames=0`)
750    ///    so no-reorder decoders don't buffer; an explicit Annex-A level from `min_h264_level` pinned
751    ///    from frame 1 so the level never bumps mid-stream; BT.709 VUI colorimetry; chroma 4:2:0 or
752    ///    4:4:4 with the matching full-range flag; repeated SPS/PPS; CABAC; no AUD; strict GOP
753    ///    target; and lookahead disabled for real-time latency.
754    /// 6. **Initialize with resize headroom**: `maxEncodeWidth` / `maxEncodeHeight` are raised to at
755    ///    least 4096×2304 (the 5.2 ceiling) so `reconfigure_resolution` can grow in place; this costs
756    ///    ~290 MiB of device memory, so a failed init retries at the exact size (in-place resize then
757    ///    falls back to a rebuild).
758    /// 7. **Register, map, and buffer**: register and map the ARGB input surface, and create a
759    ///    4-deep ring of bitstream output buffers.
760    ///
761    /// Every failure after the CUDA allocation unwinds the resources created so far — buffers,
762    /// session, context — before returning `Err`. EGL is only needed by the zero-copy dmabuf path,
763    /// so callers on the host-ARGB path pass a null `egl_display`. The retained
764    /// `init_params.encodeConfig` raw pointer is nulled before the struct is returned (it points at
765    /// a local `config` about to move); the reconfigure paths repoint it at `self.encode_config`
766    /// when they resubmit.
767    pub fn new(
768        settings: &RustCaptureSettings,
769        egl_display: *const c_void,
770    ) -> Result<Self, String> {
771        println!("[NVENC] Initializing...");
772
773        let egl = Arc::new(Self::load_egl()?);
774        let cuda = Arc::new(Self::load_cuda()?);
775        let nvenc_lib = Arc::new(Self::load_nvenc()?);

778        crate::nvgpufilter::install();

780        static LEAK_ONCE: std::sync::Once = std::sync::Once::new();

782            std::mem::forget(egl.clone());
783            std::mem::forget(cuda.clone());
784            std::mem::forget(nvenc_lib.clone());
785        });
786
787        unsafe {
788            let res = (cuda.cuInit)(0);
789            if res != CUresult::CUDA_SUCCESS {
790                return Err(format!(
791                    "Init CUDA failed: {}",
792                    Self::get_error_string(&cuda, res)

794            }

796            Self::probe_devices(&cuda);

804                    println!("[NVENC] Bound to CUDA device via PCI Bus ID: {}", pci_bus_id);

1100    /// Resize the live session to `settings` in place, folding in the current rate / QP /
1101    /// fps, without tearing it down.
1102    ///
1103    /// The NVENC session, CUDA context and bitstream buffers survive, so a resize costs a few
1104    /// milliseconds instead of a full rebuild. Flow:
1105    ///
1106    /// 1. **Reject the unchangeable**: a different encode device, a chroma-format flip (4:4:4), an
1107    ///    RC-mode flip, or dimensions of zero or beyond the init-time `maxEncode` headroom all return
1108    ///    `Err` so the caller rebuilds. Chroma and RC mode are read back from the live
1109    ///    `encode_config` (the H.264 arm of the codec-config union is the one this encoder fills).
1110    /// 2. **Release geometry-dependent state** under the pushed CUDA context: unmap / unregister /
1111    ///    free the ARGB surface, the raw-plane buffer, every cached dmabuf import, and every pinned
1112    ///    host. The raw-plane buffer and dmabuf imports are re-created lazily by their encode paths;
1113    ///    pinned hosts are dropped because the source shm segments are recreated on resize and may
1114    ///    reuse the same base addresses.
1115    /// 3. **Reconfigure the session**: update the level for the new size, the CBR bitrate + VBV or
1116    ///    the ConstQP, and the new dimensions / DAR / frame rate, then `NvEncReconfigureEncoder` with
1117    ///    `resetEncoder` and `forceIDR` so the stream restarts cleanly at the new size. Driver
1118    ///    rejection returns `Err`.

1408    /// Encode one mapped input picture and return its bitstream bytes behind the wire header.
1409    ///
1410    /// The shared tail of all three encode paths:
1411    ///
1412    /// 1. **Pick an output buffer** from the ring (`current_buffer_idx` advances modulo the ring
1413    ///    length) and submit the picture with `nvEncEncodePicture`; `force_idr` sets the force-IDR
1414    ///    pic flag.
1415    /// 2. **Lock the bitstream** (`nvEncLockBitstream`, blocking) to read the encoded bytes.
1416    /// 3. **Frame the output**: unless `omit_stripe_headers` is set, prepend the 10-byte wire header
1417    ///    — a `0x04` tag, a picture-type byte derived from the *actual* encoded `pictureType`
1418    ///    (IDR = `0x01`, I = `0x02`, P = `0x00`) rather than the `force_idr` request, the low 16 bits
1419    ///    of the frame number, a zero field, and the width and height (all big-endian).
1420    /// 4. **Emit and fan out**: append the encoded bytes, mirror them to the recording sink when one
1421    ///    is attached, unlock the bitstream, and return the framed buffer.
1422    unsafe fn submit_frame(
1423        &mut self,
1424        mapped_buffer: NV_ENC_INPUT_PTR,
1425        buffer_format: NV_ENC_BUFFER_FORMAT,
1426        frame_number: u64,
1427        force_idr: bool,
1428    ) -> Result<Vec<u8>, String> {
1429        let output_bitstream = self.bitstream_buffers[self.current_buffer_idx];
1430        self.current_buffer_idx = (self.current_buffer_idx + 1) % self.bitstream_buffers.len();
1431
1432        let mut pic_params = NV_ENC_PIC_PARAMS {
1433            version: sv(NvStruct::PicParams),
1434            inputWidth: self.width,
1435            inputHeight: self.height,
1436            inputBuffer: mapped_buffer,
1437            outputBitstream: output_bitstream,
1438            bufferFmt: buffer_format,
1439            pictureStruct: NV_ENC_PIC_STRUCT::NV_ENC_PIC_STRUCT_FRAME,
1440            encodePicFlags: if force_idr {
1441                NV_ENC_PIC_FLAGS::NV_ENC_PIC_FLAG_FORCEIDR as u32
1442            } else {
1443                0
1444            },
1445            ..Default::default()
1446        };
1447
1448        let encode_fn = self.nvenc_funcs.nvEncEncodePicture.unwrap();
1449        let res = encode_fn(self.encoder_session, &mut pic_params);
1450        if res != NVENCSTATUS::NV_ENC_SUCCESS {
1451            return Err(format!("Encode Picture failed: {:?}", res));
1452        }
1453
1454        let mut lock_params = NV_ENC_LOCK_BITSTREAM {
1455            version: sv(NvStruct::LockBitstream),
1456            outputBitstream: output_bitstream,
1457            ..Default::default()
1458        };
1459        lock_params.set_doNotWait(0);
1460
1461        let lock_fn = self.nvenc_funcs.nvEncLockBitstream.unwrap();
1462        if lock_fn(self.encoder_session, &mut lock_params) != NVENCSTATUS::NV_ENC_SUCCESS {
1463            return Err("Lock Bitstream failed".into());
1464        }
1465
1466        let data_ptr = lock_params.bitstreamBufferPtr as *const u8;
1467        let data_size = lock_params.bitstreamSizeInBytes as usize;
1468        let header_sz = if self.omit_stripe_headers { 0 } else { 10 };
1469        let mut output = Vec::with_capacity(header_sz + data_size);
1470
1471        if !self.omit_stripe_headers {
1472            let type_hdr = match lock_params.pictureType {
1473                NV_ENC_PIC_TYPE::NV_ENC_PIC_TYPE_IDR => 0x01u8,
1474                NV_ENC_PIC_TYPE::NV_ENC_PIC_TYPE_I => 0x02u8,
1475                _ => 0x00u8,
1476            };
1477            output.push(0x04);
1478            output.push(type_hdr);
1479            output.extend_from_slice(&(frame_number as u16).to_be_bytes());
1480            output.extend_from_slice(&0u16.to_be_bytes());
1481            output.extend_from_slice(&(self.width as u16).to_be_bytes());
1482            output.extend_from_slice(&(self.height as u16).to_be_bytes());
1483        }
1484

1487            output.extend_from_slice(slice);
1488        }
1489
1490        (self.nvenc_funcs.nvEncUnlockBitstream.unwrap())(self.encoder_session, output_bitstream);

1494    /// Encode a dmabuf frame zero-copy, by importing it through EGL into CUDA.
1495    ///
1496    /// Applies any pending ConstQP change, then works under the pushed CUDA context:
1497    ///
1498    /// 1. **Import once, cache by fd**: on a cache miss, build an `EGLImageKHR` from the dmabuf's
1499    ///    fd / offset / pitch / modifier, register it as a CUDA graphics resource, and map it to a
1500    ///    `CUeglFrame`; the result is memoized in `dmabuf_cache` so a recurring capture buffer pays
1501    ///    the import cost only once. Each failure destroys what it created and pops the context.
1502    /// 2. **Copy into the ARGB input**: a `cuMemcpy2D` from the mapped frame — the array plane or the
1503    ///    pitch-linear plane, per `frame_type` — into the registered ARGB surface, sanitizing pitch
1504    ///    and format for NVENC.
1505    /// 3. **Submit** the ARGB input via `submit_frame`, then pop the context.
1506    ///
1507    /// The dmabuf fd is read out before the context is pushed so an early `?` return cannot leave the
1508    /// CUDA context stack imbalanced.
1509    pub fn encode(
1510        &mut self,
1511        dmabuf: &Dmabuf,
1512        frame_number: u64,
1513        target_qp: u32,
1514        force_idr: bool,
1515    ) -> Result<Vec<u8>, String> {
1516        unsafe {
1517            self.reconfigure_if_needed(target_qp);
1518            let fd = dmabuf.handles().next().ok_or("No handles")?.as_raw_fd();
1519            let _ = (self.cuda.cuCtxPushCurrent_v2)(self.cuda_context);
1520
1521            if !self.dmabuf_cache.contains_key(&fd) {
1522                let stride = dmabuf.strides().next().unwrap_or(0) as i32;
1523                let offset = dmabuf.offsets().next().unwrap_or(0) as i32;
1524                let fmt = dmabuf.format();
1525                let modifier: u64 = fmt.modifier.into();
1526
1527                let attribs = [
1528                    EGL_WIDTH,
1529                    self.width as i32,
1530                    EGL_HEIGHT,
1531                    self.height as i32,
1532                    EGL_LINUX_DRM_FOURCC_EXT,
1533                    fmt.code as i32,
1534                    EGL_DMA_BUF_PLANE0_FD_EXT,
1535                    fd,
1536                    EGL_DMA_BUF_PLANE0_OFFSET_EXT,
1537                    offset,
1538                    EGL_DMA_BUF_PLANE0_PITCH_EXT,
1539                    stride,
1540                    EGL_DMA_BUF_PLANE0_MODIFIER_LO_EXT,
1541                    (modifier & 0xFFFFFFFF) as i32,
1542                    EGL_DMA_BUF_PLANE0_MODIFIER_HI_EXT,
1543                    (modifier >> 32) as i32,
1544                    EGL_NONE,
1545                ];
1546
1547                let egl_image = (self.egl.eglCreateImageKHR)(
1548                    self.egl_display,
1549                    ptr::null_mut(),
1550                    EGL_LINUX_DMA_BUF_EXT,
1551                    ptr::null_mut(),
1552                    attribs.as_ptr(),
1553                );
1554                if egl_image == EGL_NO_IMAGE_KHR {
1555                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1556                    return Err("Failed to create EGLImage".into());
1557                }
1558
1559                let mut cuda_resource: CUgraphicsResource = ptr::null_mut();
1560                if (self.cuda.cuGraphicsEGLRegisterImage)(&mut cuda_resource, egl_image, 1)
1561                    != CUresult::CUDA_SUCCESS
1562                {
1563                    (self.egl.eglDestroyImageKHR)(self.egl_display, egl_image);
1564                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1565                    return Err("Failed to register EGLImage".into());
1566                }
1567
1568                let mut egl_frame: CUeglFrame = std::mem::zeroed();
1569                if (self.cuda.cuGraphicsResourceGetMappedEglFrame)(
1570                    &mut egl_frame,
1571                    cuda_resource,
1572                    0,
1573                    0,
1574                ) != CUresult::CUDA_SUCCESS
1575                {
1576                    (self.cuda.cuGraphicsUnregisterResource)(cuda_resource);
1577                    (self.egl.eglDestroyImageKHR)(self.egl_display, egl_image);
1578                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1579                    return Err("Failed to map EGL frame".into());
1580                }
1581
1582                self.dmabuf_cache.insert(
1583                    fd,
1584                    CachedDmaBuf {
1585                        egl_image,
1586                        cuda_resource,
1587                        egl_frame,
1588                    },
1589                );
1590            }
1591
1592            let cached = self.dmabuf_cache.get(&fd).unwrap();
1593            let mut copy_params = CUDA_MEMCPY2D {
1594                srcMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1595                srcHost: ptr::null(),
1596                srcDevice: 0,
1597                srcArray: ptr::null_mut(),
1598                srcPitch: 0,
1599                dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1600                dstHost: ptr::null_mut(),
1601                dstDevice: self.input_device_ptr,
1602                dstArray: ptr::null_mut(),
1603                dstPitch: self.input_pitch,
1604                WidthInBytes: (self.width * 4) as usize,
1605                Height: self.height as usize,
1606                ..Default::default()
1607            };
1608
1609            if cached.egl_frame.frame_type == 0 {
1610                copy_params.srcMemoryType = CUmemorytype::CU_MEMORYTYPE_ARRAY;
1611                copy_params.srcArray = cached.egl_frame.frame.p_array[0];
1612            } else {
1613                copy_params.srcMemoryType = CUmemorytype::CU_MEMORYTYPE_DEVICE;
1614                copy_params.srcDevice = cached.egl_frame.frame.p_pitch[0] as CUdeviceptr;
1615                copy_params.srcPitch = cached.egl_frame.pitch as usize;
1616            }
1617
1618            if (self.cuda.cuMemcpy2D_v2)(&copy_params) != CUresult::CUDA_SUCCESS {
1619                (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1620                return Err("Sanitization copy failed".into());
1621            }
1622
1623            let result = self.submit_frame(
1624                self.mapped_input_buffer,
1625                NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_ARGB,
1626                frame_number,
1627                force_idr,
1628            );
1629            (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1630            result
1631        }
1632    }

1634    /// Encode a host ARGB frame by uploading it straight into the ARGB input surface, with no
1635    /// CPU-side colour conversion.
1636    ///
1637    /// The packed rows are copied host→device into the registered ARGB surface and NVENC's hardware
1638    /// CSC produces the YUV. Bytes must be in NVENC ARGB order (`B,G,R,A` in memory) — the host BGRA
1639    /// layout an XShm grab yields — and `src_stride` is the source row stride in bytes (`>= width*4`).
1640    /// Steps, under the pushed CUDA context after any pending QP change:
1641    ///
1642    /// 1. **Bounds-check** the source against `stride × (rows-1) + width*4`, erroring rather than
1643    ///    reading past a short buffer.
1644    /// 2. **Pin the source once**: unless `PIXELFLUX_NVENC_PIN=0`, page-lock each distinct source
1645    ///    base address (`cuMemHostRegister`) so the copy is a direct pinned DMA instead of a pageable
1646    ///    copy staged through a driver bounce buffer. The persistent, bounded shm sources make this a
1647    ///    one-time bounded cost; a failed registration is recorded as length `0` and never re-probed.
1648    /// 3. **Copy and submit**: `cuMemcpy2D` the rows into the ARGB surface honoring `src_stride`,
1649    ///    then `submit_frame`.
1650    pub fn encode_cpu_argb(
1651        &mut self,
1652        argb: &[u8],
1653        src_stride: usize,
1654        frame_number: u64,
1655        target_qp: u32,
1656        force_idr: bool,
1657    ) -> Result<Vec<u8>, String> {
1658        unsafe {
1659            self.reconfigure_if_needed(target_qp);
1660            let _ = (self.cuda.cuCtxPushCurrent_v2)(self.cuda_context);
1661
1662            let width_bytes = (self.width * 4) as usize;
1663            let rows = self.height as usize;
1664            let needed = if rows == 0 { 0 } else { src_stride * (rows - 1) + width_bytes };
1665            if src_stride < width_bytes || argb.len() < needed {
1666                (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1667                return Err(format!(
1668                    "ARGB buffer too small: len={} need>={} (stride={}, {}x{})",
1669                    argb.len(), needed, src_stride, self.width, self.height
1670                ));
1671            }
1672
1673            if std::env::var("PIXELFLUX_NVENC_PIN").as_deref() != Ok("0") {
1674                let base = argb.as_ptr() as usize;
1675                if let std::collections::hash_map::Entry::Vacant(e) = self.pinned_hosts.entry(base) {
1676                    let st = (self.cuda.cuMemHostRegister_v2)(argb.as_ptr() as *mut c_void, argb.len(), 0);
1677                    e.insert(if st == CUresult::CUDA_SUCCESS { argb.len() } else { 0 });
1678                }
1679            }
1680
1681            let copy = CUDA_MEMCPY2D {
1682                srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1683                srcHost: argb.as_ptr() as *const c_void,
1684                srcPitch: src_stride,
1685                dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1686                dstDevice: self.input_device_ptr,
1687                dstPitch: self.input_pitch,
1688                WidthInBytes: width_bytes,
1689                Height: rows,
1690                ..Default::default()
1691            };
1692            if (self.cuda.cuMemcpy2D_v2)(&copy) != CUresult::CUDA_SUCCESS {
1693                (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1694                return Err("ARGB host->device copy failed".into());
1695            }
1696
1697            let result = self.submit_frame(
1698                self.mapped_input_buffer,
1699                NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_ARGB,
1700                frame_number,
1701                force_idr,
1702            );
1703            (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1704            result
1705        }
1706    }

1708    /// Encode a raw planar frame — NV12 (4:2:0) or YUV444 — uploaded host→device.
1709    ///
1710    /// The planar-input counterpart to `encode_cpu_argb`, used when the caller has already produced
1711    /// YUV. The chroma format follows the session's `chromaFormatIDC` (3 ⇒ YUV444, else NV12). Flow
1712    /// under the pushed CUDA context after any pending QP change:
1713    ///
1714    /// 1. **Lazily allocate** the planar device buffer on first use: a pitched allocation tall enough
1715    ///    for three full planes (YUV444) or Y plus half-height interleaved UV (NV12), registered and
1716    ///    mapped with the matching buffer format.
1717    /// 2. **Upload each plane** with its own `cuMemcpy2D`. Every copy is bounds-checked against the
1718    ///    host slice: the Y plane is required in full (a short buffer errors), and each chroma plane
1719    ///    is copied only if its **entire** span — not merely its start offset — is present, so a
1720    ///    truncated buffer never reads past its end.
1721    /// 3. **Submit** the mapped planar input via `submit_frame`.
1722    pub fn encode_raw(
1723        &mut self,
1724        raw_data: &[u8],
1725        frame_number: u64,
1726        target_qp: u32,
1727        force_idr: bool,
1728    ) -> Result<Vec<u8>, String> {
1729        unsafe {
1730            self.reconfigure_if_needed(target_qp);
1731            let _ = (self.cuda.cuCtxPushCurrent_v2)(self.cuda_context);
1732
1733            let is_444 = self.encode_config.encodeCodecConfig.h264Config.chromaFormatIDC == 3;
1734
1735            if self.nv12_device_ptr.is_none() {
1736                let mut d_ptr: CUdeviceptr = 0;
1737                let mut pitch: usize = 0;
1738
1739                let alloc_height = if is_444 {
1740                    self.height * 3
1741                } else {
1742                    self.height + (self.height / 2)
1743                };
1744
1745                let res = (self.cuda.cuMemAllocPitch_v2)(
1746                    &mut d_ptr,
1747                    &mut pitch,
1748                    self.width as usize,
1749                    alloc_height as usize,
1750                    16,
1751                );
1752                if res != CUresult::CUDA_SUCCESS {
1753                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1754                    return Err("Failed to allocate GPU buffer for raw input".into());
1755                }
1756
1757                let buffer_fmt = if is_444 {
1758                    NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_YUV444
1759                } else {
1760                    NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_NV12
1761                };
1762
1763                let mut reg_res = NV_ENC_REGISTER_RESOURCE {
1764                    version: sv(NvStruct::RegisterResource),
1765                    resourceType:
1766                        NV_ENC_INPUT_RESOURCE_TYPE::NV_ENC_INPUT_RESOURCE_TYPE_CUDADEVICEPTR,
1767                    width: self.width,
1768                    height: self.height,
1769                    resourceToRegister: d_ptr as *mut c_void,
1770                    pitch: pitch as u32,
1771                    bufferFormat: buffer_fmt,
1772                    bufferUsage: NV_ENC_BUFFER_USAGE::NV_ENC_INPUT_IMAGE,
1773                    ..Default::default()
1774                };
1775
1776                let register_fn = self.nvenc_funcs.nvEncRegisterResource.unwrap();
1777                if register_fn(self.encoder_session, &mut reg_res) != NVENCSTATUS::NV_ENC_SUCCESS {
1778                    (self.cuda.cuMemFree_v2)(d_ptr);
1779                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1780                    return Err("Failed to register raw input buffer".into());
1781                }
1782
1783                let mut map_params = NV_ENC_MAP_INPUT_RESOURCE {
1784                    version: sv(NvStruct::MapInputResource),
1785                    registeredResource: reg_res.registeredResource,
1786                    ..Default::default()
1787                };
1788                let map_fn = self.nvenc_funcs.nvEncMapInputResource.unwrap();
1789                if map_fn(self.encoder_session, &mut map_params) != NVENCSTATUS::NV_ENC_SUCCESS {
1790                    (self.nvenc_funcs.nvEncUnregisterResource.unwrap())(
1791                        self.encoder_session,
1792                        reg_res.registeredResource,
1793                    );
1794                    (self.cuda.cuMemFree_v2)(d_ptr);
1795                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1796                    return Err("Failed to map raw input buffer".into());
1797                }
1798
1799                self.nv12_device_ptr = Some(d_ptr);
1800                self.nv12_pitch = pitch;
1801                self.nv12_registered_resource = Some(reg_res.registeredResource);
1802                self.nv12_mapped_buffer = Some(map_params.mappedResource);
1803            }
1804
1805            let dev_ptr = self.nv12_device_ptr.unwrap();
1806            let dev_pitch = self.nv12_pitch;
1807            let width_bytes = self.width as usize;
1808            let height = self.height as usize;
1809
1810            if is_444 {
1811                let plane_size = width_bytes * height;
1812                if raw_data.len() < plane_size {
1813                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1814                    return Err("raw frame smaller than the Y plane (444)".into());
1815                }
1816
1817                let copy_y = CUDA_MEMCPY2D {
1818                    srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1819                    srcHost: raw_data.as_ptr() as *const c_void,
1820                    srcPitch: width_bytes,
1821                    dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1822                    dstDevice: dev_ptr,
1823                    dstPitch: dev_pitch,
1824                    WidthInBytes: width_bytes,
1825                    Height: height,
1826                    ..Default::default()
1827                };
1828                if (self.cuda.cuMemcpy2D_v2)(&copy_y) != CUresult::CUDA_SUCCESS {
1829                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1830                    return Err("Failed to copy Y plane (444)".into());
1831                }
1832
1833                if raw_data.len() >= 2 * plane_size {
1834                    let copy_u = CUDA_MEMCPY2D {
1835                        srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1836                        srcHost: raw_data[plane_size..].as_ptr() as *const c_void,
1837                        srcPitch: width_bytes,
1838                        dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1839                        dstDevice: dev_ptr + (dev_pitch * height) as u64,
1840                        dstPitch: dev_pitch,
1841                        WidthInBytes: width_bytes,
1842                        Height: height,
1843                        ..Default::default()
1844                    };
1845                    if (self.cuda.cuMemcpy2D_v2)(&copy_u) != CUresult::CUDA_SUCCESS {
1846                        (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1847                        return Err("Failed to copy U plane (444)".into());
1848                    }
1849                }
1850
1851                if raw_data.len() >= 3 * plane_size {
1852                    let copy_v = CUDA_MEMCPY2D {
1853                        srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1854                        srcHost: raw_data[2 * plane_size..].as_ptr() as *const c_void,
1855                        srcPitch: width_bytes,
1856                        dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1857                        dstDevice: dev_ptr + (dev_pitch * height * 2) as u64,
1858                        dstPitch: dev_pitch,
1859                        WidthInBytes: width_bytes,
1860                        Height: height,
1861                        ..Default::default()
1862                    };
1863                    if (self.cuda.cuMemcpy2D_v2)(&copy_v) != CUresult::CUDA_SUCCESS {
1864                        (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1865                        return Err("Failed to copy V plane (444)".into());
1866                    }
1867                }
1868            } else {
1869                let y_size = width_bytes * height;
1870                if raw_data.len() < y_size {
1871                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1872                    return Err("raw frame smaller than the Y plane".into());
1873                }
1874                let copy_y = CUDA_MEMCPY2D {
1875                    srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1876                    srcHost: raw_data.as_ptr() as *const c_void,
1877                    srcPitch: width_bytes,
1878                    dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1879                    dstDevice: dev_ptr,
1880                    dstPitch: dev_pitch,
1881                    WidthInBytes: width_bytes,
1882                    Height: height,
1883                    ..Default::default()
1884                };
1885                if (self.cuda.cuMemcpy2D_v2)(&copy_y) != CUresult::CUDA_SUCCESS {
1886                    (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1887                    return Err("Failed to copy Y plane".into());
1888                }
1889
1890                let uv_offset = y_size;
1891                if raw_data.len() >= uv_offset + width_bytes * (height / 2) {
1892                    let copy_uv = CUDA_MEMCPY2D {
1893                        srcMemoryType: CUmemorytype::CU_MEMORYTYPE_HOST,
1894                        srcHost: raw_data[uv_offset..].as_ptr() as *const c_void,
1895                        srcPitch: width_bytes,
1896                        dstMemoryType: CUmemorytype::CU_MEMORYTYPE_DEVICE,
1897                        dstDevice: dev_ptr + (dev_pitch * height) as u64,
1898                        dstPitch: dev_pitch,
1899                        WidthInBytes: width_bytes,
1900                        Height: height / 2,
1901                        ..Default::default()
1902                    };
1903                    if (self.cuda.cuMemcpy2D_v2)(&copy_uv) != CUresult::CUDA_SUCCESS {
1904                        (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1905                        return Err("Failed to copy UV plane".into());
1906                    }
1907                }
1908            }
1909
1910            let raw_format = if self.encode_config.encodeCodecConfig.h264Config.chromaFormatIDC == 3 {
1911                NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_YUV444
1912            } else {
1913                NV_ENC_BUFFER_FORMAT::NV_ENC_BUFFER_FORMAT_NV12
1914            };
1915            let result =
1916                self.submit_frame(self.nv12_mapped_buffer.unwrap(), raw_format, frame_number, force_idr);
1917            (self.cuda.cuCtxPopCurrent_v2)(ptr::null_mut());
1918            result
1919        }
1920    }
