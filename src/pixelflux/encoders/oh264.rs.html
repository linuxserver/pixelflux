7//! Software H.264 encoder built on Cisco's OpenH264: the full-frame software path
8//! for H.264 output, opted into with `use_openh264`. Its reason to exist alongside
9//! the striped software x264 path is that it emits one continuous single-stream
10//! H.264 bitstream instead of independent per-stripe streams — that single stream
11//! records as-is and drives through one hardware decoder, so a host with no usable
12//! GPU encoder still looks to a client exactly like an NVENC/VAAPI session. Cisco's
13//! BSD license is what lets it ship redistributably; the price of admission is being
14//! 4:2:0-only and Annex-B.
15//!
16//! Keyframes are driven externally — an effectively infinite intra period plus
17//! on-demand `force_intra_frame`, never a fixed cadence — so a mostly-static screen
18//! spends no bitrate on keyframes nothing requested, matching the NVENC/x264
19//! strict-GOP streaming behavior. Host ARGB is converted to I420 with the same
20//! BT.709 path the x264 encoder uses, then fed to OpenH264 as borrowed planes.
21
22use crate::recording_sink::RecordingSink;
23use std::sync::Arc;
24use crate::RustCaptureSettings;
25use openh264::encoder::{
26    BitRate, Complexity, Encoder, EncoderConfig, FrameRate, FrameType, IntraFramePeriod, QpRange,
27    RateControlMode, UsageType,
28};
29use openh264::formats::YUVSlices;
30use openh264::OpenH264API;

32/// Suppresses OpenH264's built-in periodic IDR so the GOP is effectively infinite and a
33/// keyframe costs bitrate only when something genuinely needs one.
34///
35/// A wall-clock keyframe cadence spikes bitrate and dents quality on a screen that is not changing,
36/// so recovery keyframes are driven entirely by the pipeline's force-IDR decision
37/// (`force_intra_frame`) instead. The period is simply set so large the encoder effectively never
38/// reaches it within a session.
39const INFINITE_INTRA_PERIOD: u32 = 300_000;
40
41/// A deliberately huge bitrate value (100 Mbps, still well under H.264 level 5.2's cap) used
42/// wherever rate control must be kept from becoming the binding constraint. Two such places:
43///
44/// 1. **CRF/CQP rate budget**: passed as the target bitrate so the pinned QP — not the bitrate —
45///    always dominates rate control and the constant QP is what actually shapes each frame.
46/// 2. **Live CBR headroom**: the max bitrate a live CBR change may lift the session to.
47///    Construction pins OpenH264's max bitrate to the *starting* target (EncoderConfig exposes no
48///    separate max), which would otherwise reject any later raise above it; lifting the max to this
49///    ceiling first is what lets `set_live_bitrate` move the target freely.
50const BITRATE_CEILING_BPS: u32 = 100_000_000;
51
52/// One capture session's worth of OpenH264 state, kept alive between frames because the
53/// encoder's reference chain and rate-control state must persist across the whole session — and the
54/// I420 plane buffers are reused every frame to keep the per-frame hot path free of allocation.
55///
56/// Holds the live `Encoder`, the fixed frame dimensions, and the reusable I420 plane buffers
57/// (`y_buf` / `u_buf` / `v_buf`) that each frame's RGB-to-YUV conversion writes into before
58/// hand-off. `is_cbr` selects the rate-control mode: `true` is CBR (bitrate-mode RC driving a
59/// target bitrate), `false` is CRF/CQP (the same bitrate-mode RC but with the QP pinned to a
60/// single value). `current_bitrate_bps` tracks the currently-accepted live CBR target so a
61/// rejected change can be rolled back. `omit_stripe_headers` drops the 10-byte wire header for
62/// bare Annex-B output, and `recording_sink` is the optional raw-Annex-B fan-out.
63pub struct Openh264Encoder {
64    encoder: Encoder,
65    width: usize,
66    height: usize,
67    y_buf: Vec<u8>,
68    u_buf: Vec<u8>,
69    v_buf: Vec<u8>,
70    current_bitrate_bps: i32,
71    is_cbr: bool,
72    omit_stripe_headers: bool,
73    recording_sink: Option<Arc<RecordingSink>>,
74}
75
76impl Openh264Encoder {
77    /// Build an OpenH264 encoder from the capture settings, or `None` on init failure.
78    ///
79    /// `None` lets the caller fall back to the x264 software stripe path. Like the NVENC/VAAPI
80    /// encoders, this one writes raw Annex-B directly to `recording_sink` and self-prepends the
81    /// wire header on the buffer it returns.
82    ///
83    /// 1. **Geometry**: width and height are floored to even values (`& !1`, min 2) because 4:2:0
84    ///    chroma subsampling requires it. A 4:4:4 full-color request is only warned about —
85    ///    OpenH264 is 4:2:0-only, so it is encoded 4:2:0 regardless (NVENC/x264 honor full color).
86    ///
87    /// 2. **Thread count**: one fewer than the available parallelism — leaving headroom for capture
88    ///    and the rest of the pipeline — clamped to `[1, 4]`, since the frame is split into four
89    ///    slices and more encode threads than slices would buy nothing.
90    ///
91    /// 3. **CRF floor**: `video_crf` is clamped to `[1, 51]`. The floor is 1, not 0, because a
92    ///    zero anywhere in the QP pair makes OpenH264's `ParamValidation` discard the *entire*
93    ///    range and substitute its screen-content defaults `[26, 35]`; explicit non-zero values
94    ///    are honored (the library clips the min up to its own floor of 12).
95    ///
96    /// 4. **Base config** (shared by both rate modes): screen-content real-time usage, low
97    ///    complexity, an effectively infinite intra period, and the thread count. Frame skip is
98    ///    **enabled** so the rate controller can actually hold the target bitrate — skip-less
99    ///    bitrate control is only approximate. A skipped frame is not encoded and the next P-frame
100    ///    references the last *encoded* frame, so the reference chain stays intact. Adaptive
101    ///    quantization and background detection are set explicitly to `false` to silence the init
102    ///    warnings (both are auto-disabled for screen content anyway). Scene-change detection stays
103    ///    on: `ParamValidation` force-enables it for screen-content usage, so its content-adaptive
104    ///    IDRs are inherent to this encoder and the wire header labels them correctly. The internal
105    ///    stderr trace follows the capture's `debug_logging` flag (otherwise quiet).
106    ///
107    /// 5. **Rate control**:
108    ///    - **CBR** (`video_cbr_mode`): bitrate-mode RC targeting `video_bitrate_kbps`, with the QP
109    ///      range made explicit (`video_min_qp`, and `video_max_qp` defaulting the max to 51). The
110    ///      explicit range dodges the same zero-substitution as the CRF floor, which would
111    ///      otherwise cap the RC at QP 35 where hard content overshoots any target.
112    ///    - **CRF/CQP** (default): the same bitrate-mode RC but with an oversized bitrate budget
113    ///      (`BITRATE_CEILING_BPS`) and the QP pinned to a single value (`min == max == crf`), so
114    ///      the QP clamp always dominates and every frame uses the constant QP — matching NVENC
115    ///      CONSTQP / VAAPI CQP. Bitrate-mode with a pinned QP is what actually holds the QP
116    ///      constant: RC_OFF ignores the QP range and RC_QUALITY rejects a `min == max` range.
117    ///
118    /// After construction the encoder is primed for four fixed slices and VUI signaling via
119    /// `enable_four_slices`, then the recording sink is attached.
120    pub fn new(settings: &RustCaptureSettings, recording_sink: Option<Arc<RecordingSink>>) -> Option<Self> {
121        if settings.video_fullcolor {
122            eprintln!("[openh264] 4:4:4 full-color requested; OpenH264 is 4:2:0-only, encoding 4:2:0.");
123        }
124        let width = (settings.width.max(2) as usize) & !1;
125        let height = (settings.height.max(2) as usize) & !1;
126        let bps = (settings.video_bitrate_kbps.max(1) as u32).saturating_mul(1000);
127        let fps = if settings.target_fps < 1.0 { 30.0 } else { settings.target_fps as f32 };
128        let threads = std::thread::available_parallelism()
129            .map(|n| n.get().saturating_sub(1).clamp(1, 4))
130            .unwrap_or(1) as u16;
131        let is_cbr = settings.video_cbr_mode;
132        let crf = settings.video_crf.clamp(1, 51) as u8;
133
134        let base = EncoderConfig::new()
135            .max_frame_rate(FrameRate::from_hz(fps))
136            .usage_type(UsageType::ScreenContentRealTime)
137            .complexity(Complexity::Low)
138            .skip_frames(true)
139            .adaptive_quantization(false)
140            .background_detection(false)
141            .debug(settings.debug_logging)
142            .intra_frame_period(IntraFramePeriod::from_num_frames(INFINITE_INTRA_PERIOD))
143            .num_threads(threads);
144        let config = if is_cbr {
145            let min_qp = settings.video_min_qp.clamp(1, 51) as u8;
146            let max_qp = if settings.video_max_qp > 0 {
147                settings.video_max_qp.clamp(min_qp as i32, 51) as u8
148            } else {
149                51
150            };
151            base.bitrate(BitRate::from_bps(bps))
152                .rate_control_mode(RateControlMode::Bitrate)
153                .qp(QpRange::new(min_qp, max_qp))
154        } else {
155            base.bitrate(BitRate::from_bps(BITRATE_CEILING_BPS))
156                .rate_control_mode(RateControlMode::Bitrate)
157                .qp(QpRange::new(crf, crf))
158        };
159
160        let encoder = Encoder::with_api_config(OpenH264API::from_source(), config).ok()?;
161
162        let (cw, ch) = (width / 2, height / 2);
163        let mut me = Self {
164            encoder,
165            width,
166            height,
167            y_buf: vec![0u8; width * height],
168            u_buf: vec![0u8; cw * ch],
169            v_buf: vec![0u8; cw * ch],
170            current_bitrate_bps: bps as i32,
171            is_cbr,
172            omit_stripe_headers: settings.omit_stripe_headers,
173            recording_sink: None,
174        };
175        me.enable_four_slices();
176        me.recording_sink = recording_sink;
177        Some(me)
178    }
179
180    /// Give every frame four fixed slices and explicit VUI colour signaling (BT.709, limited
181    /// range) — the two properties a streaming client depends on that OpenH264 will not produce on
182    /// its own: without the slices, encode and decode cannot parallelize; without VUI, the picture
183    /// can arrive visibly dark.
184    ///
185    /// 1. **Four fixed slices**: both the encoder threads and client decoders slice-parallelize,
186    ///    while more than four slices upsets some Chromium decoders, so the count is pinned to four
187    ///    (`SM_FIXEDSLCNUM_SLICE`, `uiSliceNum = 4`).
188    /// 2. **VUI colour signaling**: BT.709 primaries/transfer/matrix at limited range, matching the
189    ///    RGB-to-YUV conversion. Without VUI a WebRTC receiver falls back to the SDP profile to
190    ///    infer range, so a full-range-negotiated (4:4:4) session would display this encoder's
191    ///    limited-range output unexpanded — visibly darker; x264/NVENC write VUI, so only this
192    ///    encoder needs it set here.
193    ///
194    /// The `openh264` crate initializes the underlying encoder lazily on the *first* encode and
195    /// exposes neither fixed slicing nor VUI through its config, so the routine primes the encoder
196    /// with one throwaway frame, patches the now-live `SEncParamExt` parameter set (slice mode plus
197    /// the VUI fields) via `set_option`, and lets the next encode re-init from it. It is
198    /// best-effort: any `get_option` / `set_option` failure returns early, leaving a working
199    /// single-slice encoder without VUI. Because the throwaway frame consumed the initial IDR, a
200    /// final `force_intra_frame` guarantees the first *delivered* frame is still an IDR.
201    fn enable_four_slices(&mut self) {
202        let slices = YUVSlices::new(
203            (&self.y_buf, &self.u_buf, &self.v_buf),
204            (self.width, self.height),
205            (self.width, self.width / 2, self.width / 2),
206        );
207        if self.encoder.encode(&slices).is_err() {
208            return;
209        }
210        unsafe {
211            let mut params: openh264_sys2::SEncParamExt = std::mem::zeroed();
212            let raw = self.encoder.raw_api();
213            if raw.get_option(
214                openh264_sys2::ENCODER_OPTION_SVC_ENCODE_PARAM_EXT,
215                &mut params as *mut _ as *mut std::ffi::c_void,
216            ) != 0
217            {
218                return;
219            }
220            params.sSpatialLayers[0].sSliceArgument.uiSliceMode =
221                openh264_sys2::SM_FIXEDSLCNUM_SLICE;
222            params.sSpatialLayers[0].sSliceArgument.uiSliceNum = 4;
223            {
224                let layer = &mut params.sSpatialLayers[0];
225                layer.bVideoSignalTypePresent = true;
226                layer.uiVideoFormat = 5;
227                layer.bFullRange = false;
228                layer.bColorDescriptionPresent = true;
229                layer.uiColorPrimaries = 1;
230                layer.uiTransferCharacteristics = 1;
231                layer.uiColorMatrix = 1;
232            }
233            let ret = raw.set_option(
234                openh264_sys2::ENCODER_OPTION_SVC_ENCODE_PARAM_EXT,
235                &mut params as *mut _ as *mut std::ffi::c_void,
236            );
237            if ret != 0 {
238                eprintln!("[openh264] stream-param re-init rejected ({ret}); staying single-slice without VUI");
239            }
240        }
241        self.encoder.force_intra_frame();
242    }
243
244    /// Apply a live bitrate (kbps) and/or framerate change in place, so the session keeps its
245    /// reference chain and infinite GOP instead of taking the visible reset a full rebuild would
246    /// force on the viewer.
247    ///
248    /// OpenH264 honors both through `SetOption`, so the reference chain and the infinite GOP are
249    /// preserved and the change takes effect mid-stream — this is what makes the web UI's bitrate
250    /// slider work, like NVENC and x264. The bitrate is applied only in **CBR** mode, and only when
251    /// it actually differs from the current target; in **CRF/CQP** the QP is fixed at construction,
252    /// so a CRF change instead restarts the capture and rebuilds the encoder (there is no live-QP
253    /// setter). A framerate of at least 1 fps is pushed via the frame-rate option.
254    pub fn reconfigure_rate(&mut self, bitrate_kbps: i32, fps: f64) {
255        if self.is_cbr {
256            let bps = bitrate_kbps.max(1).saturating_mul(1000);
257            if bps != self.current_bitrate_bps {
258                self.set_live_bitrate(bps);
259            }
260        }
261        if fps >= 1.0 {
262            let mut rate = fps as f32;
263            unsafe {
264                self.encoder.raw_api().set_option(
265                    openh264_sys2::ENCODER_OPTION_FRAME_RATE,
266                    std::ptr::addr_of_mut!(rate).cast(),
267                );
268            }
269        }
270    }
271
272    /// Move the live CBR target to `bps` — but raise the max-bitrate ceiling first, because
273    /// OpenH264 refuses any target above the max that construction pinned to the session's *starting*
274    /// bitrate.
275    ///
276    /// Construction pins OpenH264's max bitrate to the starting target (EncoderConfig exposes no
277    /// separate max), and `SetOption` rejects — after partially applying — any target above that
278    /// max. So the ceiling is first lifted to `BITRATE_CEILING_BPS` for both the overall stream
279    /// (`SPATIAL_LAYER_ALL`) and layer 0 (verification reads the per-layer value); if raising the
280    /// ceiling fails, the previous rate is kept and the change aborts. The target is then set on the
281    /// whole stream: on success `current_bitrate_bps` is updated; on failure the already-mutated
282    /// target is re-set back to `current_bitrate_bps` so the encoder stays in sync with the tracked
283    /// value.
284    fn set_live_bitrate(&mut self, bps: i32) {
285        for layer in [openh264_sys2::SPATIAL_LAYER_ALL, openh264_sys2::SPATIAL_LAYER_0] {
286            let ret = self.set_bitrate_option(
287                openh264_sys2::ENCODER_OPTION_MAX_BITRATE,
288                layer,
289                BITRATE_CEILING_BPS as i32,
290            );
291            if ret != 0 {
292                eprintln!(
293                    "[openh264] live bitrate change to {bps} bps failed raising the max-bitrate ceiling \
294                     (code {ret}); keeping {} bps",
295                    self.current_bitrate_bps
296                );
297                return;
298            }
299        }
300        let ret =
301            self.set_bitrate_option(openh264_sys2::ENCODER_OPTION_BITRATE, openh264_sys2::SPATIAL_LAYER_ALL, bps);
302        if ret == 0 {
303            self.current_bitrate_bps = bps;
304        } else {
305            self.set_bitrate_option(
306                openh264_sys2::ENCODER_OPTION_BITRATE,
307                openh264_sys2::SPATIAL_LAYER_ALL,
308                self.current_bitrate_bps,
309            );
310            eprintln!(
311                "[openh264] live bitrate change to {bps} bps failed (code {ret}); keeping {} bps",
312                self.current_bitrate_bps
313            );
314        }
315    }
316
317    /// Marshal one `SBitrateInfo` FFI struct and issue a single `SetOption` (BITRATE or
318    /// MAX_BITRATE) for the given layer, returning the raw return code (0 on success). Exists so the
319    /// ceiling-raise and the target-set in `set_live_bitrate` share one unsafe marshalling path
320    /// rather than open-coding the same `zeroed()`-and-cast dance twice.
321    fn set_bitrate_option(
322        &mut self,
323        option: openh264_sys2::ENCODER_OPTION,
324        layer: openh264_sys2::LAYER_NUM,
325        bps: i32,
326    ) -> i32 {
327        let mut info: openh264_sys2::SBitrateInfo = unsafe { std::mem::zeroed() };
328        info.iLayer = layer;
329        info.iBitrate = bps;
330        unsafe { self.encoder.raw_api().set_option(option, std::ptr::addr_of_mut!(info).cast()) }
331    }
332
333    /// Encode one host frame and return it framed exactly like the hardware full-frame
334    /// encoders, so the client demuxes this software path with the same code and cannot tell a
335    /// GPU-less session apart. Returns an empty buffer when the frame produced no payload.
336    ///
337    /// 1. **Keyframe**: when `force_idr` is set, `force_intra_frame` is called so this frame is
338    ///    emitted as an IDR.
339    /// 2. **Colour conversion**: `argb` (with `stride` bytes per row) is converted to I420 into the
340    ///    reusable Y/U/V plane buffers. `rgba_input` selects the source byte order — `false` is
341    ///    B,G,R,A (X11 XShm), `true` is R,G,B,A (Wayland GL readback).
342    /// 3. **Encode**: the planes are wrapped as borrowed `YUVSlices` and encoded.
343    /// 4. **Framing**: unless `omit_stripe_headers` is set, a 10-byte wire header is prepended —
344    ///    byte-for-byte the layout the NVENC/VAAPI/x264 full-frame paths emit, which is what lets a
345    ///    single client demuxer serve every encoder. Its type byte is read from the *actually
346    ///    encoded* picture type (IDR = `0x01`, I = `0x02`, P = `0x00`) rather than from `force_idr`,
347    ///    because OpenH264's scene-change detection can emit an IDR the caller never asked for, and
348    ///    the header must label a real decode entry point, not the request. It also carries the
349    ///    frame number, a zero y-start, and the width/height (all big-endian). With
350    ///    `omit_stripe_headers` the output is bare Annex-B.
351    /// 5. **Empty payload**: if the encoder produced no bitstream (e.g. a skipped frame), an empty
352    ///    vec is returned rather than a lone header — a header with no Annex-B behind it would be a
353    ///    malformed frame to the client, and the pipeline reads empty as "nothing to send".
354    /// 6. **Recording sink**: the raw Annex-B payload (past the header) is also written to the
355    ///    recording sink when one is attached, since a recorder wants the bare elementary stream,
356    ///    not the wire framing.
357    pub fn encode_host_argb(
358        &mut self,
359        argb: &[u8],
360        stride: usize,
361        frame_number: u64,
362        force_idr: bool,
363        rgba_input: bool,
364    ) -> Result<Vec<u8>, String> {
365        if force_idr {
366            self.encoder.force_intra_frame();
367        }
368        let cw = self.width / 2;
369        crate::encoders::software::convert_to_yuv_mt(
370            argb,
371            stride as u32,
372            self.width,
373            self.height,
374            rgba_input,
375            false,
376            &mut self.y_buf,
377            &mut self.u_buf,
378            &mut self.v_buf,
379            4,
380        )
381        .map_err(|e| format!("rgb-to-yuv420 failed: {e:?}"))?;
382
383        let slices = YUVSlices::new(
384            (&self.y_buf, &self.u_buf, &self.v_buf),
385            (self.width, self.height),
386            (self.width, cw, cw),
387        );
388        match self.encoder.encode(&slices) {
389            Ok(bitstream) => {
390                let header_sz = if self.omit_stripe_headers { 0 } else { 10 };
391                let mut out = Vec::with_capacity(header_sz);
392                if header_sz != 0 {
393                    let type_hdr = match bitstream.frame_type() {
394                        FrameType::IDR => 0x01u8,
395                        FrameType::I => 0x02u8,
396                        _ => 0x00u8,
397                    };
398                    out.push(0x04);
399                    out.push(type_hdr);
400                    out.extend_from_slice(&(frame_number as u16).to_be_bytes());
401                    out.extend_from_slice(&0u16.to_be_bytes());
402                    out.extend_from_slice(&(self.width as u16).to_be_bytes());
403                    out.extend_from_slice(&(self.height as u16).to_be_bytes());
404                }
405                bitstream.write_vec(&mut out);
406                if out.len() == header_sz {
407                    return Ok(Vec::new());
408                }
409                if let Some(ref sink) = self.recording_sink {
410                    sink.write_frame(&out[header_sz..]);
411                }
412                Ok(out)
413            }
414            Err(e) => Err(format!("openh264 encode failed: {e:?}")),
415        }
416    }
417}

423    /// Build a busy, motion-carrying test frame (high-entropy per-pixel content) so the
424    /// rate controller produces non-trivial P-frames. Fine under CBR (the RC raises QP to fit) but
425    /// pathological for a pinned low QP, so the CRF test uses the compressible `gradient_frame`.
426    fn busy_frame(w: usize, h: usize, t: usize) -> Vec<u8> {
427        let mut f = vec![0u8; w * h * 4];
428        for y in 0..h {
429            for x in 0..w {
430                let i = (y * w + x) * 4;
431                let v = (((x + t) * 7 + (y + t) * 13) & 0xFF) as u8;
432                f[i] = v;
433                f[i + 1] = v.wrapping_mul(3);
434                f[i + 2] = v.wrapping_add((x + t) as u8);
435                f[i + 3] = 255;
436            }
437        }
438        f
439    }

441    /// Build a smooth diagonal-gradient test frame (moderate, realistic screen-like detail):
442    /// compressible enough not to overflow at a low pinned QP, yet detailed enough that the QP
443    /// visibly scales the output size.
444    fn gradient_frame(w: usize, h: usize, t: usize) -> Vec<u8> {
445        let mut f = vec![0u8; w * h * 4];
446        for y in 0..h {
447            for x in 0..w {
448                let i = (y * w + x) * 4;
449                let v = ((x * 3 + y * 5 + t * 4) & 0xFF) as u8;
450                f[i] = v;
451                f[i + 1] = v.wrapping_add(40);
452                f[i + 2] = v.wrapping_add(80);
453                f[i + 3] = 255;
454            }
455        }
456        f

459    /// A forced first frame is emitted as a typed IDR with a valid wire header and Annex-B
460    /// payload; re-encoding identical content (no scene change) yields a typed delta (P) frame
461    /// whose header carries the passed frame number.
462    #[test]
463    fn encodes_annexb_idr_then_p() {
464        let s = RustCaptureSettings {
465            width: 128,
466            height: 96,
467            video_bitrate_kbps: 2000,
468            target_fps: 30.0,
469            ..Default::default()
470        };
471        let mut enc = Openh264Encoder::new(&s, None).expect("openh264 init");
472        let stride = 128 * 4;
473        let idr = enc.encode_host_argb(&busy_frame(128, 96, 0), stride, 0, true, false).expect("encode idr");
474        assert!(idr.len() > 10, "IDR frame should produce output");
475        assert_eq!(idr[0], 0x04);
476        assert_eq!(idr[1], 0x01, "forced first frame must be typed IDR");
477        assert_eq!(&idr[2..6], &[0, 0, 0, 0], "frame_id 0 and y_start 0");

687    /// Diagnostic that the CBR rate controller responds to its target on incompressible
688    /// noise, and that QP pinning works.
689    ///
690    /// Runs 60 noise frames each at CBR 4 Mbps / 500 kbps and at CRF QP51 / QP25, printing
691    /// bytes-per-frame and IDR counts, then asserts two invariants:
692    ///
693    /// 1. **CBR responds to its target**: 500 kbps must emit far fewer bytes/frame than 4 Mbps.
694    ///    This guards the explicit QP range set in `new` — a zero anywhere in the configured QP pair
695    ///    makes OpenH264 substitute its `[26, 35]` screen defaults, under which noise emits the same
696    ///    bytes at any target.
697    /// 2. **QP pinning**: QP51 must compress far harder than QP25.
698    ///
699    /// The RC still cannot reach the QP51 floor on this content: screen mode force-enables
700    /// scene-change detection (so every noise frame becomes an IDR) and `RcCalculateIdrQp` clamps
701    /// IDR QP to `<= 40` regardless of the session max — an upstream RC ceiling, not a config
702    /// defect.
703    #[test]
704    fn cbr_rc_diagnostic_on_noise() {
705        let (w, h) = (1280usize, 720usize);
706        let stride = w * 4;
707        let run = |cbr: bool, kbps: i32, crf: i32| -> (usize, usize) {
708            let s = RustCaptureSettings {
709                width: w as i32,
710                height: h as i32,
711                video_cbr_mode: cbr,
712                video_bitrate_kbps: kbps,
713                video_crf: crf,
714                target_fps: 60.0,
715                ..Default::default()
716            };
717            let mut e = Openh264Encoder::new(&s, None).unwrap();
718            let mut total = 0usize;
719            let mut idrs = 0usize;
720            for t in 0..60u64 {
721                let out = e
722                    .encode_host_argb(&noise_frame(w, h, t as u32), stride, t, t == 0, false)
723                    .unwrap();
724                total += out.len();
725                for i in 0..out.len().saturating_sub(4) {
726                    if out[i] == 0 && out[i + 1] == 0 && out[i + 2] == 1 && (out[i + 3] & 0x1F) == 5 {
727                        idrs += 1;
728                        break;
729                    }
730                }
731            }
732            (total / 60, idrs)
733        };
734        let (cbr4m, idr4m) = run(true, 4000, 25);
735        let (cbr500k, _) = run(true, 500, 25);
736        let (qp51, _) = run(false, 1, 51);
737        let (qp25, _) = run(false, 1, 25);
738        println!(
739            "noise 720p60 bytes/frame: CBR-4Mbps={cbr4m} (IDR frames {idr4m}/60) \
740             CBR-500kbps={cbr500k} QP51-floor={qp51} QP25={qp25}"
741        );
742        assert!(
743            cbr500k * 4 < cbr4m * 3,
744            "CBR-on-noise no longer responds to its target: 500kbps={cbr500k} B/f vs 4Mbps={cbr4m} B/f \
745             (QP51 floor {qp51}, QP25 {qp25})"
746        );
747        assert!(qp51 * 4 < qp25, "QP pinning sanity: 51 must compress far harder than 25");
748    }
