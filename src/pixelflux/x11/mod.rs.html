7//! X11 host capture: grab the root window into host memory as BGRA, composite the XFixes hardware
8//! cursor and the watermark on the CPU, and feed each frame to [`pipeline::X11Pipeline`], which owns
9//! damage/stripe/encode. The grab goes through a shared-memory segment (XShm via x11rb) rather than
10//! a plain `GetImage` for one reason: a full-screen frame is far too large to copy through the X
11//! protocol socket every tick, so XShm has the server write the pixels straight into memory this
12//! process already has mapped.
13//!
14//! `run_capture` splits the work across two threads because grabbing the next frame and encoding
15//! the previous one have no reason to wait on each other: the caller's thread grabs frames and owns
16//! the x11rb connection and the pool of shm surfaces, while a spawned encode thread owns the
17//! [`pipeline::X11Pipeline`] (and thus the encoder) and runs the delivery callback, so the two overlap for
18//! throughput. Frames hand off through a bounded `FramePool` that carries only a raw pointer and
19//! geometry — never an X object, none of which is safe to share — so nothing X-related ever crosses
20//! the thread boundary and the encoder never has to touch X. Multi-instance safety for the encoders
21//! is handled inside them (e.g. the libx264 open/close lock); each capture owns its own private xcb
22//! connection, so there is no shared X state to serialize here.
23

44/// Cross-thread controls for a running capture: a bag of atomics (plus two mutex-guarded
45/// payloads) the owning `ScreenCapture` pyclass flips from the Python thread and the capture thread
46/// reads at the top of each iteration.
47///
48/// It exists so Python never has to reach into the pipeline. The `X11Pipeline` and its encoder are
49/// only safe to touch from the capture and encode threads, so `request_idr` / rate / fps / region
50/// changes are posted here as atomic flags and applied later on the thread that owns the pipeline,
51/// rather than mutating encoder state across the thread boundary from Python.
52///
53/// 1. **Lifecycle**: `stop` ends the capture loop; `force_idr` requests an on-demand keyframe on the
54///    next processed frame.
55/// 2. **Rate control** (gated by `rate_dirty`): `bitrate_kbps`, `vbv_mult_milli` (the VBV frame-time
56///    multiplier * 1000, held as an integer for atomics; `<= 0` selects the policy default), and
57///    `fps_milli` (target fps * 1000, re-read every frame for dynamic pacing and rate control). These
58///    atomics always hold the CURRENT values, so a pipeline rebuild can carry live rates forward.
59/// 3. **Tunables** (gated by `tunables_dirty`): one `LiveTunables` struct behind `tunables`, set
60///    rarely, carrying the per-frame quality knobs for the encode thread.
61/// 4. **Capture geometry**: `capture_cursor` toggles the cursor overlay (read on every grab);
62///    `region_dirty` guards `region` = `(x, y, w, h)` (w/h `<= 0` extends to the root edge), applied
63///    by the capture thread with the same drain/recreate machinery as auto-adjust.
64pub struct Controls {
65    pub stop: AtomicBool,
66    pub force_idr: AtomicBool,
67    pub rate_dirty: AtomicBool,
68    pub bitrate_kbps: AtomicI32,
69    pub vbv_mult_milli: AtomicI32,
70    pub fps_milli: AtomicU64,
71    pub tunables_dirty: AtomicBool,
72    pub tunables: Mutex<Option<crate::LiveTunables>>,
73    pub capture_cursor: AtomicBool,
74    pub region_dirty: AtomicBool,
75    pub region: Mutex<(i32, i32, i32, i32)>,
76}
77
78impl Controls {
79    /// Seed the controls from the initial settings so the first loop iteration reads the
80    /// configured bitrate / VBV / fps / region / cursor state rather than defaults.
81    pub fn new(s: &RustCaptureSettings) -> Self {
82        Self {
83            stop: AtomicBool::new(false),
84            force_idr: AtomicBool::new(false),
85            rate_dirty: AtomicBool::new(false),
86            bitrate_kbps: AtomicI32::new(s.video_bitrate_kbps),
87            vbv_mult_milli: AtomicI32::new((s.video_vbv_multiplier * 1000.0).round() as i32),
88            fps_milli: AtomicU64::new((s.target_fps.max(1.0) * 1000.0) as u64),
89            tunables_dirty: AtomicBool::new(false),
90            tunables: Mutex::new(None),
91            capture_cursor: AtomicBool::new(s.capture_cursor),
92            region_dirty: AtomicBool::new(false),
93            region: Mutex::new((s.capture_x, s.capture_y, s.width, s.height)),
94        }
95    }
96}

98/// A shared-memory image surface: a POSIX shm segment mapped into both this process and the
99/// X server, into which `shm_get_image` writes one BGRA frame.
100///
101/// Both ends map the same physical pages, which is the whole point: a full-screen grab then costs no
102/// per-frame copy across the X socket, because the server's blit lands directly in memory the
103/// capture thread already reads. The surface owns the server-side segment id (`shmseg`), the local
104/// mapping (`addr` / `size`), and the geometry (`width` / `height` / `stride`) needed to interpret
105/// the bytes.
106struct ShmSurface {
107    shmseg: u32,
108    addr: *mut u8,
109    size: usize,
110    width: u16,
111    height: u16,
112    stride: usize,
113}
114
115impl ShmSurface {
116    /// Allocate a shm segment of `width*height*4` bytes, attach it both locally and to the X
117    /// server, and hand back a surface both ends can read.
118    ///
119    /// The segment is created with `shmget(IPC_PRIVATE)` and mapped locally via `shmat`, then
120    /// attached on the server (`shm_attach`). That attach is confirmed with a round-trip `check()`
121    /// BEFORE the segment is marked `IPC_RMID`: deleting only after both ends hold a reference means
122    /// the kernel frees the segment once this process and the server have both detached, never while
123    /// either still needs it. Every failure path unwinds the partial state (local detach and/or
124    /// `IPC_RMID`) so a failed allocation leaks neither memory nor a server attachment; a zero-sized
125    /// request is rejected up front.
126    fn create(conn: &RustConnection, width: u16, height: u16) -> Result<Self, String> {
127        let stride = width as usize * 4;
128        let size = stride * height as usize;
129        if size == 0 {
130            return Err("zero-sized capture surface".into());
131        }
132        unsafe {
133            let shmid = libc::shmget(libc::IPC_PRIVATE, size, libc::IPC_CREAT | 0o600);
134            if shmid < 0 {
135                return Err("shmget failed".into());
136            }
137            let addr = libc::shmat(shmid, std::ptr::null(), 0);
138            if addr == (-1isize) as *mut libc::c_void {
139                libc::shmctl(shmid, libc::IPC_RMID, std::ptr::null_mut());
140                return Err("shmat failed".into());
141            }
142            let shmseg = match conn.generate_id() {
143                Ok(id) => id,
144                Err(e) => {
145                    libc::shmdt(addr);
146                    libc::shmctl(shmid, libc::IPC_RMID, std::ptr::null_mut());
147                    return Err(format!("generate_id: {e}"));
148                }
149            };
150            let attach = conn
151                .shm_attach(shmseg, shmid as u32, false)
152                .map_err(|e| format!("shm_attach: {e}"))
153                .and_then(|c| c.check().map_err(|e| format!("shm_attach check: {e}")));
154            libc::shmctl(shmid, libc::IPC_RMID, std::ptr::null_mut());
155            if let Err(e) = attach {
156                libc::shmdt(addr);
157                return Err(e);
158            }
159            Ok(Self {
160                shmseg,
161                addr: addr as *mut u8,
162                size,
163                width,
164                height,
165                stride,
166            })
167        }
168    }
169
170    /// Borrow the mapped segment as a mutable byte slice for the capture blit and overlays.
171    fn as_mut_slice(&mut self) -> &mut [u8] {
172        unsafe { std::slice::from_raw_parts_mut(self.addr, self.size) }
173    }
174
175    /// Borrow the mapped segment read-only, for the stability comparison.
176    fn as_slice(&self) -> &[u8] {
177        unsafe { std::slice::from_raw_parts(self.addr, self.size) }
178    }
179
180    /// Detach the segment from the X server and then locally, clearing the mapped pointer.
181    ///
182    /// This is a manual method rather than a `Drop` impl because releasing the server-side
183    /// attachment is an X protocol request that needs the live connection, which a `Drop` could not
184    /// reach; the owner of the connection must call it before the surface is dropped. The order —
185    /// server `shm_detach` first, then the local `shmdt` — completes the release of the segment that
186    /// was marked `IPC_RMID` at creation, so the kernel reclaims it once neither end holds it.
187    fn destroy(&mut self, conn: &RustConnection) {
188        let _ = conn.shm_detach(self.shmseg);
189        let _ = conn.flush();
190        unsafe {
191            libc::shmdt(self.addr as *mut libc::c_void);
192        }
193        self.addr = std::ptr::null_mut();
194    }
195}

197/// Resolve the capture dimensions `shm_get_image` will read, bounded so the region can never
198/// run past the live root from the capture offset.
199///
200/// The bound is the reason this function exists: a grab that ran off the root edge would fail the
201/// request outright, so the size is always clamped to what is actually there. With auto-adjust (or
202/// an unset width/height `<= 0`) the capture tracks the full root minus the capture offset;
203/// otherwise the requested size is clamped to what is available from that offset. `capture_x` /
204/// `capture_y` are floored at `>= 0` exactly as `shm_get_image` consumes them, and saturating
205/// subtraction plus a final `u16` clamp (minimum 2) keep pathological settings from overflowing or
206/// collapsing to an unusable surface. H.264 (`output_mode == 1`) requires even dimensions, so both
207/// are then rounded down to a multiple of two.
208fn resolve_dims(root_w: u16, root_h: u16, s: &RustCaptureSettings) -> (u16, u16) {
209    let cap_x = s.capture_x.max(0);
210    let cap_y = s.capture_y.max(0);
211    let avail_w = (root_w as i32).saturating_sub(cap_x).max(2);
212    let avail_h = (root_h as i32).saturating_sub(cap_y).max(2);
213    let mut w = if s.auto_adjust_screen_capture_size || s.width <= 0 {
214        avail_w
215    } else {
216        s.width.min(avail_w)
217    };
218    let mut h = if s.auto_adjust_screen_capture_size || s.height <= 0 {
219        avail_h
220    } else {
221        s.height.min(avail_h)
222    };
223    w = w.clamp(2, u16::MAX as i32);
224    h = h.clamp(2, u16::MAX as i32);
225    if s.output_mode == 1 {
226        w &= !1;
227        h &= !1;
228    }
229    (w as u16, h as u16)
230}

232/// Grab one frame of the capture region into `surface` with a single XShm round-trip.
233///
234/// One `shm_get_image` request is atomic — the server never interleaves another client
235/// inside it — so the grab is a coherent snapshot of whatever the screen held at that
236/// instant. Every grabbed frame is published; whether any of it changed is decided
237/// downstream by the encoder's own per-stripe damage detection.
238fn grab_frame(
239    conn: &RustConnection,
240    root: u32,
241    surface: &ShmSurface,
242    cap_x: i16,
243    cap_y: i16,
244    cap_w: u16,
245    cap_h: u16,
246) -> Result<(), String> {
247    conn.shm_get_image(
248        root,
249        cap_x,
250        cap_y,
251        cap_w,
252        cap_h,
253        !0u32,
254        ImageFormat::Z_PIXMAP.into(),
255        surface.shmseg,
256        0,
257    )
258    .map_err(|e| format!("shm_get_image: {e}"))?
259    .reply()
260    .map_err(|e| format!("shm_get_image reply: {e}"))?;
261    Ok(())
262}

264/// Alpha-blend a source pixel (pre-split into r,g,b,a) over a BGRA destination pixel.
265///
266/// Cursor and watermark pixels are overwhelmingly either fully opaque or fully transparent, and this
267/// runs per pixel per frame on the CPU, so the two extremes are special-cased to skip the blend
268/// arithmetic entirely: an opaque source (`a == 255`) simply overwrites, a fully transparent source
269/// (`a == 0`) is left as-is, and only genuine edge pixels pay for the integer source-over. In every
270/// case only the B / G / R bytes are written; the destination's alpha byte is left as the capture
271/// delivered it.
272#[inline]
273fn blend_pixel(dst: &mut [u8], r: u8, g: u8, b: u8, a: u8) {
274    if a == 255 {
275        dst[0] = b;
276        dst[1] = g;
277        dst[2] = r;
278    } else if a > 0 {
279        let ia = 255 - a as u32;
280        dst[0] = ((b as u32 * a as u32 + dst[0] as u32 * ia) / 255) as u8;
281        dst[1] = ((g as u32 * a as u32 + dst[1] as u32 * ia) / 255) as u8;
282        dst[2] = ((r as u32 * a as u32 + dst[2] as u32 * ia) / 255) as u8;
283    }
284}

286/// Frame-space top-left of the cursor image, given the XFixes hotspot position.
287///
288/// XFixes reports the cursor position at its HOTSPOT; X draws the image with its top-left at
289/// `(pos - hot)`, and the capture origin `(cap_x, cap_y)` is subtracted to move it into frame space.
290/// The result may go negative near the frame edges, which `overlay_cursor` clips per pixel to match
291/// the server's own edge clipping.
292#[inline]
293fn cursor_image_origin(x: i16, y: i16, xhot: u16, yhot: u16, cap_x: i32, cap_y: i32) -> (i32, i32) {
294    (x as i32 - xhot as i32 - cap_x, y as i32 - yhot as i32 - cap_y)
295}

297/// Composite the XFixes cursor (ARGB `u32` per pixel) onto the BGRA frame with its top-left
298/// at `(img_x, img_y)`, blending each pixel through `blend_pixel` with per-pixel bounds clipping so
299/// an image straddling a frame edge writes only its in-frame portion.
300#[allow(clippy::too_many_arguments)]
301fn overlay_cursor(
302    frame: &mut [u8],
303    stride: usize,
304    frame_w: i32,
305    frame_h: i32,
306    cur_w: i32,
307    cur_h: i32,
308    pixels: &[u32],
309    img_x: i32,
310    img_y: i32,
311) {
312    for y in 0..cur_h {
313        let ty = img_y + y;
314        if ty < 0 || ty >= frame_h {
315            continue;
316        }
317        for x in 0..cur_w {
318            let tx = img_x + x;
319            if tx < 0 || tx >= frame_w {
320                continue;
321            }
322            let px = pixels[(y * cur_w + x) as usize];
323            let a = ((px >> 24) & 0xFF) as u8;
324            let r = ((px >> 16) & 0xFF) as u8;
325            let g = ((px >> 8) & 0xFF) as u8;
326            let b = (px & 0xFF) as u8;
327            let off = ty as usize * stride + tx as usize * 4;
328            blend_pixel(&mut frame[off..off + 4], r, g, b, a);
329        }
330    }
331}

333/// CPU watermark: the raw RGBA pixels (row-major, `w*h*4`) in host memory plus the
334/// placement / animation state, blended directly into the captured BGRA frame.
335///
336/// The blend is on the CPU because the frame is already sitting in host shm memory before it reaches
337/// any encoder, so stamping the overlay there is both the cheapest place to do it and the one place
338/// it applies identically regardless of which encoder (hardware or software) runs downstream. The
339/// pixels are kept as RGBA straight from `image`'s decode and converted per pixel as they are blended.
340struct X11Watermark {
341    pixels: Vec<u8>,
342    w: i32,
343    h: i32,
344    pos_x: i32,
345    pos_y: i32,
346    sub_x: f64,
347    sub_y: f64,
348    vel_x: f64,
349    vel_y: f64,
350    loaded: bool,
351}
352
353impl X11Watermark {
354    /// Load the watermark image at `path` into host RGBA, or return an unloaded stub.
355    ///
356    /// An empty path or any decode failure yields `loaded == false`, which every method treats as a
357    /// no-op, so a missing or broken watermark simply disables the overlay. The bouncing-animation
358    /// velocities are seeded here.
359    fn load(path: &str) -> Self {
360        let mut wm = Self {
361            pixels: Vec::new(),
362            w: 0,
363            h: 0,
364            pos_x: 0,
365            pos_y: 0,
366            sub_x: 0.0,
367            sub_y: 0.0,
368            vel_x: 2.0,
369            vel_y: 2.0,
370            loaded: false,
371        };
372        if path.is_empty() {
373            return wm;
374        }
375        if let Ok(img) = image::open(std::path::Path::new(path)) {
376            let rgba = img.to_rgba8();
377            wm.w = rgba.width() as i32;
378            wm.h = rgba.height() as i32;
379            wm.pixels = rgba.into_vec();
380            wm.loaded = wm.w > 0 && wm.h > 0;
381        }
382        wm
383    }
384
385    /// Set the watermark's top-left placement for this frame from the location enum; a stub
386    /// (unloaded) watermark returns immediately.
387    ///
388    /// The fixed corners and center are direct arithmetic. The animated mode advances a bouncing
389    /// position and reflects velocity off each frame edge, accumulating in the fractional `sub_x` /
390    /// `sub_y` rather than the integer `pos_x` / `pos_y`: a sub-pixel-per-frame velocity has to
391    /// survive between frames or the motion would either stall or snap by whole pixels, so the float
392    /// carries the remainder and `pos_x` / `pos_y` take its floor for the actual blit.
393    fn update_position(&mut self, frame_w: i32, frame_h: i32, loc_enum: i32) {
394        if !self.loaded {
395            return;
396        }
397        match WatermarkLocation::from(loc_enum) {
398            WatermarkLocation::TL => {
399                self.pos_x = 0;
400                self.pos_y = 0;
401            }
402            WatermarkLocation::TR => {
403                self.pos_x = frame_w - self.w;
404                self.pos_y = 0;
405            }
406            WatermarkLocation::BL => {
407                self.pos_x = 0;
408                self.pos_y = frame_h - self.h;
409            }
410            WatermarkLocation::BR => {
411                self.pos_x = frame_w - self.w;
412                self.pos_y = frame_h - self.h;
413            }
414            WatermarkLocation::MI => {
415                self.pos_x = (frame_w - self.w) / 2;
416                self.pos_y = (frame_h - self.h) / 2;
417            }
418            WatermarkLocation::AN => {
419                self.sub_x += self.vel_x;
420                self.sub_y += self.vel_y;
421                if self.sub_x <= 0.0 {
422                    self.sub_x = 0.0;
423                    self.vel_x = self.vel_x.abs();
424                } else if self.sub_x + self.w as f64 >= frame_w as f64 {
425                    self.sub_x = (frame_w - self.w) as f64;
426                    self.vel_x = -self.vel_x.abs();
427                }
428                if self.sub_y <= 0.0 {
429                    self.sub_y = 0.0;
430                    self.vel_y = self.vel_y.abs();
431                } else if self.sub_y + self.h as f64 >= frame_h as f64 {
432                    self.sub_y = (frame_h - self.h) as f64;
433                    self.vel_y = -self.vel_y.abs();
434                }
435                self.pos_x = self.sub_x as i32;
436                self.pos_y = self.sub_y as i32;
437            }
438            WatermarkLocation::None => (),
439        }
440    }
441
442    /// Alpha-blend the loaded watermark into the BGRA frame at its current position; a no-op
443    /// when unloaded. Clips per pixel at the frame bounds because the animated position — or a
444    /// watermark larger than the capture — can leave part of the image off-frame, and only the
445    /// in-frame portion may be written.
446    fn blend_into(&self, frame: &mut [u8], stride: usize, frame_w: i32, frame_h: i32) {
447        if !self.loaded {
448            return;
449        }
450        for y in 0..self.h {
451            let ty = self.pos_y + y;
452            if ty < 0 || ty >= frame_h {
453                continue;
454            }
455            for x in 0..self.w {
456                let tx = self.pos_x + x;
457                if tx < 0 || tx >= frame_w {
458                    continue;
459                }
460                let src = ((y * self.w + x) * 4) as usize;
461                let (r, g, b, a) = (
462                    self.pixels[src],
463                    self.pixels[src + 1],
464                    self.pixels[src + 2],
465                    self.pixels[src + 3],
466                );
467                let off = ty as usize * stride + tx as usize * 4;
468                blend_pixel(&mut frame[off..off + 4], r, g, b, a);
469            }
470        }
471    }
472}

474/// A captured raw BGRA frame held in a pooled shm surface, ready to encode.
475///
476/// It carries the surface pointer plus geometry so the encode thread reads the pixels directly (no
477/// copy) and can rebuild its pipeline when the capture size changed (auto-adjust). It also carries
478/// the pool's surface `generation`, so a rebuild is still triggered when the surfaces were recreated
479/// at the SAME size (a resize flap) and the new segments happen to reuse the old virtual addresses.
480struct RawFrame {
481    idx: usize,
482    ptr: *mut u8,
483    len: usize,
484    width: u16,
485    height: u16,
486    stride: usize,
487    generation: u64,
488}
489/// `RawFrame` is `Send`: its raw pointer addresses a pooled shm surface that the pool
490/// guarantees is not reused until the encode thread recycles this frame, so the handle is safe to
491/// move across the capture -> encode thread boundary.
492unsafe impl Send for RawFrame {}
493
494/// Mutex-guarded interior of `FramePool`: the free-surface index list and the single
495/// capture -> encode handoff slot, under one lock so moving a surface between them — acquire,
496/// publish, take, recycle — is always a single atomic step.
497struct PoolInner {
498    free: Vec<usize>,
499    slot: Option<RawFrame>,
500}
501
502/// Demand-driven capture -> encode handoff: a bounded single-slot channel over a fixed set of
503/// pooled shm surfaces.
504///
505/// The capture thread writes into a pooled surface and `publish`es it into the single `slot`; the
506/// encode thread `take`s it. Capture stays at most one frame ahead of encode because `acquire` and
507/// `publish` BLOCK (bounded) until the encoder frees a surface / drains the slot, throttling capture
508/// to the encode rate. Since X11 capture is pull-based (no backlog), throttling — rather than
509/// capturing-then-dropping — means a full-resolution shm round-trip is never spent on a frame that
510/// would be discarded, while the next capture still overlaps the current encode (the throughput win).
511/// The encoder therefore only ever sees a contiguous frame stream, keeping the H.264 reference chain
512/// valid.
513///
514/// `generation` is bumped by the capture thread each time the backing shm surfaces are destroyed and
515/// recreated. Published frames carry it so the encode thread rebuilds its pipeline — dropping encoder
516/// state keyed to surface base pointers (e.g. NVENC's pinned-host registrations) — even when a resize
517/// flap lands back on the old dimensions and the recreated segments reuse the old virtual addresses.
518struct FramePool {
519    inner: Mutex<PoolInner>,
520    cv: Condvar,
521    stop: AtomicBool,
522    generation: AtomicU64,
523}
524
525impl FramePool {
526    /// Create a pool with `n` free surfaces, an empty handoff slot, and generation 0.
527    fn new(n: usize) -> Self {
528        Self {
529            inner: Mutex::new(PoolInner { free: (0..n).collect(), slot: None }),
530            cv: Condvar::new(),
531            stop: AtomicBool::new(false),
532            generation: AtomicU64::new(0),
533        }
534    }
535
536    /// Capture: record that the surfaces were recreated by advancing the generation. Only
537    /// called after `drain_for_resize` succeeded, so no frame from the previous generation is still
538    /// in flight.
539    fn bump_generation(&self) {
540        self.generation.fetch_add(1, Ordering::Relaxed);
541    }
542
543    /// Read the current surface generation, stamped onto each published frame.
544    fn generation(&self) -> u64 {
545        self.generation.load(Ordering::Relaxed)
546    }
547
548    /// Capture: claim a free surface to write the next frame into, or `None` on stop.
549    ///
550    /// Blocks (bounded, re-checking `stop` every 20ms) until a surface is free, which throttles
551    /// capture to the encode rate so a full-resolution capture is never spent on a frame that would
552    /// be dropped. The bounded re-check means a stop that races the wait cannot hang the thread.
553    fn acquire(&self, stop: &AtomicBool) -> Option<usize> {
554        let mut g = self.inner.lock().unwrap();
555        loop {
556            if let Some(idx) = g.free.pop() {
557                return Some(idx);
558            }
559            if stop.load(Ordering::Relaxed) {
560                return None;
561            }
562            let (gg, _) = self.cv.wait_timeout(g, Duration::from_millis(20)).unwrap();
563            g = gg;
564        }
565    }
566
567    /// Capture: publish the just-captured frame into the single slot, or `false` (frame
568    /// discarded) on stop.
569    ///
570    /// Blocks (bounded, re-checking `stop` every 20ms) until the encode thread has taken the previous
571    /// frame, so capture stays at most one frame ahead and never drops under normal flow.
572    fn publish(&self, frame: RawFrame, stop: &AtomicBool) -> bool {
573        let mut g = self.inner.lock().unwrap();
574        loop {
575            if g.slot.is_none() {
576                g.slot = Some(frame);
577                drop(g);
578                self.cv.notify_all();
579                return true;
580            }
581            if stop.load(Ordering::Relaxed) {
582                return false;
583            }
584            let (gg, _) = self.cv.wait_timeout(g, Duration::from_millis(20)).unwrap();
585            g = gg;
586        }
587    }
588
589    /// Encode: block until a frame is available (`Some`) or stop is signalled (`None`).
590    ///
591    /// The wait is bounded (re-checking `stop` every 20ms) as defense-in-depth against a lost wakeup,
592    /// so a stop that races the park can never leave the encode thread blocked forever.
593    fn take(&self) -> Option<RawFrame> {
594        let mut g = self.inner.lock().unwrap();
595        loop {
596            if let Some(f) = g.slot.take() {
597                return Some(f);
598            }
599            if self.stop.load(Ordering::Acquire) {
600                return None;
601            }
602            let (gg, _) = self.cv.wait_timeout(g, Duration::from_millis(20)).unwrap();
603            g = gg;
604        }
605    }
606
607    /// Encode: return a surface to the free list after it has been encoded, waking any waiter
608    /// parked in `acquire` / `publish` / `drain_for_resize`.
609    fn recycle(&self, idx: usize) {
610        self.inner.lock().unwrap().free.push(idx);
611        self.cv.notify_all();
612    }
613
614    /// Capture: before recreating surfaces (auto-adjust / region resize), reclaim the pending
615    /// slot and wait until every surface is back in the free list, so no surface is destroyed while
616    /// the encode thread is still reading it.
617    ///
618    /// Reclaiming the slot returns any un-taken frame to the free list; the loop then waits until
619    /// `free.len() == n`, i.e. the encode thread has finished any in-flight frame. The wait is bounded
620    /// and also breaks on stop — pool shutdown OR the external `stop` — so a panicked or dead encode
621    /// thread that never recycles cannot wedge the capture thread here forever; a requested stop
622    /// unblocks it. Returns `true` if it fully drained (safe to recreate surfaces), or `false` if it
623    /// aborted on stop — in which case the caller tears down, which joins the encode thread before
624    /// destroying surfaces, so the resize-safety guarantee still holds.
625    fn drain_for_resize(&self, n: usize, stop: &AtomicBool) -> bool {
626        let mut g = self.inner.lock().unwrap();
627        if let Some(old) = g.slot.take() {
628            g.free.push(old.idx);
629        }
630        while g.free.len() < n {
631            if self.stop.load(Ordering::Acquire) || stop.load(Ordering::Relaxed) {
632                return false;
633            }
634            let (gg, _) = self.cv.wait_timeout(g, Duration::from_millis(20)).unwrap();
635            g = gg;
636        }
637        true
638    }
639
640    /// Signal every waiter to stop and wake them, ending the encode thread's `take` loop.
641    ///
642    /// The inner mutex is acquired BEFORE storing `stop` and notifying, which closes the lost-wakeup
643    /// window: with the lock held, `take` is either still before its `stop` check or already parked on
644    /// the condvar (and will receive the notify), never in the gap between the two. The notify is
645    /// issued after the guard is dropped so the woken thread does not immediately re-block on the lock
646    /// this call holds.
647    fn shutdown(&self) {
648        let g = self.inner.lock().unwrap();
649        self.stop.store(true, Ordering::Release);
650        drop(g);
651        self.cv.notify_all();
652    }
653}
654
655/// Encode thread body: consume captured frames from the pool, keep the [`pipeline::X11Pipeline`] in
656/// step with the capture size and cross-thread controls, encode, recycle, and deliver.
657///
658/// The loop runs until `FramePool::take` returns `None` (stop). For each frame:
659///
660/// 1. **(Re)build the pipeline** when it is missing, the frame size changed, or the surface
661///    generation changed. A generation change alone forces the rebuild path because recreated shm
662///    segments often reuse the old virtual base addresses, so encoder state keyed to base pointers
663///    (NVENC's pinned-host cache) must be dropped even at identical dimensions. An in-place
664///    [`X11Pipeline::reshape`] is preferred — NVENC reconfigures its live session and the striped
665///    software path just re-derives stripe state — so a resize never stalls on a full encoder
666///    re-init; encoders that cannot follow in place (VAAPI/OpenH264) report `false` and the pipeline
667///    is rebuilt. On a rebuild the old pipeline (with its GPU session/surfaces) is dropped BEFORE the
668///    new one is built, so an auto-adjust resize never holds two full encoder allocations at once
669///    (transient 2x GPU memory). The rebuilt settings pull the CURRENT bitrate / VBV / fps from the
670///    controls atomics, carrying live rate changes forward instead of reverting to the capture-start
671///    values.
672/// 2. **Apply cross-thread controls** here, on the thread that owns the pipeline: a pending
673///    `force_idr` requests a keyframe; a `rate_dirty` swap (Acquire, pairing with the setters' Release
674///    stores so the bitrate / VBV / fps payload is never seen half-applied) pushes a live rate change;
675///    a `tunables_dirty` swap applies per-frame tunables into both the local settings copy and the
676///    pipeline.
677/// 3. **Encode and hand off**: read the pooled surface directly through its raw pointer — sound
678///    because the pool guarantees the surface is not reused until it is recycled — run
679///    [`X11Pipeline::process`], recycle the surface BEFORE delivering so a slow consumer never holds a
680///    capture surface, then deliver any encoded stripes through `on_frame`.
681///
682/// The optional Unix-socket recording sink (parity with the Wayland path) is bound ONCE here and
683/// owned outside the pipeline, so pipeline rebuilds on resize keep the socket listener and any
684/// attached recorders alive. It can only carry a single full-frame H.264 stream, so configurations
685/// that cannot produce one (JPEG output, or a striped CPU encoder) are warned about up front.
686fn encode_loop<F>(pool: &FramePool, controls: &Controls, settings: &RustCaptureSettings, on_frame: &mut F)
687where
688    F: FnMut(Vec<EncodedStripe>),
689{
690    let mut psettings = settings.clone();
691    let recording_sink = RecordingSink::try_bind(&settings.recording_socket);
692    if recording_sink.is_some() {
693        if settings.output_mode == 0 {
694            eprintln!("[recording_sink] recording_socket set but output_mode is JPEG; no recordable H.264 stream.");
695        } else if settings.use_cpu && !settings.use_openh264 && !settings.video_fullframe {
696            eprintln!("[recording_sink] recording_socket set but the CPU encoder is striped; set video_fullframe=true for a recordable stream.");
697        }
698    }
699    let mut pipeline: Option<X11Pipeline> = None;
700    let (mut pw, mut ph) = (0i32, 0i32);
701    let mut pgen = 0u64;
702    let mut last_log_time = Instant::now();
703    let mut frame_count: u64 = 0;
704    let mut stripe_count: u64 = 0;
705
706    while let Some(frame) = pool.take() {
707        let (fw, fh) = (frame.width as i32, frame.height as i32);
708        if pipeline.is_none() || pw != fw || ph != fh || pgen != frame.generation {
709            let size_changed = pw != fw || ph != fh;
710            psettings.width = fw;
711            psettings.height = fh;
712            psettings.video_bitrate_kbps = controls.bitrate_kbps.load(Ordering::Relaxed);
713            psettings.video_vbv_multiplier =
714                controls.vbv_mult_milli.load(Ordering::Relaxed) as f64 / 1000.0;
715            psettings.target_fps =
716                (controls.fps_milli.load(Ordering::Relaxed).max(1) as f64) / 1000.0;
717            let reshaped = pipeline
718                .as_mut()
719                .is_some_and(|pl| pl.reshape(&psettings, size_changed));
720            if !reshaped {
721                drop(pipeline.take());
722                pipeline = Some(X11Pipeline::new(psettings.clone(), recording_sink.clone()));
723                if let Some(pl) = &pipeline {
724                    let enc_name = pl.encoder_name();
725                    let mut log_msg = format!(
726                        "[x11] Stream settings active -> Res: {}x{} | FPS: {:.1} | Encoder: {}",
727                        psettings.width, psettings.height, psettings.target_fps, enc_name
728                    );
729                    if psettings.output_mode == 0 {
730                        log_msg.push_str(&format!(" | Mode: JPEG | Quality: {}", psettings.jpeg_quality));
731                    } else {
732                        log_msg.push_str(&format!(" | Mode: H264 | CRF: {}", psettings.video_crf));
733                        if psettings.video_fullcolor {
734                            log_msg.push_str(" | Colorspace: I444 (Full Range)");
735                        } else {
736                            log_msg.push_str(" | Colorspace: I420 (Limited Range)");
737                        }
738                    }
739                    log_msg.push_str(&format!(
740                        " | Damage Thresh: {}f | Damage Dur: {}f",
741                        psettings.damage_block_threshold, psettings.damage_block_duration
742                    ));
743                    println!("{}", log_msg);
744                }
745            }
746            pw = fw;
747            ph = fh;
748            pgen = frame.generation;
749        }
750        let pl = pipeline.as_mut().unwrap();
751
752        if controls.force_idr.swap(false, Ordering::Relaxed) {
753            pl.request_idr();
754        }
755        if controls.rate_dirty.swap(false, Ordering::Acquire) {
756            let b = controls.bitrate_kbps.load(Ordering::Relaxed);
757            let v = controls.vbv_mult_milli.load(Ordering::Relaxed) as f64 / 1000.0;
758            let fps = (controls.fps_milli.load(Ordering::Relaxed).max(1) as f64) / 1000.0;
759            pl.update_rate(b, v, fps);
760        }
761        if controls.tunables_dirty.swap(false, Ordering::Acquire) {
762            if let Some(t) = controls.tunables.lock().unwrap().take() {
763                t.apply_to(&mut psettings);
764                pl.update_tunables(&t);
765            }
766        }
767
768        let buf = unsafe { std::slice::from_raw_parts(frame.ptr, frame.len) };
769        let stripes = pl.process(buf, frame.stride);
770        pool.recycle(frame.idx);
771        if !stripes.is_empty() {
772            frame_count += 1;
773            stripe_count += stripes.len() as u64;
774            on_frame(stripes);
775        }
776
777        let now = Instant::now();
778        let elapsed = now.duration_since(last_log_time).as_secs_f64();
779        if elapsed >= 1.0 {
780            if settings.debug_logging {
781                let actual_fps = frame_count as f64 / elapsed;
782                let stripes_per_sec = stripe_count as f64 / elapsed;
783                println!(
784                    "[x11] Res: {}x{} Encoder: {} EncFPS: {:.2} EncStripes/s: {:.2}",
785                    psettings.width, psettings.height, pl.encoder_name(), actual_fps, stripes_per_sec
786                );
787            }
788            frame_count = 0;
789            stripe_count = 0;
790            last_log_time = now;
791        }
792    }
793}

795/// Run the X11 capture pipeline until `stop` is set, splitting capture and encode across two
796/// threads that overlap for throughput.
797///
798/// This (the caller's) thread performs setup and then the capture loop; a spawned encode thread runs
799/// `encode_loop` and invokes `on_frame(stripes)` once per encoded frame. The split lets capture and
800/// encode overlap, and because frames are throttled/dropped as RAW frames before encode, the delivered
801/// H.264 stays a valid contiguous reference chain.
802///
803/// 1. **Setup**: connect to X (a private connection) and require a 32-bpp BGRA root — the Z-pixmap
804///    byte depth must be 4, which modern servers use for depth 24/32 — then negotiate XShm and
805///    XFixes. XFixes is always negotiated (one round-trip) so the cursor overlay can be toggled on
806///    live even when capture started without it. Initial dimensions and origin are resolved from the
807///    settings and the live root geometry, and a `FramePool` of `POOL_N` = 3 shm surfaces is
808///    allocated — the working set (one in-capture, one in-slot, one in-encode) that a one-frame-ahead
809///    demand-driven capture needs, which also keeps the memory cost (`3 * W*H*4`, significant at 4K)
810///    bounded.
811/// 2. **Encode thread**: spawned with a raised scheduling priority; it reports its thread id back
812///    through `encode_tid_tx` so the caller can detect a re-entrant stop issued from inside the
813///    delivery callback.
814/// 3. **Capture loop** (until `stop`): fps is re-read each iteration for live pacing — sleep to the
815///    next frame deadline, or `yield_now` when already behind instead of busy-spinning.
816///    - **Live region change** (`region_dirty`): re-target the grab origin immediately (an x/y pan
817///      needs no surface work); a size change reuses the drain/recreate path below.
818///    - **Auto-adjust**: on a root geometry change, drain in-flight frames then recreate the surfaces
819///      and bump the generation (which the encode thread turns into a reshape or rebuild). The geometry
820///      is re-resolved AFTER the drain, because a slow drain can outlast another geometry change (a
821///      fast flap) and recreating at a stale size would make the next `shm_get_image` exceed the root;
822///      a fully-reverted flap keeps its surfaces as-is. A stop that races the drain breaks out to
823///      teardown.
824///    - **Grab**: acquire a pooled surface (blocks until the encoder frees one), then `shm_get_image`
825///      the region into it synchronously (`reply()` waits).
826///    - **Overlays**: composite the XFixes cursor (drawn at its hotspot-offset origin; live-toggleable)
827///      and the watermark onto the BGRA pixels on the CPU.
828///    - **Publish**: hand the finished frame to the encode thread (blocks until the slot is free;
829///      never drops). A stop observed while waiting in acquire/publish exits the loop.
830/// 4. **Teardown**: stop and JOIN the encode thread BEFORE destroying the shm surfaces it may still be
831///    reading, preserving the resize-safety guarantee.
832///
833/// Blocking; intended to run on a dedicated thread. The X connection and shm surfaces live on this
834/// thread and the encoder lives on the encode thread — nothing X-related crosses the boundary.
835pub fn run_capture<F>(
836    settings: RustCaptureSettings,
837    controls: Arc<Controls>,
838    encode_tid_tx: Sender<thread::ThreadId>,
839    on_frame: F,
840) -> Result<(), String>
841where
842    F: FnMut(Vec<EncodedStripe>) + Send + 'static,
843{
844    let (conn, screen_num) =
845        x11rb::connect(None).map_err(|e| format!("X11 connect failed: {e}"))?;
846    let root = conn.setup().roots[screen_num].root;
847    let root_depth = conn.setup().roots[screen_num].root_depth;
848
849    let bpp = conn
850        .setup()
851        .pixmap_formats
852        .iter()
853        .find(|f| f.depth == root_depth)
854        .map(|f| f.bits_per_pixel)
855        .unwrap_or(32);
856    if bpp != 32 {
857        return Err(format!(
858            "unsupported root depth {root_depth} ({bpp} bpp); only 32-bpp BGRA is supported"
859        ));
860    }
861
862    conn.shm_query_version()
863        .map_err(|e| format!("shm_query_version: {e}"))?
864        .reply()
865        .map_err(|e| format!("XShm unavailable: {e}"))?;
866    conn.xfixes_query_version(5, 0)
867        .map_err(|e| format!("xfixes_query_version: {e}"))?
868        .reply()
869        .map_err(|e| format!("XFixes unavailable: {e}"))?;
870
871    let geo = conn
872        .get_geometry(root)
873        .map_err(|e| format!("get_geometry: {e}"))?
874        .reply()
875        .map_err(|e| format!("get_geometry reply: {e}"))?;
876
877    let mut rsettings = settings.clone();
878    let (mut cap_w, mut cap_h) = resolve_dims(geo.width, geo.height, &rsettings);
879    let mut cap_x = rsettings.capture_x.max(0) as i16;
880    let mut cap_y = rsettings.capture_y.max(0) as i16;
881
882    const POOL_N: usize = 3;
883    let mut surfaces: Vec<ShmSurface> = Vec::with_capacity(POOL_N);
884    for _ in 0..POOL_N {
885        surfaces.push(ShmSurface::create(&conn, cap_w, cap_h)?);
886    }
887    let pool = Arc::new(FramePool::new(POOL_N));
888
889    let mut watermark = X11Watermark::load(&settings.watermark_path);
890
891    let enc_pool = pool.clone();
892    let enc_controls = controls.clone();
893    let enc_settings = settings.clone();
894    let encode_thread = thread::spawn(move || {
895        crate::boost_thread_priority(-10);
896        let _ = encode_tid_tx.send(thread::current().id());
897        let mut on_frame = on_frame;
898        encode_loop(&enc_pool, &enc_controls, &enc_settings, &mut on_frame);
899    });
900
901    let mut next_frame = Instant::now();
902    // Frames between root-geometry polls (auto-adjust); ~0.5s at 60fps.
903    const GEOMETRY_POLL_FRAMES: i32 = 30;
904    let mut geometry_check = 0i32;
905    let mut grab_failures = 0u32;
906
907    let result = (|| -> Result<(), String> {
908        while !controls.stop.load(Ordering::Relaxed) {
909            let fps = (controls.fps_milli.load(Ordering::Relaxed).max(1) as f64) / 1000.0;
910            let frame_dur = Duration::from_secs_f64(1.0 / fps.max(1.0));
911            let now = Instant::now();
912            if now < next_frame {
913                std::thread::sleep(next_frame - now);
914            } else {
915                std::thread::yield_now();
916            }
917            next_frame += frame_dur;
918            let now = Instant::now();
919            if next_frame < now {
920                next_frame = now;
921            }
922            if controls.stop.load(Ordering::Relaxed) {
923                break;
924            }
925
926            if controls.region_dirty.swap(false, Ordering::Acquire) {
927                let (nx, ny, nw, nh) = *controls.region.lock().unwrap();
928                rsettings.capture_x = nx;
929                rsettings.capture_y = ny;
930                rsettings.width = nw;
931                rsettings.height = nh;
932                // An explicit live size pins the region: auto-adjust (from the start
933                // settings) would glue resolve_dims back to the ROOT size, undoing the
934                // re-target on the next geometry poll. w/h <= 0 keep following the root.
935                rsettings.auto_adjust_screen_capture_size = nw <= 0 || nh <= 0;
936                cap_x = nx.max(0) as i16;
937                cap_y = ny.max(0) as i16;
938                if let Some(g) = conn.get_geometry(root).ok().and_then(|c| c.reply().ok()) {
939                    let (fw, fh) = resolve_dims(g.width, g.height, &rsettings);
940                    if fw != cap_w || fh != cap_h {
941                        if !pool.drain_for_resize(POOL_N, &controls.stop) {
942                            break;
943                        }
944                        for s in surfaces.iter_mut() {
945                            s.destroy(&conn);
946                        }
947                        surfaces.clear();
948                        cap_w = fw;
949                        cap_h = fh;
950                        for _ in 0..POOL_N {
951                            surfaces.push(ShmSurface::create(&conn, cap_w, cap_h)?);
952                        }
953                        pool.bump_generation();
954                    }
955                }
956            }
957
958            // Root geometry is polled on a cadence, not per frame: the reply costs a
959            // round-trip that would otherwise precede every grab, and external size
960            // changes are rare (live resizes arrive through region_dirty instead).
961            // Grab-failure recovery enters regardless of auto_adjust: a fixed-size
962            // region has no other re-clamp path after an external root shrink, and
963            // without one the capture dies at the failure cap instead of recovering.
964            geometry_check = geometry_check.saturating_sub(1);
965            if (settings.auto_adjust_screen_capture_size || grab_failures > 0) && geometry_check <= 0 {
966                geometry_check = GEOMETRY_POLL_FRAMES;
967                if let Some(g) = conn.get_geometry(root).ok().and_then(|c| c.reply().ok()) {
968                    let (nw, nh) = resolve_dims(g.width, g.height, &rsettings);
969                    if nw != cap_w || nh != cap_h {
970                        if !pool.drain_for_resize(POOL_N, &controls.stop) {
971                            break;
972                        }
973                        let (fw, fh) = conn
974                            .get_geometry(root)
975                            .ok()
976                            .and_then(|c| c.reply().ok())
977                            .map(|g| resolve_dims(g.width, g.height, &rsettings))
978                            .unwrap_or((nw, nh));
979                        if fw != cap_w || fh != cap_h {
980                            for s in surfaces.iter_mut() {
981                                s.destroy(&conn);
982                            }
983                            surfaces.clear();
984                            cap_w = fw;
985                            cap_h = fh;
986                            for _ in 0..POOL_N {
987                                surfaces.push(ShmSurface::create(&conn, cap_w, cap_h)?);
988                            }
989                            pool.bump_generation();
990                        }
991                    }
992                }
993            }
994
995            let idx = match pool.acquire(&controls.stop) {
996                Some(i) => i,
997                None => break,
998            };
999            let surface = &mut surfaces[idx];
1000            if let Err(e) = grab_frame(&conn, root, surface, cap_x, cap_y, cap_w, cap_h) {
1001                // Most likely an external root resize between geometry polls made
1002                // the grab run past the root edge: return the surface, re-poll
1003                // geometry immediately, and only give up if it never recovers.
1004                pool.recycle(idx);
1005                grab_failures += 1;
1006                geometry_check = 0;
1007                if grab_failures > 120 {
1008                    return Err(e);
1009                }
1010                continue;
1011            }
1012            grab_failures = 0;
1013
1014            let frame_w = cap_w as i32;
1015            let frame_h = cap_h as i32;
1016            let stride = surface.stride;
1017            let buf = surface.as_mut_slice();
1018
1019            if controls.capture_cursor.load(Ordering::Relaxed) {
1020                if let Some(c) = conn
1021                    .xfixes_get_cursor_image()
1022                    .ok()
1023                    .and_then(|c| c.reply().ok())
1024                {
1025                    if c.width > 0 && c.height > 0 {
1026                        // Translate from root coordinates using the LIVE grab origin
1027                        // (cap_x/cap_y follow region_dirty pans and clamp negatives
1028                        // exactly like the grab itself), never the immutable startup
1029                        // settings — those go stale on the first live region move.
1030                        let (img_x, img_y) = cursor_image_origin(
1031                            c.x,
1032                            c.y,
1033                            c.xhot,
1034                            c.yhot,
1035                            cap_x as i32,
1036                            cap_y as i32,
1037                        );
1038                        overlay_cursor(
1039                            buf,
1040                            stride,
1041                            frame_w,
1042                            frame_h,
1043                            c.width as i32,
1044                            c.height as i32,
1045                            &c.cursor_image,
1046                            img_x,
1047                            img_y,
1048                        );
1049                    }
1050                }
1051            }
1052
1053            if watermark.loaded {
1054                watermark.update_position(frame_w, frame_h, settings.watermark_location_enum);
1055                watermark.blend_into(buf, stride, frame_w, frame_h);
1056            }
1057
1058            let published = pool.publish(
1059                RawFrame {
1060                    idx,
1061                    ptr: surface.addr,
1062                    len: surface.size,
1063                    width: cap_w,
1064                    height: cap_h,
1065                    stride,
1066                    generation: pool.generation(),
1067                },
1068                &controls.stop,
1069            );
1070            if !published {
1071                break;
1072            }
1073        }
1074        Ok(())
1075    })();
1076
1077    pool.shutdown();
1078    let _ = encode_thread.join();
1079    for s in surfaces.iter_mut() {
1080        s.destroy(&conn);
1081    }
1082    result
1083}
