14//! # pixelflux
15//!
16//! A high-performance screen capture and encoding pipeline exposed as a Python extension via
17//! PyO3. It supports two independent backends — **X11** (XShm + XFixes) and **Wayland**
18//! (a headless [Smithay](https://github.com/Smithay/smithay) compositor) — and a shared
19//! encoding layer that dispatches to software (x264/JPEG striping, OpenH264) or hardware
20//! (NVENC, VA-API) encoders based on the available GPU and operator settings.
21//!
22//! ## Crate structure
23//!
24//! | Module | Purpose |
25//! |--------|---------|
26//! | [`encoders`] | Encoder backends: software x264/JPEG, OpenH264, NVENC, VA-API, watermark overlay |
27//! | [`wayland`] | Headless Smithay compositor, cursor rendering |
28//! | [`x11`] | X11/XShm capture loop and stripe dispatch |
29//! | [`pipeline`] | Frame-processing policy shared by both backends (send/QP/keyframe decisions) |
30//! | [`recording_sink`] | Unix-socket H.264 fan-out for external recording |
31//! | [`computer_use`] | HTTP API for AI-agent desktop control (screenshots, input injection) |
32//! | [`nvgpufilter`] | Multi-GPU NVENC device filtering via ioctl |
33//!
34//! ## Data flow
35//!
36//! ```text
37//! Python  ──►  CaptureSettings  ──►  X11 / Wayland backend
38//!                                         │
39//!                                    frame pixels
40//!                                         │
41//!                                    ┌────┴────┐
42//!                                    │ Encoder │  (NVENC / VAAPI / x264 / OpenH264 / JPEG)
43//!                                    └────┬────┘
44//!                                         │
45//!                                   EncodedStripe(s)
46//!                                         │
47//!                                    Python callback
48//! ```

141    /// Size the CBR VBV/HRD buffer so rate control has enough slack to hold quality steady
142    /// without letting end-to-end latency drift upward.
143    ///
144    /// The size is expressed as a multiple of one frame's bit budget (`bitrate_bps / fps`) rather
145    /// than a fixed byte count so a live bitrate or framerate change rescales the buffer with it,
146    /// preserving the same latency behavior at every operating point.
147    ///
148    /// # Arguments
149    ///
150    /// * `bitrate_bps` - Target bitrate in bits per second.
151    /// * `fps` - Target frames per second.
152    /// * `keyframe_interval_s` - Seconds between scheduled keyframes; `<= 0` for infinite GOP.
153    /// * `multiplier` - Explicit buffer multiplier; `<= 0` selects the policy default (1.5 on
154    ///   infinite GOP, 3 when keyframe interval is active).
155    ///
156    /// # Returns
157    ///
158    /// VBV buffer size in bits, clamped to `[1, u32::MAX]`.
159    pub fn vbv_bits(bitrate_bps: u32, fps: f64, keyframe_interval_s: f64, multiplier: f64) -> u32 {
160        let frame_bits = bitrate_bps as f64 / fps.max(1.0);
161        let mult = if multiplier > 0.0 {
162            multiplier
163        } else if keyframe_interval_s > 0.0 {
164            3.0
165        } else {
166            1.5
167        };
168        (frame_bits * mult).round().max(1.0).min(u32::MAX as f64) as u32
169    }
170}

189fn get_process_rss_bytes() -> usize {
190    if let Ok(contents) = std::fs::read_to_string("/proc/self/statm") {
191        if let Some(rss_pages) = contents.split_whitespace().nth(1) {
192            if let Ok(pages) = rss_pages.parse::<usize>() {
193                return pages * 4096;
194            }
195        }
196    }
197    0
198}

201    let mut total_size = 0;
202    if let Ok(entries) = std::fs::read_dir("/dev/shm") {
203        for entry in entries.flatten() {
204            if let Ok(metadata) = entry.metadata() {
205                total_size += metadata.len();
206            }
207        }
208    }
209    total_size

212fn calculate_memory_threshold(width: i32, height: i32) -> usize {
213    let frame_size = (width.max(0) as usize)
214        .saturating_mul(height.max(0) as usize)
215        .saturating_mul(4);
216    let base_app_memory: usize = 300 * 1024 * 1024;
217    let buffer_allowance = frame_size.saturating_mul(20);
218    let min_threshold: usize = 1536 * 1024 * 1024;
219    base_app_memory
220        .saturating_add(buffer_allowance)
221        .max(min_threshold)
222}

240/// Export the offscreen GBM render target as a Dmabuf so the very same GPU pixels can be both
241/// rendered into and encoded with no intervening copy — the linchpin of the zero-copy capture path.
242/// The returned dmabuf is the one handle the GLES renderer binds as its framebuffer AND a hardware
243/// encoder (NVENC through CUDA, or VAAPI) imports to read those pixels directly, which only works if
244/// the buffer is described precisely enough (fd, stride, DRM modifier) for the importer to interpret
245/// it. One ARGB8888 plane is all that is carried because the compositor's offscreen target is exactly
246/// that single-plane format.
247fn create_dmabuf_from_bo(bo: &BufferObject<()>) -> Dmabuf {
248    let fd = bo.fd().expect("Failed to get FD from GBM BO");
249    let modifier = bo.modifier();
250    let stride = bo.stride();
251    let width = bo.width();
252    let height = bo.height();
253
254    let drm_modifier = Modifier::from(Into::<u64>::into(modifier));
255
256    let mut builder = Dmabuf::builder(
257        (width as i32, height as i32),
258        Fourcc::Argb8888,
259        drm_modifier,
260        DmabufFlags::empty(),
261    );
262
263    builder.add_plane(fd, 0, 0, stride);
264    builder.build().expect("Failed to build Dmabuf from GBM BO")
265}

417/// Marshal a Python settings object into the plain owned Rust value both capture backends run on.
418///
419/// Fields are read by attribute name (`getattr`), not by position, so a caller can pass any object
420/// exposing the `CaptureSettings` attributes — including a subclass carrying extras. Newer/optional
421/// fields fall back to a default when absent rather than erroring, so a caller built against an
422/// older schema still starts. Both the Wayland and X11 entry points route through this one reader
423/// to prevent drift.
424///
425/// # Arguments
426///
427/// * `settings` - A Python object exposing `CaptureSettings` attributes (`capture_width`,
428///   `capture_height`, `target_fps`, `jpeg_quality`, `video_crf`, etc.).
429///
430/// # Returns
431///
432/// An owned [`RustCaptureSettings`] on success, or a Python exception if a required field is
433/// missing or has the wrong type.

435    let watermark_path_obj = settings.getattr("watermark_path")?;

437        s
438    } else if let Ok(b) = watermark_path_obj.extract::<Vec<u8>>() {
439        String::from_utf8_lossy(&b).into_owned()
440    } else {
441        String::new()
442    };
443
444    let scale = settings
445        .getattr("scale")
446        .ok()
447        .and_then(|x| x.extract().ok())
448        .unwrap_or(1.0);
449
450    Ok(RustCaptureSettings {
451        width: settings.getattr("capture_width")?.extract()?,
452        height: settings.getattr("capture_height")?.extract()?,
453        scale,

586/// A DRM card's identity as the kernel reports it, read from the device's
587/// sysfs `uevent` (DRIVER=, PCI_ID=, OF_COMPATIBLE_n=) with per-file fallbacks
588/// (`vendor`, `modalias`, the `driver` symlink). uevent is uniform across buses
589/// (PCI, platform/devicetree, USB) and readable in unprivileged containers.
590struct CardIdentity {
591    driver: String,
592    pci_vendor: Option<u32>,
593    compatibles: Vec<String>,
594}

596/// Read a DRM card's `CardIdentity` from sysfs.
597///
598/// The device's `uevent` file is the primary source (`DRIVER=`, `PCI_ID=`, `OF_COMPATIBLE_n=`)
599/// because it is uniform across buses and readable in unprivileged containers. Each field has a
600/// fallback for cards whose `uevent` omits it: the `vendor` file for the PCI vendor, `modalias`
601/// (`of:...C<compatible>`) for the devicetree compatibles, and the `driver` symlink for the name.
602fn read_card_identity(device: &std::path::Path) -> CardIdentity {
603    let mut id = CardIdentity { driver: String::new(), pci_vendor: None, compatibles: Vec::new() };
604    if let Ok(uevent) = std::fs::read_to_string(device.join("uevent")) {
605        for line in uevent.lines() {
606            if let Some(v) = line.strip_prefix("DRIVER=") {
607                id.driver = v.trim().to_lowercase();
608            } else if let Some(v) = line.strip_prefix("PCI_ID=") {
609                id.pci_vendor = v.split(':').next().and_then(|h| u32::from_str_radix(h, 16).ok());
610            } else if line.starts_with("OF_COMPATIBLE_") && !line.starts_with("OF_COMPATIBLE_N") {
611                if let Some(v) = line.split_once('=').map(|x| x.1) {
612                    id.compatibles.push(v.trim().to_lowercase());
613                }
614            }
615        }
616    }
617    if id.pci_vendor.is_none() {
618        id.pci_vendor = std::fs::read_to_string(device.join("vendor"))
619            .ok()
620            .and_then(|v| u32::from_str_radix(v.trim().trim_start_matches("0x"), 16).ok());
621    }
622    if id.compatibles.is_empty() {
623        if let Ok(modalias) = std::fs::read_to_string(device.join("modalias")) {
624            let modalias = modalias.trim();
625            if let Some(rest) = modalias.strip_prefix("of:") {
626                id.compatibles
627                    .extend(rest.split('C').skip(1).map(|c| c.to_lowercase()));
628            }
629        }
630    }
631    if id.driver.is_empty() {
632        id.driver = std::fs::read_link(device.join("driver"))
633            .map(|p| p.file_name().map(|n| n.to_string_lossy().to_lowercase()).unwrap_or_default())
634            .unwrap_or_default();
635    }
636    id
637}

670/// Does a card match the requested token? Accepted token forms, checked against
671/// the identity the kernel itself reports: a kernel DRIVER name (exact, no
672/// table), a raw PCI vendor ID ("0x10de"/"10de"), a devicetree vendor prefix
673/// (literal first segment of any compatible), or a human vendor name resolved
674/// through the small embedded alias maps above.
675fn card_matches_token(token: &str, id: &CardIdentity) -> bool {
676    if !id.driver.is_empty() && token == id.driver {
677        return true;
678    }
679    if let Some(vid) = id.pci_vendor {
680        if u32::from_str_radix(token.trim_start_matches("0x"), 16) == Ok(vid) {
681            return true;
682        }
683        if let Some((_, ids)) = VENDOR_PCI_IDS.iter().find(|(n, _)| *n == token) {
684            if ids.contains(&vid) {
685                return true;
686            }
687        }
688    }
689    if !id.compatibles.is_empty() {
690        let prefix = OF_PREFIX_ALIASES
691            .iter()
692            .find(|(n, _)| *n == token)
693            .map(|(_, p)| *p)
694            .unwrap_or(token);
695        let want = format!("{prefix},");
696        if id.compatibles.iter().any(|c| c.starts_with(&want)) {
697            return true;
698        }
699    }
700    false
701}

717/// Resolve a usable `/dev/dri/renderD*` node, optionally matching a vendor/driver token.
718///
719/// Cards under `/sys/class/drm` are walked in numeric order, skipping cards with no render node
720/// (e.g. IPMI/VGA). When `/sys/class/drm` is unreadable (container without `/sys`), falls
721/// through to scanning `/dev/dri` directly — that fallback has no device identity so a `token`
722/// request cannot be satisfied there.
723///
724/// # Arguments
725///
726/// * `token` - Optional vendor/driver filter: a kernel driver name, PCI vendor ID (`"0x10de"`),
727///   devicetree vendor prefix, or human vendor name (`"nvidia"`, `"intel"`). `None` picks the
728///   first available node.
729///
730/// # Returns
731///
732/// A `/dev/dri/renderD*` path, or `None` if no matching node exists.
733fn auto_select_render_node(token: Option<&str>) -> Option<String> {
734    let mut cards: Vec<(u32, std::path::PathBuf)> = std::fs::read_dir("/sys/class/drm")
735        .into_iter()
736        .flatten()
737        .flatten()
738        .filter_map(|e| {
739            let num = e.file_name().into_string().ok()?.strip_prefix("card")?.parse::<u32>().ok()?;
740            Some((num, e.path()))
741        })
742        .collect();
743    cards.sort_by_key(|(n, _)| *n);
744    for (_, path) in &cards {
745        if let Some(t) = token {
746            if !card_matches_token(t, &read_card_identity(&path.join("device"))) {
747                continue;
748            }
749        }
750        if let Ok(drm_entries) = std::fs::read_dir(path.join("device/drm")) {
751            for de in drm_entries.flatten() {
752                let name = de.file_name().into_string().unwrap_or_default();
753                if name.starts_with("renderD") {
754                    let dev = format!("/dev/dri/{}", name);
755                    if std::path::Path::new(&dev).exists() {
756                        return Some(dev);
757                    }
758                }
759            }
760        }
761    }
762    if token.is_some() {
763        return None;
764    }
765    let mut nodes: Vec<String> = std::fs::read_dir("/dev/dri")
766        .ok()?
767        .flatten()
768        .filter_map(|e| e.file_name().into_string().ok())
769        .filter(|n| n.starts_with("renderD"))
770        .collect();
771    nodes.sort();
772    nodes.first().map(|n| format!("/dev/dri/{}", n))
773}

788/// Interior state of `WlFramePool`: the free-buffer list plus the single publish slot.
789struct WlPoolInner {
790    free: Vec<(usize, Vec<u8>)>,
791    slot: Option<WlFrame>,
792}

794/// Render->encode handoff for the Wayland readback paths, mirroring the X11 FramePool's
795/// single-slot non-dropping design with one deliberate difference: the calloop thread is also
796/// the compositor + input dispatcher, so it must NEVER block on the pool. `try_begin` hands
797/// out a buffer only while the publish slot is empty, so `publish` cannot block, and a
798/// saturated encoder throttles capture by SKIPPING ticks (compositor damage accumulates via
799/// buffer age, so nothing is lost). Every published frame is encoded, in order: the H.264
800/// reference chain stays contiguous exactly as on X11.
801pub struct WlFramePool {
802    inner: Mutex<WlPoolInner>,
803    cv: Condvar,
804    stop: AtomicBool,
805}
806
807impl WlFramePool {
808    /// Pre-allocate all `n` capture buffers up front (each `buf_len` bytes, all initially
809    /// free) so the steady-state render/encode loop hands buffers around without ever allocating on
810    /// the hot path.
811    fn new(n: usize, buf_len: usize) -> Self {
812        Self {
813            inner: Mutex::new(WlPoolInner {
814                free: (0..n).map(|i| (i, vec![0u8; buf_len])).collect(),
815                slot: None,
816            }),
817            cv: Condvar::new(),
818            stop: AtomicBool::new(false),
819        }
820    }
821
822    /// Calloop: reserve a buffer for the next render/readback, NON-blocking. None means the
823    /// encoder is still behind (slot full or every buffer in flight) -- skip this tick.
824    /// Because the calloop is the only producer, a successful reservation guarantees the
825    /// following `publish` finds the slot empty (the consumer only ever drains it).
826    fn try_begin(&self) -> Option<(usize, Vec<u8>)> {
827        let mut g = self.inner.lock().unwrap();
828        if g.slot.is_some() {
829            return None;
830        }
831        g.free.pop()
832    }
833
834    /// Calloop: hand the filled buffer to the encode thread. Never blocks (see `try_begin`).
835    fn publish(&self, frame: WlFrame) {
836        let mut g = self.inner.lock().unwrap();
837        debug_assert!(g.slot.is_none());
838        g.slot = Some(frame);
839        drop(g);
840        self.cv.notify_all();
841    }
842
843    /// Calloop: return an unused reservation (render failed, readback skipped).
844    fn cancel(&self, id: usize, buf: Vec<u8>) {
845        self.inner.lock().unwrap().free.push((id, buf));
846    }
847
848    /// Encode: block until a frame is published (Some) or the pool shuts down (None). The
849    /// wait is bounded, re-checking `stop` as defense-in-depth against a lost wakeup.
850    fn take(&self) -> Option<WlFrame> {
851        let mut g = self.inner.lock().unwrap();
852        loop {
853            if let Some(f) = g.slot.take() {
854                return Some(f);
855            }
856            if self.stop.load(Ordering::Acquire) {
857                return None;
858            }
859            let (gg, _) = self.cv.wait_timeout(g, Duration::from_millis(20)).unwrap();
860            g = gg;
861        }
862    }
863
864    /// Encode: return an encoded frame's buffer so the calloop can capture into it again.
865    /// No notify: the only waiter (take) waits on the slot, and try_begin never blocks.
866    fn recycle(&self, id: usize, buf: Vec<u8>) {
867        self.inner.lock().unwrap().free.push((id, buf));
868    }
869
870    /// Store stop under the lock so take() can't check stop==false and then park after the
871    /// notify already fired (lost wakeup); notify after unlocking.
872    fn shutdown(&self) {
873        let g = self.inner.lock().unwrap();
874        self.stop.store(true, Ordering::Release);
875        drop(g);
876        self.cv.notify_all();
877    }
878}

1029/// Encode-thread body for the Wayland readback paths: drain published frames and run the
1030/// full encode dispatch, owning the encoders for the life of the capture.
1031///
1032/// Owning the encoders on this one thread (created, driven, and dropped here) is what lets the
1033/// calloop overlap the next render/readback with this encode — the same capture‖encode split used
1034/// on X11, minus the renderer, which is genuinely calloop-affine (EGL/GBM/dmabuf and the pixman
1035/// targets). Each published frame is processed in order so the H.264 reference chain stays
1036/// contiguous:
1037///
1038/// 1. **Apply cross-thread changes**: live tunables and rate/VBV/fps updates are read here, on the
1039///    thread that owns the encoders (each `Acquire` swap pairs with the command handler's `Release`
1040///    store, so a payload is never seen half-applied). The IDR request is swapped as late as
1041///    possible, so a request that arrived while this frame was in flight is honored one pipeline
1042///    stage earlier than the next publish.
1043/// 2. **Dispatch by encoder**: a hardware session runs `decide_hw_fullframe`, then colorspace-
1044///    converts only the frames actually being encoded (RGBA vs BGRA source chosen by the renderer)
1045///    into the reused NV12/YUV444 buffer and calls `encode_raw`; OpenH264 encodes host pixels
1046///    directly; otherwise `encode_cpu` runs the striped x264/JPEG path with compositor damage. The
1047///    software H.264 path keeps an infinite GOP, forcing an IDR only on an explicit request or the
1048///    configured interval, and an explicit request also forces a full JPEG resend for joiners.
1049/// 3. **Recycle then deliver**: the capture buffer is recycled BEFORE delivery so a slow consumer
1050///    never pins one, then the stripes go to the delivery thread through a single-slot `send` whose
1051///    blocking is the backpressure that overlaps delivery with the next render + encode.
1052///
1053/// On exit the hardware session is handed back to the calloop so a restart can reuse it in place
1054/// when the new settings stay compatible (a plain `StopCapture` just drops it).
1055fn wayland_encode_loop(pool: &WlFramePool, cfg: WlEncodeConfig) -> Option<GpuEncoder> {
1056    crate::boost_thread_priority(-10);
1057    let mut settings = cfg.settings;
1058    let (mut video_encoder, mut openh264_encoder) =
1059        build_readback_encoders(&settings, cfg.try_gpu, cfg.prior, cfg.recording_sink.clone());
1060    if cfg.try_gpu && video_encoder.is_none() && openh264_encoder.is_none() {
1061        println!("[Wayland] Decision: No GPU Encoder available -> Using CPU Software Encoding.");
1062    }
1063    if cfg.recording_sink.is_some()
1064        && settings.output_mode == 1
1065        && video_encoder.is_none()
1066        && openh264_encoder.is_none()
1067        && !settings.video_fullframe
1068    {
1069        eprintln!(
1070            "[recording_sink] WARNING: recording_socket is set but the CPU encoder is running in \
1071             multi-stripe mode. This produces N independent sub-frame H.264 streams that \
1072             cannot be muxed together. Set video_fullframe=true on the Python CaptureSettings \
1073             (or use a working GPU encoder) to produce a recordable single-stream output."
1074        );
1075    }
1076    let n_stripes = wayland_stripe_count(
1077        &settings,
1078        video_encoder.is_some() || openh264_encoder.is_some(),
1079    );
1080    cfg.stats.n_stripes.store(n_stripes as u32, Ordering::Relaxed);
1081    *cfg.stats.desc.lock().unwrap() = encoder_desc(
1082        &settings,
1083        video_encoder.as_ref(),
1084        openh264_encoder.is_some(),
1085        false,
1086    );
1087    log_stream_settings(&settings, n_stripes, video_encoder.as_ref(), openh264_encoder.is_some());
1088
1089    let width = settings.width;
1090    let height = settings.height;
1091    let mut stripes: Vec<StripeState> = Vec::with_capacity(MAX_STRIPE_CAPACITY);
1092    let mut hw_state = StripeState::default();
1093    let mut nv12_buffer: Vec<u8> = if video_encoder.is_some() {
1094        let n = (width * height) as usize;
1095        vec![0u8; if settings.video_fullcolor { n * 3 } else { n * 3 / 2 }]
1096    } else {
1097        Vec::new()
1098    };
1099
1100    while let Some(mut f) = pool.take() {
1101        if cfg.controls.tunables_dirty.swap(false, Ordering::Acquire) {
1102            if let Some(t) = cfg.controls.tunables.lock().unwrap().take() {
1103                t.apply_to(&mut settings);
1104            }
1105        }
1106        if cfg.controls.rate_dirty.swap(false, Ordering::Acquire) {
1107            settings.video_bitrate_kbps = cfg.controls.bitrate_kbps.load(Ordering::Relaxed);
1108            settings.video_vbv_multiplier =
1109                cfg.controls.vbv_mult_milli.load(Ordering::Relaxed) as f64 / 1000.0;
1110            let fps = (cfg.controls.fps_milli.load(Ordering::Relaxed) as f64) / 1000.0;
1111            if fps > 0.0 {
1112                settings.target_fps = fps;
1113            }
1114            match video_encoder.as_mut() {
1115                Some(GpuEncoder::Nvenc(enc)) => enc.reconfigure_rate(&settings),
1116                Some(GpuEncoder::Vaapi(enc)) => enc.reconfigure_rate(&settings),
1117                None => {}
1118            }
1119            if let Some(enc) = openh264_encoder.as_mut() {
1120                enc.reconfigure_rate(settings.video_bitrate_kbps, settings.target_fps);
1121            }
1122        }
1123
1124        let requested_idr = cfg.controls.force_idr.swap(false, Ordering::Relaxed);
1125
1126        let mut out: Vec<EncodedStripe> = Vec::new();
1127        if let Some(ref mut encoder) = video_encoder {
1128            let decision = crate::pipeline::decide_hw_fullframe(
1129                &mut hw_state,
1130                &settings,
1131                f.frame_id,
1132                !f.damage.is_empty(),
1133                f.is_animated,
1134                requested_idr,
1135            );
1136            if decision.send {
1137                let w = width as u32;
1138                let h = height as u32;
1139                if settings.video_fullcolor {
1140                    let y_size = (w * h) as usize;
1141                    let (y_plane, rest) = nv12_buffer.split_at_mut(y_size);
1142                    let (u_plane, v_plane) = rest.split_at_mut(y_size);
1143                    let mut planar_image = yuv::YuvPlanarImageMut {
1144                        y_plane: BufferStoreMut::Borrowed(y_plane),
1145                        y_stride: w,
1146                        u_plane: BufferStoreMut::Borrowed(u_plane),
1147                        u_stride: w,
1148                        v_plane: BufferStoreMut::Borrowed(v_plane),
1149                        v_stride: w,
1150                        width: w,
1151                        height: h,
1152                    };
1153                    let _ = if cfg.use_gpu {
1154                        yuv::rgba_to_yuv444(&mut planar_image, &f.buf, w * 4, YuvRange::Full, YuvStandardMatrix::Bt709, YuvConversionMode::Fast)
1155                    } else {
1156                        yuv::bgra_to_yuv444(&mut planar_image, &f.buf, w * 4, YuvRange::Full, YuvStandardMatrix::Bt709, YuvConversionMode::Fast)
1157                    };
1158                } else {
1159                    let y_size = (w * h) as usize;
1160                    let (y_plane, uv_plane) = nv12_buffer.split_at_mut(y_size);
1161                    let mut planar_image = YuvBiPlanarImageMut {
1162                        y_plane: BufferStoreMut::Borrowed(y_plane),
1163                        y_stride: w,
1164                        uv_plane: BufferStoreMut::Borrowed(uv_plane),
1165                        uv_stride: w,
1166                        width: w,
1167                        height: h,
1168                    };
1169                    let csc = if cfg.use_gpu {
1170                        yuv::rgba_to_yuv_nv12(&mut planar_image, &f.buf, w * 4, YuvRange::Limited, YuvStandardMatrix::Bt709, YuvConversionMode::Fast)
1171                    } else {
1172                        yuv::bgra_to_yuv_nv12(&mut planar_image, &f.buf, w * 4, YuvRange::Limited, YuvStandardMatrix::Bt709, YuvConversionMode::Fast)
1173                    };
1174                    if let Err(e) = csc {
1175                        eprintln!("[wl-encode] NV12 CSC failed: {e:?}");
1176                    }
1177                }
1178                let force_idr_for_recording = cfg
1179                    .recording_sink
1180                    .as_ref()
1181                    .map(|s| s.should_force_idr())
1182                    .unwrap_or(false);
1183                let force_idr = decision.force_idr || force_idr_for_recording;
1184                let result = match encoder {
1185                    GpuEncoder::Nvenc(enc) => {
1186                        enc.encode_raw(&nv12_buffer, f.frame_id as u64, decision.target_qp, force_idr)
1187                    }
1188                    GpuEncoder::Vaapi(enc) => {
1189                        enc.encode_raw(&nv12_buffer, f.frame_id as u64, decision.target_qp, force_idr)
1190                    }
1191                };
1192                match result {
1193                    Ok(data) if !data.is_empty() => out.push(EncodedStripe {
1194                        data,
1195                        data_type: 2,
1196                        stripe_y_start: 0,
1197                        stripe_height: height,
1198                        frame_id: f.frame_id as i32,
1199                    }),
1200                    Ok(_) => {}
1201                    Err(e) => eprintln!("HW Encode Error: {}", e),
1202                }
1203            }
1204        } else if let Some(ref mut enc) = openh264_encoder {
1205            let decision = crate::pipeline::decide_hw_fullframe(
1206                &mut hw_state,
1207                &settings,
1208                f.frame_id,
1209                !f.damage.is_empty(),
1210                f.is_animated,
1211                requested_idr,
1212            );
1213            if decision.send {
1214                let force_idr_for_recording = cfg
1215                    .recording_sink
1216                    .as_ref()
1217                    .map(|s| s.should_force_idr())
1218                    .unwrap_or(false);
1219                let force_idr = decision.force_idr || force_idr_for_recording;
1220                let stride = (width * 4) as usize;
1221                match enc.encode_host_argb(&f.buf, stride, f.frame_id as u64, force_idr, cfg.use_gpu) {
1222                    Ok(data) if !data.is_empty() => out.push(EncodedStripe {
1223                        data,
1224                        data_type: 2,
1225                        stripe_y_start: 0,
1226                        stripe_height: height,
1227                        frame_id: f.frame_id as i32,
1228                    }),
1229                    Ok(_) => {}
1230                    Err(e) => eprintln!("OpenH264 Encode Error: {}", e),
1231                }
1232            }
1233        } else {
1234            let mut damage = std::mem::take(&mut f.damage);
1235            if f.is_animated {
1236                damage.push(Rectangle::new((0, 0).into(), (width, height).into()));
1237            }
1238            let force_idr_all = requested_idr
1239                || (settings.output_mode == 1
1240                    && crate::pipeline::periodic_idr_due(&settings, f.frame_id));
1241            out = encoders::software::encode_cpu(
1242                &mut stripes,
1243                &f.buf,
1244                width,
1245                height,
1246                &damage,
1247                &settings,
1248                f.frame_id,
1249                cfg.use_gpu,
1250                false,
1251                cfg.recording_sink.as_ref(),
1252                force_idr_all,
1253            );
1254        }
1255
1256        let WlFrame { id, buf, .. } = f;
1257        pool.recycle(id, buf);
1258        if !out.is_empty() {
1259            cfg.stats.frames.fetch_add(1, Ordering::Relaxed);
1260            cfg.stats.stripes.fetch_add(out.len() as u32, Ordering::Relaxed);
1261            let _ = cfg.deliver_tx.send(out);
1262        }
1263    }
1264    if settings.debug_logging {
1265        println!(
1266            "[Wayland] Encode thread exiting (hw={}, openh264={}, stripes={}).",
1267            video_encoder.is_some(),
1268            openh264_encoder.is_some(),
1269            stripes.len()
1270        );
1271    }
1272    video_encoder
1273}

1275/// Compose the encoder half of the 1 s debug log line (backend, colorspace, frame mode) for
1276/// whichever thread owns the encoders.
1277fn encoder_desc(
1278    settings: &RustCaptureSettings,
1279    video_encoder: Option<&GpuEncoder>,
1280    openh264: bool,
1281    zero_copy: bool,
1282) -> String {
1283    if settings.output_mode == 0 {
1284        return format!("JPEG Q:{}", settings.jpeg_quality);
1285    }
1286    let copy_mode = if zero_copy { "ZeroCopy" } else { "Readback" };
1287    let backend = if openh264 {
1288        "OpenH264".to_string()
1289    } else {
1290        match video_encoder {
1291            Some(GpuEncoder::Nvenc(_)) => format!("NVENC ({})", copy_mode),
1292            Some(GpuEncoder::Vaapi(_)) => format!("VAAPI ({})", copy_mode),
1293            None => "CPU".to_string(),
1294        }
1295    };
1296    let is_444 = !openh264
1297        && match video_encoder {
1298            Some(GpuEncoder::Nvenc(_)) => settings.video_fullcolor,
1299            Some(_) => false,
1300            None => settings.video_fullcolor,
1301        };
1302    let cs_str = if is_444 { "CS_IN:I444" } else { "CS_IN:I420" };
1303    let range_str = if is_444 { "FR" } else { "LR" };
1304    let frame_str = if video_encoder.is_some() || openh264 || settings.video_fullframe {
1305        "FF"
1306    } else {
1307        "Striped"
1308    };
1309    format!("H264 ({}) {} {} {} CRF:{}", backend, cs_str, range_str, frame_str, settings.video_crf)
1310}

1312/// Decide how many horizontal stripes a frame is split into, which is really the choice of how
1313/// much encode parallelism to spend on it. A full-frame session (`fullframe_encoder`: a HW or
1314/// OpenH264 encoder, or a forced `video_fullframe`) is one contiguous H.264 stream and so is always a
1315/// SINGLE stripe; the striped CPU paths instead fan the frame out across cores — bounded by
1316/// `height/64` for H.264 so no stripe is shorter than a macroblock row, or by `height` for JPEG — so
1317/// a many-core host encodes a frame in parallel without over-subscribing a tiny one.
1318fn wayland_stripe_count(settings: &RustCaptureSettings, fullframe_encoder: bool) -> usize {
1319    let num_cores = std::thread::available_parallelism()
1320        .map(|n| n.get())
1321        .unwrap_or(1);
1322    if settings.output_mode == 1 {
1323        if fullframe_encoder || settings.video_fullframe {
1324            1
1325        } else {
1326            let min_h = 64;
1327            if settings.height < min_h {
1328                1
1329            } else {
1330                num_cores.min((settings.height / min_h) as usize).max(1)
1331            }
1332        }
1333    } else {
1334        num_cores.min(settings.height as usize).max(1)
1335    }
1336}

1416/// The main execution loop of the Wayland backend.
1417///
1418/// This function is the central nervous system of the backend. It runs on its own thread and owns
1419/// the entire lifecycle of the headless Wayland compositor:
1420///
1421/// 1. **Initialization**: builds the `calloop` event loop and the Wayland display, raises
1422///    libwayland's per-client buffer limit when the newer setter is available (resolved at runtime
1423///    so the module still loads against older libwayland), and brings up the rendering pipeline —
1424///    GBM/EGL hardware acceleration on the resolved DRM render node, falling back to software
1425///    rendering (Pixman) when no node is usable.
1426/// 2. **State management**: constructs and holds the `AppState` — the Wayland globals (compositor,
1427///    seat, SHM, shell, dmabuf, selections, and the rest), the single virtual `HEADLESS-1` output,
1428///    the frame buffer and encode pools, and the window / cursor / encode bookkeeping.
1429/// 3. **Event dispatch**:
1430///    - **Command channel**: control messages from the Python thread — start/stop, input injection,
1431///      keymap and clipboard operations, live rate / tunable changes, and the computer-use queries.
1432///    - **Wayland socket**: accepts client connections and drives the compositor protocol.
1433/// 4. **StartCapture reconfigure**: reprograms the virtual output's mode / scale / refresh, resizes
1434///    the framebuffer and offscreen GBM buffer, and fullscreens mapped toplevels. The encode device
1435///    is resolved here: an operator's explicit `encode_node_index` (-1 software, >= 0 a device)
1436///    always wins, and only the unset `-2` sentinel is filled from the auto-picked render node.
1437///    H.264 output masks the dimensions even, because 4:2:0 needs even width and height.
1438/// 5. **Encode-path choice + render loop**: only a same-GPU GLES session encodes zero-copy on this
1439///    calloop thread, because the dmabuf and its EGL context are calloop-affine; every readback
1440///    flavor (OpenH264, striped x264/JPEG, Pixman, or a cross-GPU hardware encoder) builds its
1441///    encoders on the dedicated encode thread instead. A high-frequency timer composites the mapped
1442///    windows plus cursor and watermark, applies the shared paint-over / recovery-IDR policy, and
1443///    delivers the encoded stripes back to Python through the frame callback. The zero-copy encode
1444///    waits the GL render fence first, so a hardware encoder reading the dmabuf through CUDA/VA
1445///    never maps a half-rasterized (torn) frame.
1446/// 6. **Thread lifecycle**: the Python frame callback runs on a dedicated delivery thread so its
1447///    GIL never stalls calloop input / control dispatch, and in readback mode the encoders run on
1448///    the `wl-encode` thread. On a restart or stop the encode thread is torn down before the
1449///    delivery thread — it feeds the delivery sender and must be gone first — and the retained
1450///    callbacks are dropped and gated by a process-shutdown flag so nothing fires into a finalizing
1451///    interpreter.
1452fn run_wayland_thread(
1453    command_rx: smithay::reexports::calloop::channel::Channel<ThreadCommand>,
1454    command_tx: smithay::reexports::calloop::channel::Sender<ThreadCommand>,
1455    initial_width: i32,
1456    initial_height: i32,
1457    explicit_dri_node: String,
1458    auto_gpu_selected: bool,
1459    cursor_size: i32,
1460) {
1461    let width: i32 = if initial_width > 0 { initial_width } else { 1024 };
1462    let height: i32 = if initial_height > 0 { initial_height } else { 768 };
1463
1464    let mut event_loop = EventLoop::<AppState>::try_new().expect("Unable to create event_loop");
1465    let display: Display<AppState> = Display::new().unwrap();
1466    let dh: DisplayHandle = display.handle();
1467    unsafe {
1468        if let Ok(lib) = libloading::Library::new("libwayland-server.so.0") {
1469            if let Ok(set_max) = lib.get::<unsafe extern "C" fn(*mut std::ffi::c_void, usize)>(
1470                b"wl_display_set_default_max_buffer_size\0",
1471            ) {
1472                set_max(
1473                    dh.backend_handle().display_ptr() as *mut std::ffi::c_void,
1474                    10 * 1024 * 1024,
1475                );
1476            }
1477            std::mem::forget(lib);
1478        }
1479    }
1480
1481    let dri_node = explicit_dri_node;
1482
1483    let mut use_gpu = !dri_node.is_empty();
1484    let render_node_path = dri_node.clone();
1485
1486    let mut gles_renderer = None;
1487    let mut pixman_renderer = None;
1488    let mut offscreen_buffer: Option<(BufferObject<()>, Dmabuf)> = None;
1489    let mut dmabuf_global = None;
1490    let mut gbm_device_raw = None;
1491    let mut dmabuf_state = DmabufState::new();
1492
1493    let mut gpu_success = false;
1494    if use_gpu {
1495        println!("[Wayland] Initializing GL Renderer using device: {}", dri_node);
1496        let init_res: Result<(), String> = (|| {
1497            let device_path = std::path::Path::new(&dri_node);
1498            let file = File::options().read(true).write(true).open(device_path)
1499                .map_err(|e| format!("Failed to open render device: {}", e))?;
1500            let file_for_alloc = file.try_clone()
1501                .map_err(|e| format!("Failed to clone file for GBM Allocator: {}", e))?;
1502            let gbm_allocator = RawGbmDevice::new(file_for_alloc)
1503                .map_err(|_| "Failed to create Raw GBM Device")?;
1504            let gbm = GbmDevice::new(file)
1505                .map_err(|_| "Failed to create GBM device")?;
1506            let egl = unsafe { EGLDisplay::new(gbm) }
1507                .map_err(|_| "Failed to create EGL display")?;
1508            let context = EGLContext::new(&egl)
1509                .map_err(|_| "Failed to create EGL context")?;
1510            let mut renderer = unsafe { GlesRenderer::new(context) }
1511                .map_err(|_| "Failed to init GlesRenderer")?;
1512            
1513            if let Err(e) = renderer.bind_wl_display(&dh) {
1514                println!("[Wayland] Warning: Failed to bind EGL to Wayland Display (Optional): {:?}", e);
1515            }
1516
1517            let formats = Bind::<Dmabuf>::supported_formats(&renderer)
1518                .ok_or("Failed to query formats")?
1519                .into_iter()
1520                .collect::<Vec<_>>();
1521
1522            let node = DrmNode::from_path(device_path)
1523                .map_err(|_| "Failed to create DrmNode")?;
1524            let dmabuf_default_feedback = DmabufFeedbackBuilder::new(node.dev_id(), formats.clone()).build();
1525
1526            dmabuf_global = Some(if let Ok(default_feedback) = dmabuf_default_feedback {
1527                dmabuf_state.create_global_with_default_feedback::<AppState>(&dh, &default_feedback)
1528            } else {
1529                dmabuf_state.create_global::<AppState>(&dh, formats)
1530            });
1531
1532            let bo = gbm_allocator.create_buffer_object(
1533                width as u32, height as u32, GbmFormat::Argb8888, BufferObjectFlags::RENDERING
1534            ).map_err(|_| "Failed to allocate GBM buffer")?;
1535
1536            let dmabuf = create_dmabuf_from_bo(&bo);
1537            offscreen_buffer = Some((bo, dmabuf));
1538            gbm_device_raw = Some(gbm_allocator);
1539            gles_renderer = Some(renderer);
1540            Ok(())
1541        })();
1542
1543        match init_res {
1544            Ok(_) => gpu_success = true,
1545            Err(e) => {
1546                println!("[Wayland] GPU Initialization failed: {}. Falling back to Software Renderer (Pixman).", e);
1547                use_gpu = false;
1548            }
1549        }
1550    }
1551
1552    if !gpu_success {
1553        if !dri_node.is_empty() && !use_gpu {
1554        } else {
1555            println!("[Wayland] No render node. Initializing Software Renderer (Pixman).");
1556        }
1557        pixman_renderer = Some(PixmanRenderer::new().expect("Failed to init PixmanRenderer"));
1558        use_gpu = false;
1559    }
1560
1561    let compositor_state = CompositorState::new::<AppState>(&dh);
1562    let fractional_scale_state = FractionalScaleManagerState::new::<AppState>(&dh);
1563    let shm_state = ShmState::new::<AppState>(&dh, vec![]);
1564    let output_state = OutputManagerState::new_with_xdg_output::<AppState>(&dh);
1565    let mut seat_state = SeatState::new();
1566    let shell_state = XdgShellState::new::<AppState>(&dh);
1567    let space = Space::default();
1568    let layer_shell_state = WlrLayerShellState::new::<AppState>(&dh);
1569    let data_device_state = DataDeviceState::new::<AppState>(&dh);
1570    let data_control_state = DataControlState::new::<AppState, _>(&dh, None, |_| true);
1571    let virtual_keyboard_state = VirtualKeyboardManagerState::new::<AppState, _>(&dh, |_client| true);
1572    let pointer_warp_state = PointerWarpManager::new::<AppState>(&dh);
1573    let relative_pointer_state = RelativePointerManagerState::new::<AppState>(&dh);
1574    let pointer_constraints_state = PointerConstraintsState::new::<AppState>(&dh);

3425struct ScState {

3427    backend: u8,
3428    controls: Option<Arc<crate::x11::Controls>>,
3429    handle: Option<thread::JoinHandle<()>>,

3450/// Unified capture handle exposed to Python. Drives the X11 capture directly or delegates to the
3451/// shared Wayland backend, chosen at `start_capture` time. Exposes start_capture / stop_capture /
3452/// request_idr_frame / update_* / is_capturing, plus the Wayland input-injection methods.
3453#[pyclass]
3454struct ScreenCapture {
3455    id: u64,
3456    inner: Mutex<ScState>,
3457}

3459impl ScreenCapture {
3460    /// Stop this capture: signal the capture thread, drop the live controls, and join.
3461    ///
3462    /// The path forks on the backend. A **Wayland** capture only tells the shared compositor to
3463    /// stop when this instance still owns it — ownership is claimed-and-cleared atomically so a
3464    /// stale stop cannot tear down a capture another instance just started. An **X11** capture joins
3465    /// its capture thread (which also joins the encode thread) and then its deliver thread,
3466    /// releasing the GIL first because the deliver thread runs the Python callback and holding the
3467    /// GIL across the joins would deadlock. A re-entrant stop arriving on the capture, encode, or
3468    /// deliver thread cannot join itself, so it detaches and lets the threads exit on the stop flag.
3469    fn stop_internal(&self, py: Python<'_>) -> PyResult<()> {
3470        let (handle, deliver_handle, same_thread, backend, controls) = {
3471            let mut st = self.inner.lock().unwrap();
3472            if let Some(c) = &st.controls {
3473                c.stop.store(true, Ordering::Relaxed);
3474            }
3475            let cur = Some(thread::current().id());
3476            if st.encode_thread_id.is_none() {
3477                if let Some(rx) = st.encode_tid_rx.as_ref() {
3478                    if let Ok(id) = rx.try_recv() {
3479                        st.encode_thread_id = Some(id);
3480                    }
3481                }
3482            }
3483            let same = st.cap_thread_id == cur
3484                || st.encode_thread_id == cur
3485                || st.deliver_thread_id == cur;
3486            let controls = st.controls.take();
3487            let handle = st.handle.take();
3488            let deliver_handle = st.deliver_handle.take();
3489            let backend = st.backend;
3490            st.backend = 0;
3491            st.cap_thread_id = None;
3492            st.encode_thread_id = None;
3493            st.encode_tid_rx = None;
3494            st.deliver_thread_id = None;
3495            (handle, deliver_handle, same, backend, controls)
3496        };
3497        if let Some(c) = &controls {
3498            live_x11().lock().unwrap().retain(|x| !Arc::ptr_eq(x, c));
3499        }
3500        if backend == 2 {
3501            if WAYLAND_OWNER
3502                .compare_exchange(self.id, 0, Ordering::AcqRel, Ordering::Relaxed)
3503                .is_ok()
3504            {
3505                if let Some(slot) = WAYLAND_BACKEND.get() {
3506                    if let Some(be) = slot.lock().unwrap().as_ref() {
3507                        let _ = be.bind(py).borrow().stop_capture();
3508                    }
3509                }
3510            }
3511        } else {
3512            if same_thread {
3513                // Detach: the threads exit on the stop flag once the callback returns.
3514                drop(handle);
3515                drop(deliver_handle);
3516            } else {
3517                py.detach(|| {
3518                    if let Some(h) = handle {
3519                        let _ = h.join();
3520                    }
3521                    // The capture join above ends the encode thread, dropping the
3522                    // delivery sender; the deliver thread then drains its one
3523                    // queued frame and exits, so this join is bounded.
3524                    if let Some(h) = deliver_handle {
3525                        let _ = h.join();
3526                    }
3527                });
3528            }
3529        }
3530        Ok(())
3531    }
3532}
