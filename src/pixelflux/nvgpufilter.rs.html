7//! Multi-GPU NVENC `GET_ATTACHED_IDS` / `GET_PROBED_IDS` ioctl filter — it exists so a container
8//! handed only a subset of the host's GPUs can still open an NVENC session.
9//!
10//! The problem: on NVIDIA driver 570-595, `libnvidia-encode` / `libcuda` / `libnvcuvid` enumerate
11//! every *host* GPU via the RM `GET_ATTACHED_IDS` ioctl and try to peer-init each one — including
12//! GPUs the container never exposed. A GPU whose `/dev/nvidiaX` node is absent then makes
13//! `nvEncOpenEncodeSessionEx` fail with UNSUPPORTED_DEVICE, so the session cannot open at all even
14//! though a perfectly usable GPU is right there in the container.
15//!
16//! The fix is to strip the unreachable GPUs out of that enumeration response before the libraries
17//! act on it. It GOT-patches `ioctl` in those NVIDIA libraries *only* — deliberately not an
18//! LD_PRELOAD object, which would shadow every `ioctl` in the process and need its own recursion
19//! guard. Because this crate's own GOT is left untouched, the wrapper's inner `ioctl` still resolves
20//! to the real libc instead of re-entering itself.
21//!
22//! Everything hinges on at least one host GPU being hidden from the container, since that is the
23//! only situation the bug arises in: on 565-or-before / 610-or-later drivers (enumeration already
24//! correct) and whenever the container can see every host GPU, the strict-subset rule downstream
25//! makes the whole filter a no-op.
26

28use std::ffi::CStr;
29use std::sync::atomic::{AtomicPtr, AtomicU64, Ordering};
30use std::sync::Once;

32/// The ioctl command number (`_IOC_NR`, bits 0-7) that identifies the RM control escape —
33/// the low byte the request gate keys on to recognize `NV_RM_CONTROL_REQUEST`. Kept as a named
34/// constant only for documentation and the `ioc_nr_extracts_low_byte` unit test, hence
35/// `#[allow(dead_code)]`.
36#[allow(dead_code)]
37const NV_ESC_RM_CONTROL: c_ulong = 0x2A;
38/// The RM control ioctl the gate recognizes: `_IOWR('F'=0x46, NR=0x2A,
39/// sizeof(NVOS54_PARAMETERS)=32)`, encoding DIR=READ|WRITE, TYPE=0x46, NR=0x2A, SIZE=0x20.
40///
41/// Matching is by DIR|TYPE|NR with the encoded `_IOC_SIZE` deliberately masked off (via
42/// `ioc_no_size`): a driver whose `NVOS54_PARAMETERS` has a different `sizeof` encodes a different
43/// size and would slip past an exact-request compare, yet it is still the very same command.
44/// Matching the NR alone would be too loose and pinning the exact encoded size too tight, so the
45/// gate keys on DIR|TYPE|NR and validates the true per-command parameter layout separately via
46/// `ctrl.params_size`. This is the deliberate middle ground.
47const NV_RM_CONTROL_REQUEST: c_ulong = 0xC020_462A;
48/// Locates the `_IOC_SIZE` field so it can be cleared for size-agnostic request matching:
49/// the 14-bit encoded parameter size lives at bit 16 of an ioctl request (`asm-generic/ioctl.h`).
50/// Paired with `IOC_SIZEMASK` by `ioc_no_size`.
51const IOC_SIZESHIFT: u32 = 16;
52/// 14-bit mask covering the `_IOC_SIZE` field (`(1 << 14) - 1`); shifted by `IOC_SIZESHIFT`
53/// and cleared from a request by `ioc_no_size` so the driver's param-struct size cannot affect the
54/// match.
55const IOC_SIZEMASK: c_ulong = 0x3FFF;
56/// Identifies the attached-GPU enumeration response the filter rewrites — RM control command
57/// `NV0000_CTRL_CMD_GPU_GET_ATTACHED_IDS`, matched against `NvRmControlParams::cmd`.
58const GPU_GET_ATTACHED_IDS: u32 = 0x0201;
59/// Identifies the probed-GPU enumeration response the filter rewrites — RM control command
60/// `NV0000_CTRL_CMD_GPU_GET_PROBED_IDS`, matched against `NvRmControlParams::cmd`.
61const GPU_GET_PROBED_IDS: u32 = 0x0214;
62/// The RM ABI caps attached GPUs at 32, so this is the fixed `gpuIds` array capacity in both
63/// enumeration params structs — and the bound the filter's scans and scratch buffers are sized to.
64const MAX_ATTACHED_GPUS: usize = 32;
65/// Marks the end of the live ids: the RM fills unused `gpuIds` slots with this sentinel, so
66/// the first `INVALID_GPU_ID` terminates the populated prefix the filter scans and rewrites.
67const INVALID_GPU_ID: u32 = 0xFFFF_FFFF;
68/// The `params_size` that marks an ATTACHED response and tells it apart from PROBED: both
69/// structs open with the same `gpuIds[32]`, so only the exact total size (here `gpuIds[32]` =
70/// `4 * MAX_ATTACHED_GPUS`) disambiguates them before the shared leading array is rewritten.
71const ATTACHED_PARAMS_SIZE: usize = 4 * MAX_ATTACHED_GPUS;
72/// The `params_size` that marks a PROBED response and tells it apart from ATTACHED: this
73/// struct is `gpuIds[32]` followed by `excludedGpuIds[32]` (`4 * MAX_ATTACHED_GPUS * 2`). Only the
74/// shared leading `gpuIds[32]` is rewritten; the trailing `excludedGpuIds[32]` lists GPUs to
75/// exclude, not to use, so it is deliberately left untouched.
76const PROBED_PARAMS_SIZE: usize = 4 * MAX_ATTACHED_GPUS * 2;

78/// The param block the RM control ioctl hands back (`NVOS54_PARAMETERS`, 32 bytes, pointed at
79/// by `arg`) — modeled here so the filter can identify and validate an enumeration call before it
80/// touches the GPU-id array it carries.
81///
82/// The rewrite reads `cmd` to identify the enumeration API, `status` and `params` to confirm the
83/// call succeeded and carries a payload, and `params_size` to validate the exact payload layout
84/// before dereferencing `params` (a `u64` user pointer to the command-specific struct).
85#[repr(C)]
86struct NvRmControlParams {
87    h_client: u32,
88    h_object: u32,
89    cmd: u32,
90    flags: u32,
91    params: u64,
92    params_size: u32,
93    status: u32,
94}

96/// The `DT_*` `.dynamic` tags the GOT patch needs to locate a library's symbol, string, and
97/// relocation tables — spelled out here because libc exposes `PT_DYNAMIC` but not these individual
98/// values. `DT_NULL` (0) terminates the array.
99const DT_NULL: i64 = 0;
100/// ELF `.dynamic` tag `DT_PLTRELSZ` (2): total byte size of the PLT relocation table.
101const DT_PLTRELSZ: i64 = 2;
102/// ELF `.dynamic` tag `DT_RELA` (7): address of the general relocation table (`.rela.dyn`).
103const DT_RELA: i64 = 7;
104/// ELF `.dynamic` tag `DT_RELASZ` (8): total byte size of the `.rela.dyn` relocation table.
105const DT_RELASZ: i64 = 8;
106/// ELF `.dynamic` tag `DT_STRTAB` (5): address of the dynamic string table (symbol names).
107const DT_STRTAB: i64 = 5;
108/// ELF `.dynamic` tag `DT_SYMTAB` (6): address of the dynamic symbol table.
109const DT_SYMTAB: i64 = 6;
110/// ELF `.dynamic` tag `DT_JMPREL` (23): address of the PLT relocation table (`.rela.plt`).
111const DT_JMPREL: i64 = 23;

113/// Why the filter also handles eager GOT binds: `R_X86_64_GLOB_DAT` (6) names a GOT slot
114/// bound eagerly, the form `-fno-plt` builds use — and NVIDIA libraries route `ioctl` through such a
115/// slot in `.rela.dyn`. Alongside `R_X86_64_JUMP_SLOT`, it names a GOT entry holding a function
116/// pointer the filter repoints.
117const R_X86_64_GLOB_DAT: u32 = 6;
118/// The classic lazily-bound PLT GOT slot the filter repoints: `R_X86_64_JUMP_SLOT` (7), found
119/// in `.rela.plt`, names a GOT entry holding a function pointer — the common case, paired with
120/// `R_X86_64_GLOB_DAT` for the `-fno-plt` case.
121const R_X86_64_JUMP_SLOT: u32 = 7;

123/// Identity of the RM control device, cached so the hook only rewrites responses that truly
124/// came from it: the `rdev` of `/dev/nvidiactl`, resolved once at `install()`. A value of 0 means it
125/// was never resolved, in which case the fd-identity gate is skipped and matching falls back to the
126/// request/size checks alone.
127static NVIDIACTL_RDEV: AtomicU64 = AtomicU64::new(0);

129/// One `Elf64_Dyn` entry, modeled so the GOT patch can iterate a library's `.dynamic` array:
130/// a `d_tag` (`DT_*`) paired with `d_un`, the 64-bit `d_val`/`d_ptr` union interpreted per tag.
131#[repr(C)]
132struct Elf64Dyn {
133    d_tag: i64,
134    d_un: u64,
135}

137/// One `Elf64_Rela` relocation entry, modeled so the patch can find each GOT slot and the
138/// symbol it binds: `r_offset` (the target GOT slot, base-relative), `r_info` (packed symbol index +
139/// relocation type, split by `elf64_r_sym`/`elf64_r_type`), and the unused `r_addend`.
140#[repr(C)]
141struct Elf64Rela {
142    r_offset: u64,
143    r_info: u64,
144    r_addend: i64,
145}

147/// Extract the symbol-table index from an `Elf64_Rela` `r_info` field (its high 32 bits) —
148/// how the patch learns which symbol a relocation binds, so it can check for `ioctl`.
149#[inline]
150fn elf64_r_sym(info: u64) -> u64 {
151    info >> 32
152}
153
154/// Extract the relocation type from an `Elf64_Rela` `r_info` field (its low 32 bits) — how
155/// the patch tells a `JUMP_SLOT`/`GLOB_DAT` GOT entry from relocations it must ignore.
156#[inline]
157fn elf64_r_type(info: u64) -> u32 {
158    (info & 0xffff_ffff) as u32
159}

161/// Extract the `_IOC_NR` command byte (bits 0-7) of an ioctl request — the field that names
162/// the command. Used only by documentation and the `ioc_nr_extracts_low_byte` unit test, hence
163/// `#[allow(dead_code)]`.
164#[allow(dead_code)]
165#[inline]
166fn ioc_nr(req: c_ulong) -> c_ulong {
167    req & 0xFF
168}
169
170/// Strip the `_IOC_SIZE` field from a request (leaving DIR|TYPE|NR) so the gate stays bound
171/// to the exact RM control command without coupling to any one driver's param-struct size — a driver
172/// that changes `sizeof(NVOS54_PARAMETERS)` must still match.
173#[inline]
174fn ioc_no_size(req: c_ulong) -> c_ulong {
175    req & !(IOC_SIZEMASK << IOC_SIZESHIFT)
176}

178/// The reachability test the whole filter turns on: a GPU id is kept only if its
179/// `/dev/nvidia{minor}` node is actually present in the container, checked here via `access(F_OK)` on
180/// a NUL-terminated path.
181fn node_present(minor: u32) -> bool {
182    let path = format!("/dev/nvidia{}\0", minor);
183    unsafe { libc::access(path.as_ptr() as *const c_char, libc::F_OK) == 0 }
184}

186/// Map an RM `gpuId` to the `/dev/nvidia` minor that `node_present` needs: the id only
187/// carries a PCI address, not a device minor, so this bridges the two by scanning
188/// `/proc/driver/nvidia/gpus`, returning -1 when no match is found.
189///
190/// 1. **Extract the PCI address from the id**: the PCI address is encoded in `gpuId >> 8`, so
191///    `want_bus` is the low bus byte `(gpuId >> 8) & 0xFF` and `want_full` is the full
192///    `gpuId >> 8` (domain folded in for larger ids).
193/// 2. **Match a /proc entry**: each subdirectory is named by its PCI address
194///    `domain:bus:slot.func` in hex; split on `:` and `.` (expecting four fields) and parse the
195///    domain and bus. An entry matches when its bus equals `want_bus` **or** its combined
196///    `(domain << 8) | bus` equals `want_full` — the dual test covers ids that fold the domain in.
197/// 3. **Read the minor**: from the matched entry's `information` file, parse the `Device Minor: N`
198///    line and return `N`. The scan processes only the first matching directory (it `break`s
199///    afterward) and returns -1 if no minor line is present.
200fn gpuid_to_minor(gpu_id: u32) -> i32 {
201    let want_bus = (gpu_id >> 8) & 0xFF;
202    let want_full = gpu_id >> 8;
203    let dir = match std::fs::read_dir("/proc/driver/nvidia/gpus") {
204        Ok(d) => d,
205        Err(_) => return -1,
206    };
207    for ent in dir.flatten() {
208        let name = ent.file_name();
209        let name = match name.to_str() {
210            Some(s) => s,
211            None => continue,
212        };
213        let parts: Vec<&str> = name.split([':', '.']).collect();
214        if parts.len() != 4 {
215            continue;
216        }
217        let dom = u32::from_str_radix(parts[0], 16);
218        let bus = u32::from_str_radix(parts[1], 16);
219        let (dom, bus) = match (dom, bus) {
220            (Ok(d), Ok(b)) => (d, b),
221            _ => continue,
222        };
223        if bus != want_bus && ((dom << 8) | bus) != want_full {
224            continue;
225        }
226        let info = format!("/proc/driver/nvidia/gpus/{}/information", name);
227        if let Ok(text) = std::fs::read_to_string(&info) {
228            for line in text.lines() {
229                if let Some(rest) = line.strip_prefix("Device Minor:") {
230                    if let Ok(m) = rest.trim().parse::<i32>() {
231                        return m;
232                    }
233                }
234            }
235        }
236        break;
237    }
238    -1
239}

241/// Drop the hidden GPUs from a `gpuIds[]` array in place — but only ever a strict subset, so
242/// the filter can never accidentally blank out every GPU and strand the container with none. Ids
243/// that pass `keep` are compacted to the front and the rest back-filled with `INVALID_GPU_ID`.
244///
245/// 1. **Scan the populated prefix**: iterate until the first `INVALID_GPU_ID`, counting `total`
246///    live ids and collecting the kept ones into a scratch array (`nkept`).
247/// 2. **Commit only a strict subset** (`0 < nkept < total`): copy the kept prefix back over `ids`
248///    and set every trailing slot to `INVALID_GPU_ID`.
249/// 3. **Fail-safe**: if none survive or all survive, leave the array untouched — never blank out
250///    every GPU, and never do redundant work when nothing was dropped.
251///
252/// The `keep` predicate is injected so this stays free of `/proc` and `/dev` access and is
253/// unit-testable in isolation.
254fn filter_ids(ids: &mut [u32; MAX_ATTACHED_GPUS], keep: impl Fn(u32) -> bool) {
255    let mut kept = [0u32; MAX_ATTACHED_GPUS];
256    let (mut total, mut nkept) = (0usize, 0usize);
257    for &id in ids.iter() {
258        if id == INVALID_GPU_ID {
259            break;
260        }
261        total += 1;
262        if keep(id) {
263            kept[nkept] = id;
264            nkept += 1;
265        }
266    }
267    if nkept > 0 && nkept < total {
268        ids[..nkept].copy_from_slice(&kept[..nkept]);
269        for slot in ids[nkept..].iter_mut() {
270            *slot = INVALID_GPU_ID;
271        }
272    }
273}

275/// The seam where a hidden GPU gets scrubbed out: the `ioctl` wrapper installed into the
276/// NVIDIA libraries' GOT slots, which forwards to the real `libc::ioctl` and then post-processes the
277/// response. The three steps below exist only to make interposing on `ioctl` safe:
278///
279/// 1. **Call through**: invoke the genuine `libc::ioctl`. Only this crate's own GOT is left
280///    unpatched, so this inner call resolves to libc rather than back into this wrapper (no
281///    recursion).
282/// 2. **Filter the result** under `catch_unwind`: `rewrite_attached_ids` may drop hidden GPUs from
283///    the response. A panic must never unwind across this `extern "C"` boundary — the compiler
284///    guard would abort the whole process — so any panic is caught and the unmodified real result
285///    is returned instead.
286/// 3. **Preserve `errno`**: the real syscall's `errno` is saved before the `/proc` + `/dev` lookups
287///    inside the filter and restored afterward, so a caller reading `errno` sees the syscall's value.
288unsafe extern "C" fn filtered_ioctl(fd: c_int, req: c_ulong, arg: *mut c_void) -> c_int {
289    let rc = libc::ioctl(fd, req as _, arg);
290    let saved_errno = *libc::__errno_location();
291    let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
292        rewrite_attached_ids(fd, rc, req, arg);
293    }));
294    *libc::__errno_location() = saved_errno;
295    rc
296}

298/// Scrub the hidden GPUs out of a successful `GET_ATTACHED_IDS` / `GET_PROBED_IDS` response so
299/// the NVIDIA libraries never try to peer-init a GPU the container cannot reach: it drops every id
300/// whose `/dev/nvidia` node is absent. Split from `filtered_ioctl` so the whole thing runs under
301/// `catch_unwind`.
302///
303/// The guards run cheap-to-expensive and each returns early — leaving the response untouched — the
304/// moment one fails, so the overwhelmingly common non-enumeration ioctl pays almost nothing:
305///
306/// 1. **Request gate**: the syscall succeeded (`rc == 0`), `arg` is non-null, and the request is
307///    the RM_CONTROL ioctl matched by DIR|TYPE|NR (size-agnostic, via `ioc_no_size`).
308/// 2. **fd identity gate**: when `NVIDIACTL_RDEV` was resolved at install, `fstat` the fd and
309///    require a character device whose `rdev` equals `/dev/nvidiactl`'s — so only that char
310///    device's responses are rewritten. An `fstat` failure passes through untouched; if the rdev
311///    was never resolved the gate is skipped and matching relies on the request/size checks alone.
312/// 3. **Payload gate**: the RM call itself reports success (`ctrl.status == 0`) and carries a
313///    parameter pointer (`ctrl.params != 0`).
314/// 4. **Command + exact-size dispatch**: `cmd` selects ATTACHED or PROBED, and `params_size` must
315///    equal that command's exact struct size. Requiring the exact size both disambiguates the two
316///    (their layouts share a leading array) and stops a lying size from steering a rewrite under a
317///    different layout. Both APIs are filtered so behavior is uniform across driver versions —
318///    buggy 570-595 where it actually drops GPUs, and correct drivers where the strict-subset rule
319///    makes it a no-op.
320/// 5. **Filter the leading `gpuIds[32]`**: `filter_ids` keeps an id only when `gpuid_to_minor`
321///    resolves it and the corresponding `/dev/nvidia` node is present. For PROBED the trailing
322///    `excludedGpuIds[]` is deliberately left untouched (it lists GPUs to exclude, not to use).
323///
324/// When `PIXELFLUX_GPU_FILTER_DEBUG` is set, the before/after live-id counts are logged.
325unsafe fn rewrite_attached_ids(fd: c_int, rc: c_int, req: c_ulong, arg: *mut c_void) {
326    if rc != 0 || ioc_no_size(req) != ioc_no_size(NV_RM_CONTROL_REQUEST) || arg.is_null() {
327        return;
328    }
329    let cached = NVIDIACTL_RDEV.load(Ordering::Relaxed);
330    if cached != 0 {
331        let mut st: libc::stat = std::mem::zeroed();
332        if libc::fstat(fd, &mut st) != 0 {
333            return;
334        }
335        if (st.st_mode & libc::S_IFMT) != libc::S_IFCHR || st.st_rdev as u64 != cached {
336            return;
337        }
338    }
339    let ctrl = &mut *(arg as *mut NvRmControlParams);
340    if ctrl.status != 0 || ctrl.params == 0 {
341        return;
342    }
343    let which = match ctrl.cmd {
344        GPU_GET_ATTACHED_IDS if ctrl.params_size as usize == ATTACHED_PARAMS_SIZE => "ATTACHED",
345        GPU_GET_PROBED_IDS if ctrl.params_size as usize == PROBED_PARAMS_SIZE => "PROBED",
346        _ => return,
347    };
348    let ids = &mut *(ctrl.params as *mut [u32; MAX_ATTACHED_GPUS]);
349    let debug = std::env::var_os("PIXELFLUX_GPU_FILTER_DEBUG").is_some();
350    let before = ids.iter().take_while(|&&id| id != INVALID_GPU_ID).count();
351    filter_ids(ids, |id| {
352        let minor = gpuid_to_minor(id);
353        minor >= 0 && node_present(minor as u32)
354    });
355    if debug {
356        let after = ids.iter().take_while(|&&id| id != INVALID_GPU_ID).count();
357        eprintln!("[pixelflux] GET_{which}_IDS intercepted: {before} host GPU(s) -> {after} kept");
358    }
359}

361/// Reports the current `PROT_*` protection of the page holding `addr` (from
362/// `/proc/self/maps`, or -1 when the address is not found or maps is unreadable) for two reasons the
363/// GOT patch depends on: so it can refuse to dereference an address that a wrong relative/absolute
364/// guess landed in an unreadable page, and so it can restore a patched GOT page to its true original
365/// protection rather than a hardcoded read-only.
366///
367/// Each maps line begins `lo-hi perms ...`; the address range and the `rwxp` permission string are
368/// the first two whitespace fields. For the line whose `[lo, hi)` range contains `addr`, the `r`,
369/// `w`, and `x` characters are translated into `PROT_READ`/`PROT_WRITE`/`PROT_EXEC`.
370fn page_prot(addr: usize) -> i32 {
371    let text = match std::fs::read_to_string("/proc/self/maps") {
372        Ok(t) => t,
373        Err(_) => return -1,
374    };
375    for line in text.lines() {
376        let mut it = line.split_whitespace();
377        let range = match it.next() {
378            Some(r) => r,
379            None => continue,
380        };
381        let perms = match it.next() {
382            Some(p) => p,
383            None => continue,
384        };
385        let mut rr = range.split('-');
386        let lo = rr.next().and_then(|s| usize::from_str_radix(s, 16).ok());
387        let hi = rr.next().and_then(|s| usize::from_str_radix(s, 16).ok());
388        if let (Some(lo), Some(hi)) = (lo, hi) {
389            if addr >= lo && addr < hi {
390                let b = perms.as_bytes();
391                let mut prot = 0;
392                if b.first() == Some(&b'r') {
393                    prot |= libc::PROT_READ;
394                }
395                if b.get(1) == Some(&b'w') {
396                    prot |= libc::PROT_WRITE;
397                }
398                if b.get(2) == Some(&b'x') {
399                    prot |= libc::PROT_EXEC;
400                }
401                return prot;
402            }
403        }
404    }
405    -1
406}

408/// Turn a `DT_*` `d_ptr` into an absolute address across libc implementations that disagree
409/// on what it holds: glibc pre-relocates these pointers to absolute, while musl leaves them
410/// file-relative, so the patch cannot assume either. The heuristic treats a value below the load
411/// `base` as a relative offset to add to `base`, and a value at or above `base` as already absolute.
412#[inline]
413fn dyn_addr(base: usize, v: u64) -> usize {
414    let v = v as usize;
415    if v < base {
416        base + v
417    } else {
418        v
419    }
420}

422/// Repoint every `ioctl` GOT slot in one loaded library to `filtered_ioctl` by walking its
423/// `PT_DYNAMIC` array — the only way to interpose without an LD_PRELOAD object — and it must cover
424/// both the lazy-PLT and `-fno-plt` relocation forms because NVIDIA ships the latter.
425///
426/// 1. **Parse `.dynamic`**: iterate the `Elf64Dyn` entries until `DT_NULL`, recording the dynamic
427///    symbol table, string table, PLT relocation table (`DT_JMPREL`/`DT_PLTRELSZ`) and general
428///    relocation table (`DT_RELA`/`DT_RELASZ`). Table addresses are resolved through `dyn_addr` to
429///    handle the glibc-absolute vs musl-relative pointer conventions.
430/// 2. **Sanity-gate the tables**: bail if the symbol or string table is missing, and — via the
431///    `readable` closure over `page_prot` — bail if a table's mapping is known but lacks read
432///    permission (an address whose mapping cannot be determined is presumed readable), so a wrong
433///    relative/absolute guess from `dyn_addr` cannot fault on first dereference. Also bail if the
434///    page size is unavailable.
435/// 3. **Patch both relocation tables**: hand each present, adequately-sized, readable table to
436///    `patch_reloc_table`. The PLT table (`.rela.plt`) carries the classic lazily-bound `JUMP_SLOT`
437///    entries; the general table (`.rela.dyn`) carries the `GLOB_DAT` slot that `-fno-plt` NVIDIA
438///    builds use to bind `ioctl`.
439unsafe fn patch_ioctl_got(base: usize, dynp: *const Elf64Dyn) {
440    let mut symtab: *const libc::Elf64_Sym = std::ptr::null();
441    let mut strtab: *const c_char = std::ptr::null();
442    let mut jmprel: *const Elf64Rela = std::ptr::null();
443    let mut pltrelsz: usize = 0;
444    let mut rela: *const Elf64Rela = std::ptr::null();
445    let mut relasz: usize = 0;
446
447    let mut d = dynp;
448    while (*d).d_tag != DT_NULL {
449        match (*d).d_tag {
450            DT_SYMTAB => symtab = dyn_addr(base, (*d).d_un) as *const libc::Elf64_Sym,
451            DT_STRTAB => strtab = dyn_addr(base, (*d).d_un) as *const c_char,
452            DT_JMPREL => jmprel = dyn_addr(base, (*d).d_un) as *const Elf64Rela,
453            DT_PLTRELSZ => pltrelsz = (*d).d_un as usize,
454            DT_RELA => rela = dyn_addr(base, (*d).d_un) as *const Elf64Rela,
455            DT_RELASZ => relasz = (*d).d_un as usize,
456            _ => {}
457        }
458        d = d.add(1);
459    }
460    if symtab.is_null() || strtab.is_null() {
461        return;
462    }
463    let readable = |p: usize| {
464        let prot = page_prot(p);
465        prot < 0 || (prot & libc::PROT_READ) != 0
466    };
467    if !readable(symtab as usize) || !readable(strtab as usize) {
468        return;
469    }
470
471    let page = unsafe { libc::sysconf(libc::_SC_PAGESIZE) } as c_long;
472    if page <= 0 {
473        return;
474    }
475    let page = page as usize;
476    let ent = std::mem::size_of::<Elf64Rela>();
477    if !jmprel.is_null() && pltrelsz >= ent && readable(jmprel as usize) {
478        patch_reloc_table(base, symtab, strtab, jmprel, pltrelsz / ent, page);
479    }
480    if !rela.is_null() && relasz >= ent && readable(rela as usize) {
481        patch_reloc_table(base, symtab, strtab, rela, relasz / ent, page);
482    }
483}

485/// Repoint exactly the `ioctl` GOT slots in one relocation table to `filtered_ioctl` and
486/// nothing else — every candidate entry is name-checked so no other symbol the library imports is
487/// disturbed. It handles both the `JUMP_SLOT` and `GLOB_DAT` entry forms.
488///
489/// For each entry: skip it unless the relocation type is `R_X86_64_JUMP_SLOT` or
490/// `R_X86_64_GLOB_DAT` with a nonzero symbol index, then resolve the symbol name through
491/// `symtab`/`strtab` and skip unless it is exactly `ioctl`. For a match:
492///
493/// 1. **Locate the slot**: the GOT entry lives at `base + r_offset`; `pg` is its page-aligned base.
494/// 2. **Make the page writable**: read the slot's current protection via `page_prot` (defaulting to
495///    read/write when maps is unreadable) and `mprotect` the page to `PROT_READ | PROT_WRITE`.
496/// 3. **Publish the new pointer atomically**: store `filtered_ioctl` into the slot with an
497///    `AtomicPtr` `Release` store, since another thread may be dispatching through this GOT slot
498///    concurrently and the write must not tear.
499/// 4. **Restore protection**: return the page to its original protection rather than a hardcoded
500///    read-only — under partial RELRO these libraries lazily bind through this page, so leaving it
501///    read-only would fault the next symbol resolve.
502unsafe fn patch_reloc_table(
503    base: usize,
504    symtab: *const libc::Elf64_Sym,
505    strtab: *const c_char,
506    rela: *const Elf64Rela,
507    count: usize,
508    page: usize,
509) {
510    for i in 0..count {
511        let r = rela.add(i);
512        let rtype = elf64_r_type((*r).r_info);
513        if rtype != R_X86_64_JUMP_SLOT && rtype != R_X86_64_GLOB_DAT {
514            continue;
515        }
516        let sym_idx = elf64_r_sym((*r).r_info) as usize;
517        if sym_idx == 0 {
518            continue;
519        }
520        let name_off = (*symtab.add(sym_idx)).st_name as usize;
521        let name = CStr::from_ptr(strtab.add(name_off));
522        if name.to_bytes() != b"ioctl" {
523            continue;
524        }
525        let slot = (base + (*r).r_offset as usize) as *mut *mut c_void;
526        let pg = (slot as usize & !(page - 1)) as *mut c_void;
527        let mut orig = page_prot(slot as usize);
528        if orig < 0 {
529            orig = libc::PROT_READ | libc::PROT_WRITE;
530        }
531        if libc::mprotect(pg, page, libc::PROT_READ | libc::PROT_WRITE) == 0 {
532            let ap = &*(slot as *const AtomicPtr<c_void>);
533            ap.store(filtered_ioctl as *mut c_void, Ordering::Release);
534            libc::mprotect(pg, page, orig);
535        }
536    }
537}

539/// `dl_iterate_phdr` callback: for each loaded object matching a targeted NVIDIA library,
540/// find its `PT_DYNAMIC` segment and hand it to `patch_ioctl_got`. Returns 0 to keep iterating.
541///
542/// 1. **Panic firewall**: the whole body runs under `catch_unwind` because a panic must not unwind
543///    across this `extern "C"` boundary into the C `dl_iterate_phdr` — that would abort the
544///    process; a caught panic just ends this object's processing and iteration continues.
545/// 2. **Tight library match**: only `libnvcuvid`, `libnvidia-encode`, and `libcuda.so` are patched
546///    — the libraries that issue the enumeration ioctl (`libnvidia-encode` calls through
547///    `libnvcuvid`). Unrelated modules (`libcudart`, `libnvidia-ml`, `libnvidia-glcore`, …) and
548///    objects with no name are skipped so their `ioctl` bindings are left untouched.
549/// 3. **Patch each `PT_DYNAMIC`**: for a matched object, walk its program headers and call
550///    `patch_ioctl_got` on every dynamic segment, using the object's load address as the base.
551unsafe extern "C" fn patch_phdr_cb(
552    info: *mut libc::dl_phdr_info,
553    _size: libc::size_t,
554    _data: *mut c_void,
555) -> c_int {
556    let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
557        let info = &*info;
558        if info.dlpi_name.is_null() || *info.dlpi_name == 0 {
559            return;
560        }
561        let name = CStr::from_ptr(info.dlpi_name).to_string_lossy();
562        if !name.contains("libnvcuvid")
563            && !name.contains("libnvidia-encode")
564            && !name.contains("libcuda.so")
565        {
566            return;
567        }
568        let base = info.dlpi_addr as usize;
569        for i in 0..info.dlpi_phnum as isize {
570            let ph = &*info.dlpi_phdr.offset(i);
571            if ph.p_type == libc::PT_DYNAMIC {
572                patch_ioctl_got(base, (base + ph.p_vaddr as usize) as *const Elf64Dyn);
573            }
574        }
575    }));
576    0
577}

579/// True when at least one host GPU is hidden from the container — the only situation that
580/// can trigger the peer-init bug, so it gates whether the filter installs at all.
581///
582/// Compares two counts: `host` is the number of GPUs the kernel driver knows, from the non-dot
583/// entries of `/proc/driver/nvidia/gpus`; `visible` is the number of `/dev/nvidia{0..31}` nodes
584/// actually present in the container. GPUs are hidden when `host > visible`, and the `visible > 0`
585/// clause ensures there is still a usable GPU (a container with no GPUs at all is not this case).
586fn has_hidden_gpus() -> bool {
587    let host = std::fs::read_dir("/proc/driver/nvidia/gpus")
588        .map(|d| d.flatten().filter(|e| !e.file_name().to_string_lossy().starts_with('.')).count())
589        .unwrap_or(0);
590    let visible = (0..MAX_ATTACHED_GPUS as u32).filter(|&m| node_present(m)).count();
591    host > visible && visible > 0
592}

594/// Install the `GET_ATTACHED_IDS`/`GET_PROBED_IDS` GOT filter, at most once and only when a
595/// host GPU is hidden from the container. Idempotent and safe to call before every NVENC session
596/// open (guarded by a `Once`).
597///
598/// 1. **Escape hatch**: if `PIXELFLUX_DISABLE_GPU_FILTER` is set, log and return without patching
599///    (also useful for A/B testing the filter).
600/// 2. **Cache the nvidiactl identity**: `stat` `/dev/nvidiactl` and store its `rdev` in
601///    `NVIDIACTL_RDEV` so the ioctl hook can later verify an fd's identity before rewriting. If the
602///    `stat` fails the value stays 0, which makes the hook skip that identity gate.
603/// 3. **Patch only when needed**: if `has_hidden_gpus()` reports GPUs hidden from the container,
604///    run `dl_iterate_phdr` over `patch_phdr_cb` to repoint the `ioctl` GOT slots in the targeted
605///    NVIDIA libraries. Otherwise nothing is patched — the filter is a strict no-op.
606pub fn install() {
607    static ONCE: Once = Once::new();
608    ONCE.call_once(|| {
609        if std::env::var_os("PIXELFLUX_DISABLE_GPU_FILTER").is_some() {
610            eprintln!("[pixelflux] multi-GPU NVENC filter disabled via PIXELFLUX_DISABLE_GPU_FILTER");
611            return;
612        }
613        unsafe {
614            let mut st: libc::stat = std::mem::zeroed();
615            if libc::stat(c"/dev/nvidiactl".as_ptr(), &mut st) == 0 {
616                NVIDIACTL_RDEV.store(st.st_rdev as u64, Ordering::Relaxed);
617            }
618        }
619        if has_hidden_gpus() {
620            unsafe {
621                libc::dl_iterate_phdr(Some(patch_phdr_cb), std::ptr::null_mut());
622            }
623            eprintln!("[pixelflux] multi-GPU NVENC ioctl filter installed (GET_ATTACHED_IDS/GET_PROBED_IDS)");
624        }
625    });
626}

682    /// DIR|TYPE|NR but encodes a different `_IOC_SIZE` (here 0x30, versus the canonical 0x20 =
