7//! Out-of-band H.264 recording sink. Its reason to exist is separation: a
8//! recorder needs the exact encoded stream the viewers see, but neither side
9//! should be able to disturb the other — so this fans the encoder's raw Annex-B
10//! elementary stream to clients on a private Unix domain socket, entirely apart
11//! from the live viewer transport (WebSocket / WebRTC). A recorder attaching,
12//! stalling, or detaching cannot perturb what viewers receive, and recording
13//! works even with no viewer connected at all. Recording is also strictly
14//! optional: with no socket path configured, `RecordingSink::try_bind` returns
15//! `None` and there is no sink for the caller to write frames to.
16
17use std::fs;
18use std::io::{ErrorKind, Write};
19use std::os::unix::net::UnixListener;
20use std::sync::atomic::{AtomicBool, AtomicU32, Ordering};
21use std::sync::{Arc, Mutex};
22use std::thread;
23use std::time::Duration;
24
25use crossbeam_channel::{bounded, Sender, TrySendError};

27/// Bounds how long a single `write` to a recorder's socket may block, so a
28/// stalled reader cannot wedge its writer thread indefinitely. Without it, a
29/// recorder that stops draining would leave `write` blocked forever, and the
30/// writer could never again check its `stop` flag or make progress; the timeout
31/// instead surfaces as a soft `TimedOut`/`WouldBlock` that `write_all_frame`
32/// retries, keeping the thread responsive to teardown between attempts while a
33/// merely-slow-but-healthy reader still gets its frames.
34const WRITE_TIMEOUT: Duration = Duration::from_millis(100);
35
36/// Throttles the accept thread's poll loop, trading a little connection
37/// pickup latency for not burning a CPU on empty polls. The listener is
38/// deliberately nonblocking so the thread can observe `shutdown` between attempts
39/// instead of parking forever inside `accept()`; the price of that choice is that
40/// a bare `accept()` returns at once when nothing is pending, so the thread sleeps
41/// this long before trying again. The same interval also caps how long teardown
42/// waits for the thread to notice `shutdown` and exit.
43const ACCEPT_POLL_INTERVAL: Duration = Duration::from_millis(50);
44
45/// Bounds how long the recorded stream can run without a decode entry
46/// point — the reason a periodic keyframe cadence exists at all. An infinite GOP
47/// would leave a saved file decodable only from its very first frame and
48/// unrecoverable after any gap, so `should_force_idr` requests a fresh IDR every
49/// this-many frames to keep the recording seekable and self-healing. A newly
50/// connected client short-circuits the same counter to force one immediately (see
51/// `bind`), so a mid-stream joiner need not wait out the cadence.
52const DEFAULT_KEYINT_FRAMES: u32 = 60;
53
54/// Bounds each recorder's backlog so one slow reader can neither grow
55/// memory without limit nor push back on the shared encode thread. Fan-out is a
56/// non-blocking `try_send` into this per-client queue; a reader that falls this
57/// far behind is judged unable to keep up and dropped, sacrificing that one
58/// recording to protect the live pipeline and every other client. The depth is
59/// the jitter tolerance — deep enough to ride out a brief hiccup, shallow enough
60/// to cap the memory a doomed client can waste.
61const CLIENT_QUEUE_CAP: usize = 256;
62
63/// The sink's handle to one connected recorder — deliberately just the
64/// feed end and a kill switch, never the socket itself. Giving each client its
65/// own bounded channel and a dedicated writer thread that alone owns the stream
66/// is what keeps one slow reader from stalling the fan-out or its peers; the sink
67/// therefore holds only what it needs, `tx` to enqueue frames and `stop` to tear
68/// the writer down.
69struct ClientHandle {
70    tx: Sender<Arc<Vec<u8>>>,
71    /// Signals the writer thread to exit promptly rather than draining
72    /// whatever frames are still queued, once the client is dropped from the
73    /// sink (or the sink itself shuts down).
74    stop: Arc<AtomicBool>,
75}
76
77/// Taps the encoder's H.264 elementary stream to recorders over a Unix
78/// socket, kept out of band from the live viewer transport so recording neither
79/// perturbs nor depends on what viewers see. That separation is the point: a
80/// recorder can attach, stall, or detach without touching the WebSocket / WebRTC
81/// path, and recording runs even with no viewer connected — which a tee off the
82/// viewer stream could not promise.
83///
84/// `bind` owns the listener and spawns the accept thread; each accepted
85/// connection gets its own bounded queue and writer thread (see `ClientHandle`)
86/// so one slow reader cannot stall frame delivery to the others or to the caller
87/// (the encode thread). `frames_since_idr` and `keyint_frames` drive
88/// `should_force_idr`, which the encoder folds into its own keyframe decision —
89/// forcing an IDR on every new connection and on a periodic cadence — so a
90/// recorder that joins mid-stream always lands on a decodable entry point.
91pub struct RecordingSink {
92    path: String,
93    clients: Arc<Mutex<Vec<ClientHandle>>>,
94    shutdown: Arc<AtomicBool>,
95    frames_since_idr: Arc<AtomicU32>,
96    keyint_frames: u32,
97}
98
99impl RecordingSink {
100    /// The best-effort entry point to recording: yields a shared sink only
101    /// when recording is both configured and successfully bound, and `None`
102    /// otherwise — because recording is an optional tap that must never block or
103    /// crash capture. An empty `settings_path` means the feature is simply off; a
104    /// `bind` failure (e.g. the path cannot be bound) is logged to stderr and
105    /// swallowed rather than propagated, so a broken sink degrades to "no
106    /// recording" instead of taking the pipeline down with it. Success is wrapped
107    /// in an `Arc` so it can be shared with whatever encoder(s) feed it.
108    pub fn try_bind(settings_path: &str) -> Option<Arc<Self>> {
109        if settings_path.is_empty() {
110            return None;
111        }
112        match Self::bind(settings_path.to_string()) {
113            Ok(sink) => Some(Arc::new(sink)),
114            Err(e) => {
115                eprintln!("[recording_sink] bind failed: {:?}", e);
116                None
117            }
118        }
119    }
120
121    /// Stand up the socket and the one long-lived thread that owns it —
122    /// together the entire listening half of the sink. A single dedicated accept
123    /// thread, rather than polling folded into the encode path, is what lets
124    /// recorders connect at any point during capture without the frame path ever
125    /// waiting on connection setup. The steps, and why each is here:
126    ///
127    /// 1. **Stale socket cleanup**: best-effort `remove_file` on `path` first,
128    ///    since `UnixListener::bind` fails if a socket file is already there
129    ///    (e.g. left over from a prior crashed run).
130    /// 2. **Nonblocking listener**: `accept()` never blocks the accept
131    ///    thread, so the thread can also check the `shutdown` flag promptly
132    ///    between connection attempts instead of sitting blocked in
133    ///    `accept()`.
134    /// 3. **Startup IDR arming**: `frames_since_idr` starts at `u32::MAX`, so
135    ///    the very first `should_force_idr()` call after bind is already due
136    ///    — regardless of whether any client has connected by the time the
137    ///    first frame is encoded.
138    /// 4. **Accept loop** (runs on the spawned thread until `shutdown`):
139    ///    - **New connection**: sets a `WRITE_TIMEOUT` write timeout on the
140    ///      stream, abandoning the connection if that fails (a bad fd is not
141    ///      worth retrying). A dedicated writer thread is spawned that pulls
142    ///      frames from a fresh bounded channel, writing each via
143    ///      `write_all_frame` and exiting (with a logged reason) on any
144    ///      error — soft errors are already retried inside
145    ///      `write_all_frame`, so an error here means a hard failure, and the
146    ///      loop also exits early if `stop` is set. The channel's sender and
147    ///      the `stop` flag become a `ClientHandle` pushed into the shared
148    ///      `clients` list, and `frames_since_idr` resets to `u32::MAX` so the
149    ///      encoder's next frame is forced to an IDR — the newly joined
150    ///      client's (and, since the stream is shared, every other client's)
151    ///      next frame is guaranteed decodable.
152    ///    - **`WouldBlock`**: no connection is pending; sleeps
153    ///      `ACCEPT_POLL_INTERVAL` before polling again.
154    ///    - **Other error**: logs and backs off 500ms so a persistent accept
155    ///      failure does not spin the thread hot.
156    fn bind(path: String) -> std::io::Result<Self> {
157        let _ = fs::remove_file(&path);
158
159        let listener = UnixListener::bind(&path)?;
160        listener.set_nonblocking(true)?;
161
162        let clients: Arc<Mutex<Vec<ClientHandle>>> = Arc::new(Mutex::new(Vec::new()));
163        let shutdown = Arc::new(AtomicBool::new(false));
164
165        let frames_since_idr = Arc::new(AtomicU32::new(u32::MAX));
166
167        let clients_acc = clients.clone();
168        let shutdown_acc = shutdown.clone();
169        let frames_since_idr_acc = frames_since_idr.clone();
170        let path_log = path.clone();
171
172        thread::spawn(move || {
173            eprintln!("[recording_sink] listening on {}", path_log);
174            while !shutdown_acc.load(Ordering::Relaxed) {
175                match listener.accept() {
176                    Ok((stream, _)) => {
177                        if let Err(e) = stream.set_write_timeout(Some(WRITE_TIMEOUT)) {
178                            eprintln!("[recording_sink] set_write_timeout failed: {:?}", e);
179                            continue;
180                        }
181
182                        let (tx, rx) = bounded::<Arc<Vec<u8>>>(CLIENT_QUEUE_CAP);
183                        let stop = Arc::new(AtomicBool::new(false));
184                        let stop_writer = stop.clone();
185                        thread::spawn(move || {
186                            let mut stream = stream;
187                            for buf in rx.iter() {
188                                if stop_writer.load(Ordering::Relaxed) {
189                                    break;
190                                }
191                                if let Err(e) = write_all_frame(&mut stream, &buf, &stop_writer) {
192                                    eprintln!(
193                                        "[recording_sink] writer thread exiting; write failed: {:?}",
194                                        e
195                                    );
196                                    break;
197                                }
198                            }
199                        });
200
201                        let mut guard = clients_acc.lock().unwrap();
202                        guard.push(ClientHandle { tx, stop });
203
204                        frames_since_idr_acc.store(u32::MAX, Ordering::Relaxed);
205                        eprintln!(
206                            "[recording_sink] client connected; total {}; requesting IDR",
207                            guard.len()
208                        );
209                    }
210                    Err(e) if e.kind() == ErrorKind::WouldBlock => {
211                        thread::sleep(ACCEPT_POLL_INTERVAL);
212                    }
213                    Err(e) => {
214                        eprintln!("[recording_sink] accept error: {:?}", e);
215                        thread::sleep(Duration::from_millis(500));
216                    }
217                }
218            }
219            eprintln!("[recording_sink] listener thread exiting");
220        });
221
222        Ok(Self {
223            path,
224            clients,
225            shutdown,
226            frames_since_idr,
227            keyint_frames: DEFAULT_KEYINT_FRAMES,
228        })
229    }
230
231    /// The sink's answer to "does the recorded stream need a keyframe
232    /// now?", which the encoder folds into its own keyframe decision — the single
233    /// hook through which recording bends the encoder's GOP to its needs (an
234    /// immediate entry point for a joiner, a periodic one for seekability) without
235    /// the encoder having to track recorders itself. The counter lives here
236    /// because only the sink knows when a client connected.
237    ///
238    /// `frames_since_idr` is atomically incremented on every call, and `idr_due`
239    /// is evaluated against the count *before* that increment. A due result resets
240    /// the counter to zero, so the following `keyint_frames` calls return `false`
241    /// again. Bind time and every new client connection force the counter to
242    /// `u32::MAX` (see `bind`), which makes it due on the very next call — that is
243    /// how a joining client is guaranteed to receive an IDR as its first frame.
244    pub fn should_force_idr(&self) -> bool {
245        let prev = self.frames_since_idr.fetch_add(1, Ordering::Relaxed);
246        if idr_due(prev, self.keyint_frames) {
247            self.frames_since_idr.store(0, Ordering::Relaxed);
248            true
249        } else {
250            false
251        }
252    }
253
254    /// Hand one encoded frame to every recorder without ever blocking the
255    /// caller — which matters because the caller is the real-time encode thread,
256    /// and a stall here would hitch the live stream for viewers, not just for
257    /// recorders. Every choice below serves that constraint: non-blocking sends, a
258    /// single shared copy, and dropping a client rather than waiting on it.
259    ///
260    /// `data` is a raw Annex-B H.264 byte slice (a full frame or access unit).
261    ///
262    /// 1. **Early-outs**: an empty `data` is ignored, and — the common case
263    ///    when the socket is configured but nothing has connected yet — an
264    ///    empty client list returns before paying for the per-frame heap
265    ///    allocation below.
266    /// 2. **One shared copy**: `data` is copied once into an `Arc<Vec<u8>>` so
267    ///    every client's queue holds a cheap clone of the same allocation
268    ///    instead of one copy per client.
269    /// 3. **Non-blocking fan-out**: each client's bounded channel gets a
270    ///    `try_send`, which never blocks the caller. A `Full` queue (the
271    ///    client's writer thread cannot keep up with the incoming rate) or a
272    ///    `Disconnected` sender (the writer thread has exited) marks that
273    ///    client index for removal; a healthy send is silently a no-op here.
274    /// 4. **Removal**: marked indices are removed in reverse order with
275    ///    `swap_remove` so earlier indices stay valid as later ones are
276    ///    removed. Each removed client's `stop` flag is set so its writer
277    ///    thread notices and exits promptly instead of draining up to
278    ///    `CLIENT_QUEUE_CAP` queued frames first.
279    pub fn write_frame(&self, data: &[u8]) {
280        if data.is_empty() {
281            return;
282        }
283
284        let mut clients = self.clients.lock().unwrap();
285        if clients.is_empty() {
286            return;
287        }
288
289        let buf = Arc::new(data.to_vec());
290
291        let mut to_remove: Vec<usize> = Vec::new();
292
293        for (idx, client) in clients.iter().enumerate() {
294            match client.tx.try_send(buf.clone()) {
295                Ok(()) => {}
296                Err(TrySendError::Full(_)) => {
297                    eprintln!("[recording_sink] dropping slow client (idx {})", idx);
298                    to_remove.push(idx);
299                }
300                Err(TrySendError::Disconnected(_)) => {
301                    eprintln!("[recording_sink] dropping disconnected client (idx {})", idx);
302                    to_remove.push(idx);
303                }
304            }
305        }
306
307        for idx in to_remove.into_iter().rev() {
308            let removed = clients.swap_remove(idx);
309            removed.stop.store(true, Ordering::Relaxed);
310        }
311    }
312}

314impl Drop for RecordingSink {
315    /// Tear the sink down promptly and completely — stop accepting, end
316    /// every writer thread, free the socket path — so no thread or bound node
317    /// outlives the sink. The subtlety it exists to handle is that a writer thread
318    /// can be stuck in either of two states, and each needs a different nudge to
319    /// exit; miss either and that thread leaks.
320    ///
321    /// `shutdown` governs only the accept thread's loop condition (it stops taking
322    /// connections and exits on its next poll); writer threads never look at it.
323    /// Each writer is instead released by two signals, one per possible state: its
324    /// `stop` flag is set first so a writer mid-retry inside `write_all_frame`
325    /// aborts rather than finishing a stalled write, then `clients.clear()` drops
326    /// every queue sender, disconnecting each channel to wake a writer parked idle
327    /// in `rx.iter()` waiting for its next frame. Clearing the list here also makes
328    /// teardown immediate rather than leaving the writers alive until the accept
329    /// thread happens to exit and release its own clone of the client list. The
330    /// socket file is removed last so a later bind at the same path does not
331    /// collide with a leftover node.
332    fn drop(&mut self) {
333        self.shutdown.store(true, Ordering::Relaxed);
334        if let Ok(mut clients) = self.clients.lock() {
335            for client in clients.iter() {
336                client.stop.store(true, Ordering::Relaxed);
337            }
338            clients.clear();
339        }
340        let _ = fs::remove_file(&self.path);
341    }
342}

344/// Get one whole frame onto a recorder's socket, or fail cleanly — never
345/// leave a partial frame behind, because a truncated Annex-B NAL corrupts the
346/// recorded stream from that point on. Since the socket carries a write timeout,
347/// even a healthy reader can absorb only part of a `write` at a time, so this
348/// resumes from the last written offset across soft timeouts instead of
349/// surrendering the connection mid-frame.
350///
351/// 1. **Why retry at all**: the stream has a write timeout set, so a
352///    slow-but-healthy reader can make `write` return `TimedOut`/`WouldBlock`
353///    after only some leading bytes of a frame land. Bailing out there would
354///    emit a truncated Annex-B NAL, so the loop instead retries the unwritten
355///    remainder until the whole frame lands or a hard error occurs.
356/// 2. **Soft errors** (`TimedOut`, `WouldBlock`, `Interrupted`): `written`
357///    already reflects any partial progress, so these just loop back and
358///    retry the remainder. The stream's write timeout naturally paces the
359///    `TimedOut` case; a small sleep is added for `WouldBlock` so a socket
360///    that returns it without actually blocking does not spin the retry loop
361///    hot.
362/// 3. **Bounding a permanently stuck client**: this function alone would
363///    retry forever against a dead peer. That is bounded elsewhere — the
364///    client's bounded queue fills up, `write_frame` drops it from the client
365///    list and sets `stop`, which this loop checks at the top of every
366///    iteration, aborting with an error as soon as it is set.
367fn write_all_frame<W: Write>(stream: &mut W, buf: &[u8], stop: &AtomicBool) -> std::io::Result<()> {
368    let mut written = 0usize;
369    while written < buf.len() {
370        if stop.load(Ordering::Relaxed) {
371            return Err(std::io::Error::other(
372                "writer stopped (client dropped)",
373            ));
374        }
375        match stream.write(&buf[written..]) {
376            Ok(0) => {
377                return Err(std::io::Error::new(
378                    ErrorKind::WriteZero,
379                    "failed to write whole frame",
380                ));
381            }
382            Ok(n) => written += n,
383            Err(ref e) if e.kind() == ErrorKind::TimedOut => {}
384            Err(ref e) if e.kind() == ErrorKind::WouldBlock => {
385                thread::sleep(Duration::from_millis(5));
386            }
387            Err(ref e) if e.kind() == ErrorKind::Interrupted => {}
388            Err(e) => return Err(e),
389        }
390    }
391    Ok(())
392}

394/// The pure cadence arithmetic behind `should_force_idr`, pulled out so
395/// its off-by-one and its degenerate-cadence guard can be reasoned about and
396/// unit-tested free of the atomics and threading that surround the real counter.
397///
398/// Due when `frames_since_idr >= keyint_frames - 1`, computed with
399/// `saturating_sub` so a `keyint_frames` of 0 cannot underflow and instead safely
400/// makes every call due. That `- 1` is what makes a cadence of `keyint_frames`
401/// fire on the `keyint_frames`-th call after `should_force_idr` last reset the
402/// counter to zero.
403fn idr_due(frames_since_idr: u32, keyint_frames: u32) -> bool {
404    frames_since_idr >= keyint_frames.saturating_sub(1)
405}

407#[cfg(test)]
408mod tests {
409    use super::idr_due;
410
411    /// The very first call after construction is always due:
412    /// `frames_since_idr` starts at `u32::MAX`, which is what lets a client
413    /// that joins before any frame has been encoded still receive an IDR as
414    /// its first frame.
415    #[test]
416    fn first_frame_forces_idr() {
417        assert!(idr_due(u32::MAX, 60));
418    }
419
420    /// A cadence of 60 is not due until the 59th `frames_since_idr`
421    /// value (the 60th call after a reset) and stays due afterward, matching
422    /// the reset-on-due bookkeeping in `should_force_idr`.
423    #[test]
424    fn cadence_matches_keyint() {
425        let keyint = 60;
426        assert!(!idr_due(0, keyint));
427        assert!(!idr_due(58, keyint));
428        assert!(idr_due(59, keyint));
429        assert!(idr_due(60, keyint));
430    }
431
432    /// A `keyint_frames` of 0 or 1 cannot underflow the saturating
433    /// subtraction and instead makes every call due, at any frame count.
434    #[test]
435    fn keyint_zero_and_one_do_not_underflow() {
436        assert!(idr_due(0, 0));
437        assert!(idr_due(u32::MAX, 0));
438        assert!(idr_due(0, 1));
439        assert!(idr_due(1, 1));
440    }
